int ShardKvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
    // lab_shard: fill in your code

  RaftServer& server = GetRaftServer();

  shardGetEvent = Reactor::CreateSpEvent<IntEvent>();
  shared_ptr<MultiStringMarshallable> s =


      make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);

  auto cmdptr_m = dynamic_pointer_cast<Marshallable>(s);

  if (server.serverState != LEADER) {
    return KV_NOTLEADER;
  }
  is_leader = server.Start(cmdptr_m, &index, &term);
  shardGetEvent->Wait(1000000);
  if (shardGetEvent->status_ == Event::TIMEOUT) {
    return KV_TIMEOUT;
  } else if (shardGetEvent->get() == 1) {
    *v = kv_store_[k];
    Log_info("{GET SUCCESS} SERVERID = %d key = %s value = %d", server.loc_id_,
             k.c_str(), v->c_str());

    return KV_SUCCESS;
  }
  Log_info("{GET} SERVERID = %d oid = %d key = %s", server.loc_id_, oid,
           k.c_str());

  return KV_TIMEOUT;

}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code

     Coroutine::CreateRun([this, &m]() {
    RaftServer& server = GetRaftServer();
    server.mtx_.lock();
    auto v = (MultiStringMarshallable*)(&m);
    // if (v->data_.size() == 0){
    //   Log_info("Receiveing SnapShot");
    //   kv_store_ = v->kvStore;
    //   server.mtx_.unlock();
    //   return;
    // }
    if (v->data_[0] == "SHARD_MASTER"){
      Log_info("{SHARDKV}");
      return;
    }

    uint64_t oid = stoul(v->data_[0]);
    string operation = v->data_[1];

    if (operation == "put") {
      string key = v->data_[2];
      string value = v->data_[3];
      kv_store_[key] = value;
      Log_info(
          "{PUT OnNextCommand} SERVERID = %d key = %s value = %s is_leader = "
          "%d",
          server.loc_id_, key.c_str(), kv_store_[key].c_str(), is_leader);

      if (server.serverState == LEADER) {
        server.mtx_.unlock();
        shardPutEvent->Set(1);
      }
    } else if (operation == "append") {
      string key = v->data_[2];
      string value = v->data_[3];
      if (kv_store_.find(key) == kv_store_.end()) {
        kv_store_[key] = value;
        Log_info(
            "{APPEND KEY NOT FOUND} SERVERID = %d key = %s value = %s "
            "is_leader = %d",
            server.loc_id_, key.c_str(), kv_store_[key].c_str(), is_leader);
      } else {
        kv_store_[key] += value;
        Log_info(
            "{APPEND KEY FOUND} SERVERID = %d key = %s value = %s is_leader = "
            "%d",
            server.loc_id_, key.c_str(), kv_store_[key].c_str(), is_leader);
      }
      if (server.serverState == LEADER) {
        server.mtx_.unlock();
        shardAppendEvent->Set(1);
      }
    } else if (operation == "get") {
      if (server.serverState == LEADER) {
        server.mtx_.unlock();
        shardGetEvent->Set(1);
      }
    }

    server.mtx_.unlock();
  });

}
