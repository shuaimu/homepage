int ShardKvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
    // lab_shard: fill in your code
  // get configurations and check if the sid belongs this partition/groupid_t
  std::unique_lock<std::recursive_mutex> lck(mtx_);
  string op_id = to_string(oid);


  auto it1 = outstanding_requests_.find(op_id);
  if (it1 == outstanding_requests_.end()) {
    outstanding_requests_.insert({op_id, Reactor::CreateSpEvent<IntEvent>()});
  }

  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(op_id);
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  // TODO : should add current config number too
  shared_ptr<Marshallable> s1 = s; 
  RaftServer& raftSvr = this->GetRaftServer();
  uint64_t term, index;
  bool is_leader;

  raftSvr.GetState(&is_leader, &term);
  if (!is_leader) return KV_NOTLEADER;
  raftSvr.Start(s1, &index, &term);

  outstanding_requests_[op_id]->Wait(1000000); // 1 sec wait
  auto cond1 = outstanding_requests_[op_id]->status_ != Event::TIMEOUT;
  outstanding_requests_.erase(op_id);
  if (cond1) {
    return KV_SUCCESS;
  }
  return KV_TIMEOUT;
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
  auto v = (MultiStringMarshallable*)(&m);
  string op = v->data_[0];
  string op_type = v->data_[1];
  string k = v->data_[2];
  RaftServer& raftSvr = this->GetRaftServer();
  if (op_type == "put") {
    Log_info("Server with svr-id=%d Came to onNextCommand with marshallable opid=%s op_type=%s, key=%s, value=%s", raftSvr.loc_id_, op.c_str(), op_type.c_str(), v->data_[2].c_str(), v->data_[3].c_str());
    string value = v->data_[3];
    auto it = kv_store_.find(k);
    if (it != kv_store_.end()) {
      it->second = value;
    } else {
      kv_store_[k] = value;
    }
  } else if (op_type == "append") {
    Log_info("Server with svr-id=%d Came to onNextCommand with marshallable opid=%s op_type=%s, key=%s, value=%s", raftSvr.loc_id_, op.c_str(), op_type.c_str(), v->data_[2].c_str(), v->data_[3].c_str());
    string value = v->data_[3];
    auto it = kv_store_.find(k);
    if (it != kv_store_.end()) { 
      it->second = it->second + value;
    } else {
      kv_store_[k] = value;
    }
  } else if (op_type == "get") {
    Log_info("Server with svr-id=%d Came to onNextCommand with marshallable opid=%s op_type=%s, key=%s, value=%s", raftSvr.loc_id_, op.c_str(), op_type.c_str(), v->data_[2].c_str(), kv_store_.find(k)->second.c_str());
    string value = kv_store_[k];
    v->data_[3] = value;
  }
  // outstanding_requests_.push_back(op);
  auto it1 = outstanding_requests_.find(op);
  if (it1 != outstanding_requests_.end()) {
    outstanding_requests_[op]->Set(1);
  }
}
