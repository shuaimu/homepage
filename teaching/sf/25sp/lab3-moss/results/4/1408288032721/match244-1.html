int ShardKvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
    // lab_shard: fill in your code
     auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  shared_ptr<Marshallable> cmd = s;
  uint64_t index = 0;
  uint64_t term = 0;
  bool isLeader = 0;
  *v = "";
  this->GetRaftServer().GetState(&isLeader,&term);
  
  if(!isLeader)
  	return KV_NOTLEADER;
  //Log_info("The partition id of Request Get is %d",this->GetRaftServer().partition_id_);

  bool appended = this->GetRaftServer().Start(cmd,&index,&term);
  operationStatus[to_string(oid)] = 0; //Pending
  Coroutine::Sleep(900000);
  //Log_info("Timeout?%d",operationStatus[to_string(oid)]);
  if(operationStatus[to_string(oid)]!=1)
  {
      //Log_info("Timeout has occured in ShradKV");
      return KV_TIMEOUT;
  }
  if(operationStatus[to_string(oid)]==1 && mapValue.find(k)!=mapValue.end())
  {
  	//Log_info("get%s",k);
  	*v = mapValue[k];
  	return KV_SUCCESS;
  }
  
  return KV_SUCCESS;
}

    auto s = (MultiStringMarshallable*)(&m);
  string oid = s->data_[0];
  string operation = s->data_[1];
  string key = s->data_[2];
  operationStatus[oid] = 1; //Completed
  if(operation.compare("put")==0 || (operation.compare("append")==0 && mapValue.find(key)==mapValue.end()))
  {
  	string value = s->data_[3];
    //Log_info("I am in Put of shardKV");
