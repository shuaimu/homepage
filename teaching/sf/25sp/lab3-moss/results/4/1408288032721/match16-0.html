int ShardKvClient::Op(function<int(uint32_t*)> func) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      leader_idx_ = (leader_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      leader_idx_ = (leader_idx_+1) % 5;
    }
  }

}

int ShardKvClient::Put(const string& k, const string& v) {
  uint32_t shardId = Key2Shard(k);
  auto p = (ShardMasterProxy*)commo_->rpc_proxies_.at(0);
  ShardConfig configToServe;


  uint32_t retValue;
  p->Query(-1, &retValue,&configToServe);
  vector<uint32_t> servers = configToServe.group_servers_map_[configToServe.shard_group_map_[shardId]];
  Log_info("Size is : %d", servers.size());
  Log_info("group size is : %d", configToServe.shard_group_map_[shardId]);


  return Op([&](uint32_t* r)->int{
    return Proxy(servers[0]).Put(GetNextOpId(), k, v, r);
  });
}

ShardKvProxy& ShardKvClient::Proxy(siteid_t site_id) {
  verify(commo_);
  auto p = (ShardKvProxy*)commo_->rpc_proxies_.at(site_id);
  return *p; 
}
