int ShardKvClient::Op(function<int(uint32_t*)> func) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT)
    {
      leader_idx_ = (leader_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      leader_idx_ = (leader_idx_+1) % 5;
    }
  }

}



int ShardKvClient::Put(const string& k, const string& v) {
  shardid_t shardId = Key2Shard(k);
  auto sm=(ShardMasterProxy*)commo_->rpc_proxies_.at(0);
  ShardConfig config;
  uint32_t returnValue;
  sm->Query(-1, &returnValue, &config);
  uint32_t groupId = config.shard_group_map_.at(shardId);
  uint32_t serverId = config.group_servers_map_.at(groupId)[0];
  leader_idx_ = serverId;
  Log_info("Sending Put to %d", leader_idx_);
  return Op([&](uint32_t *r) -> int
    { return Proxy(leader_idx_).Put(GetNextOpId(), k, v, r); });
  }

ShardKvProxy& ShardKvClient::Proxy(siteid_t site_id) {
  verify(commo_);
  int i = 0;
  auto server = (ShardKvProxy*)commo_;
  int matched = 0;
  while (i < 4 && matched == 0)
  {
    auto proxies = commo_->rpc_par_proxies_[i];
    for (auto& p : proxies) {
      if (p.first == site_id)
      {
        server = (ShardKvProxy*) p.second;
        matched = 1;
        break;
      }
    }
    i += 1;
  }
  return *server;
}
