int ShardKvClient::Op(function<int(uint32_t*)> func) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      leader_idx_ = (leader_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      leader_idx_ = (leader_idx_+1) % 5;
    }
  }

}

std::shared_ptr<ShardMasterClient> CreateSmClient() {
  auto shardmaster_cli = make_shared<ShardMasterClient>();
  shardmaster_cli->commo_ = new Communicator();
  return shardmaster_cli;
}

int ShardKvClient::Put(const string& k, const string& v) {
  auto sm_cli = CreateSmClient();
  ShardConfig cfg;
  sm_cli->Query(-1, &cfg);
  auto shardId = Key2Shard(k) + 1;
  
  Log_info("config number %d", cfg.number);
  Log_info("gid for shard %d", cfg.shard_group_map_[shardId]);
  auto gid = cfg.shard_group_map_[shardId];
  vector<uint32_t> ss = cfg.group_servers_map_[gid];


  return Op([&](uint32_t* r)->int{
    return Proxy(ss[leader_idx_]).Put(GetNextOpId(), k, v, r);
  });
}

ShardKvProxy& ShardKvClient::Proxy(siteid_t site_id) {
  verify(commo_);
  auto p = (ShardKvProxy*)commo_->rpc_proxies_.at(site_id);
  return *p; 
}
