int ShardKvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
  // lab_shard: fill in your code
  Log_info("Entering put");
  auto evPut = Reactor::CreateSpEvent<IntEvent>();
  opidMap.insert({to_string(oid), evPut});
  uint64_t idx = 0;
  uint64_t term = 0;
  Log_info("put working for oid %llu", oid);

  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);

  auto sendCMD = dynamic_pointer_cast<Marshallable>(s);
  if (!(GetRaftServer().Start(sendCMD, &idx, &term))) {
    // Log_info("not the leader!");
    return KV_NOTLEADER;
  }
  evPut->Wait(900000);
  if (evPut->status_ == Event::TIMEOUT) {
    Log_info("PUT ShardKV | Returning timeout");
    return KV_TIMEOUT;
  }
  else {
    // opidMap.at(oid) = ev;
    Log_info("PUT ShardKv | Returning success");
    opidMap.erase(to_string(oid));


    return KV_SUCCESS;
  }
  return KV_TIMEOUT;
}

int ShardKvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
    // lab_shard: fill in your code
  auto evGet = Reactor::CreateSpEvent<IntEvent>();
  opidMap.insert({to_string(oid), evGet});
  uint64_t idx = 0;
  uint64_t term = 0;
  Log_info("Get working for oid %d", oid);
  Log_info("Len of hashmap %d", hashmap.size());
  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  s->data_.push_back("GET_DUMMY_VALUE");
  auto sendCMD = dynamic_pointer_cast<Marshallable>(s);
  Log_info("GET: value for given key %s", k.c_str());
  if (!(GetRaftServer().Start(sendCMD, &idx, &term))) {
    Log_info("not the leader!");
    return KV_NOTLEADER;
  }
  evGet->Wait(900000);
  if (evGet->status_ == Event::TIMEOUT) {
    Log_info("GET | returning Timeout");
    return KV_TIMEOUT;
  }
  else {
    Log_info("GET | Returning success");
    auto value = hashmap.find(k); 
    *v = value->second;
    opidMap.erase(to_string(oid));


    return KV_SUCCESS;
  }
  return KV_TIMEOUT;
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
  auto v = (MultiStringMarshallable*)(&m);
  string value = v->data_[3];
  string key = v->data_[2];
  string operation = v->data_[1];
  string opid = v->data_[0];
  Log_info("opid %s", opid.c_str());
  if (operation=="get") {
    Log_info("getting get");
    
  }
  if (operation=="put") {
    if (hashmap.find(key) != hashmap.end()) {
      hashmap[key.c_str()] = value.c_str();
    } else {
      hashmap.insert({key.c_str(),value.c_str()});
    }
  }
  if (operation == "append") {
    if (hashmap.find(key) != hashmap.end()) {
      hashmap.at(key.c_str()) += value.c_str();
    }
    else {
      hashmap.insert({key.c_str(),value.c_str()});
    }
  }auto output = hashmap.find(key);
  Log_info("Final value for Key %s: %s | Length of hashmap %d",(output->first).c_str(),(output->second).c_str(), hashmap.size());
  auto eventval = opidMap.find(opid.c_str());
  if (eventval == opidMap.end()) {
    Log_info("already complete");
  } else {
    Log_info("value!!!! %s", (eventval->first).c_str());
    auto ev = eventval->second;
    ev->Set(1);
  }
  
}
