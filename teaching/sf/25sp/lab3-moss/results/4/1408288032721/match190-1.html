  int ShardKvServer::CollectData(map<string, string> *data)
  {
    Log_debug("collect data at server %d", GetRaftServer().loc_id_);
    if (GetRaftServer().status != RaftServer::LEADER)
    {
      return KV_NOTLEADER;
    }
    for (auto x : kv_store_)
    {
      data->insert(x);
    }
    return KV_SUCCESS;
  }

  void ShardKvServer::OnNextCommand(Marshallable &m)
  {
    // lab_shard: fill in your code
    auto v = (MultiStringMarshallable *)(&m);
    /* your code here */
    // int size = v->data_.size();
    // for (int i = 0; i < size; i++)
    // {
    //   printf("info %d: %s \n", i, v->data_.at(i));
    // }
    // Log_debug("before getting log");
    std::unique_lock<std::mutex> lock(shard_kv_cv_m);
    // Log_debug("after getting log");

    if (v->data_.size() > 1)
    {
      Log_debug("inside on next command with data.at(1):%s", v->data_.at(1));
      if (v->data_.at(1) == "put")
      {
        put(v->data_.at(2), v->data_.at(3));
      }
      else if (v->data_.at(1) == "append")
      {
        append(v->data_.at(2), v->data_.at(3));
      }
      else if (v->data_.at(1) == "reconfig")
      {
        cur_config = v->sc;
        Log_debug("config updated to %d", cur_config.number);
      }
      else if (v->data_.at(1) == "update")
      {

        map<string, string> received_data = deserializeMap(v->data_.at(2));
        Log_debug("server %d starts updating, received data size: %d", GetRaftServer().loc_id_, received_data.size());
        kv_store_.erase(kv_store_.begin(), kv_store_.end());
        kv_store_.insert(received_data.begin(), received_data.end());
      }
    }
  }
