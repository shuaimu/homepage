int64_t ShardKvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;
  ret = ret << 32;
  ret = ret + op_id_cnt_++; 
  return ret;
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
    auto v = (MultiStringMarshallable*)(&m);
  auto commands = v->data_;
  string cmdType = commands[1];
  RaftServer& raftServer = GetRaftServer();
  (*processedQueries).insert(commands[0]);
  if(cmdType == "put") {
    Log_info("(OnNextCommand PUT) called for kvServer=%d type=APPEND", raftServer.site_id_);
    string key = commands[2];
    (kv_store_)[key] = commands[3];
  } else if (cmdType == "append") {
    Log_info("(OnNextCommand APPEND) called for kvServer=%d type=PUT", raftServer.site_id_);
    string key = commands[2];
   (kv_store_)[key] += commands[3];
  } else {
    Log_info("(OnNextCommand GET) called for kvServer=%d type=GET", raftServer.site_id_);
  }
}
