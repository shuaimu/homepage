  int64_t KvServer::GetNextOpId()
  {

    ret = ret + op_id_cnt_++;

  /**
   * @brief Put a key-value pair into the map.
   * 
   * @param oid The operation ID.
   * @param k The key.
   * @param v The value.
   * @return int KV_SUCCESS if the operation is successful, KV_NOTLEADER if the server is not the leader, KV_TIMEOUT if the operation times out.
   * 
   * @pre The key must not exist in the map.
   * @post The key-value pair will be added to the map.
   * 
   * @note The key-value pair will be added to the map.
  */
  int KvServer::Put(const uint64_t &oid,
                    const string &k,
                    const string &v)
  {
    /*
    Your are recommended to use MultiStringMarshallable as the format
    for the log entries. Here is an example how you can use it.
    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(to_string(op_id));
    s->data_.push_back("put");
    s->data_.push_back(k);
    s->data_.push_back(v);
    */
    /* your code here */
    RaftServer &rs = GetRaftServer();

    std::lock_guard<std::recursive_mutex> lock(rs.mtx_);

    std::string oid_str = to_string(oid);

    uint64_t index;
    uint64_t term;
    bool is_leader;



    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(oid_str);
    s->data_.push_back("put");
    s->data_.push_back(k);
    s->data_.push_back(v);

    std::shared_ptr<Marshallable> ret = s;
    is_leader = rs.Start(ret, &index, &term);
    if (is_leader)
    {
      auto startTime = std::chrono::high_resolution_clock::now();
      while (true)
      {
        // Check if the timeout has elapsed
        if ((std::chrono::high_resolution_clock::now() - startTime) > timeoutDuration)
        {
          // Handle the timeout scenario
          // Log_info("KvServer::Put -> Timeout {ID: %d, Value: %s}", oid, v.c_str());
          return KV_TIMEOUT;
        }

        if (getStore().count(oid_str) != 0)
        {
          getStore()[k] = v;
          // Log_info("KvServer::Put -> Success {Key: %s, Value: %s}", k.c_str(), v.c_str());
          return KV_SUCCESS;
        }
        // Sleep for a short interval to avoid busy-waiting
        Coroutine::Sleep(1000);
      }
    }
    // Log_info("KvServer::Put -> Not leader");
    return KV_NOTLEADER;
  }

  /**
   * @brief Append a value to the value of a key.
   *
   * @param oid The operation ID.
   * @param k The key of the operation.
   * @param v The value to append.
   * @return int KV_SUCCESS if the operation is successful, KV_NOTLEADER if the server is not the leader, KV_TIMEOUT if the operation times out.
   *
   * @pre The key must exist in the map. If it doesn't, it will be added.
   * @post The value of the key will be appended with the new value or the new value will be added to the map.
   *
   */
  int KvServer::Append(const uint64_t &oid,
                       const string &k,
                       const string &v)
  {
    /* your code here */
    RaftServer &rs = GetRaftServer();

    std::lock_guard<std::recursive_mutex> lock(rs.mtx_);

    std::string oid_str = to_string(oid);

    uint64_t index;
    uint64_t term;
    bool is_leader;



    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(oid_str);
    s->data_.push_back("append");
    s->data_.push_back(k);
    s->data_.push_back(v);

    std::shared_ptr<Marshallable> ret = s;
    is_leader = rs.Start(ret, &index, &term);
    if (is_leader)
    {
      auto startTime = std::chrono::high_resolution_clock::now();
      while (true)
      {
        // Check if the timeout has elapsed
        if ((std::chrono::high_resolution_clock::now() - startTime) > timeoutDuration)
        {
          // Handle the timeout scenario
          return KV_TIMEOUT;
        }
        if (getStore().count(oid_str) != 0)
        {
          if (getStore().count(k) == 0)
          {
            // Key doesn't exist, so add it to the map with the new value.
            getStore()[k] = v;
            // Log_info("KvServer::Append -> Doesn't exist before! Success {Key: %s, Value: %s}", k.c_str(), v.c_str());
          }
          else
          {
            // Key exists in the map, so append the new value to the existing value.
            getStore()[k] += v;
            // Log_info("KvServer::Append -> Exists before! Success {Key: %s, Value: %s}", k.c_str(), v.c_str());
          }
          return KV_SUCCESS;
        }
        // Sleep for a short interval to avoid busy-waiting
        Coroutine::Sleep(1000);
      }
    }
    // Log_info("KvServer::Append -> Not leader");
    return KV_NOTLEADER;
  }

  /**
   * @brief Get the value of a key.
   *
   * @param oid The operation ID.
   * @param k The key.
   * @param v A reference to the value.
   * @return int KV_SUCCESS if the operation is successful, KV_NOTLEADER if the server is not the leader, KV_TIMEOUT if the operation times out.
   *
   * @pre The key must exist in the map.
   * @post The value of the key will be returned.
   *
   */
  int KvServer::Get(const uint64_t &oid,
                    const string &k,
                    string *v)
  {
    /* your code here */
    RaftServer &rs = GetRaftServer();
    std::lock_guard<std::recursive_mutex> lock(rs.mtx_);

    std::string oid_str = to_string(oid);

    uint64_t index;
    uint64_t term;
    bool is_leader;

    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(oid_str);
    s->data_.push_back("get");
    s->data_.push_back(k);

    std::shared_ptr<Marshallable> ret = s;

    is_leader = rs.Start(ret, &index, &term);
    if (is_leader)
    {
      auto startTime = std::chrono::high_resolution_clock::now();
      while (true)
      {
        // Check if the timeout has elapsed
        if ((std::chrono::high_resolution_clock::now() - startTime) > timeoutDuration)
        {
          // Handle the timeout scenario
          // Log_info("KvServer::Get -> Timeout {ID: %s}", opID);
          return KV_TIMEOUT;
        }

        if (getStore().count(oid_str) != 0)
        {
          if (getStore().count(k) != 0)
          {
            // Key exists in the map, so retrieve its value.
            *v = getStore()[k];
            // Log_info("KvServer::Get -> Success {Key: %s, Value: %s}", k.c_str(), v->c_str());
            return KV_SUCCESS; // Return true to indicate success.
          }
          return KV_TIMEOUT;
        }
        // Sleep for a short interval to avoid busy-waiting
        Coroutine::Sleep(1000);
      }
    }
    // Log_info("KvServer::Get -> Not leader");
    return KV_NOTLEADER;
  }

  /**
   * @brief This function is called when a new command is committed by raft.
   *
   * @param m The command.
   *
   * @pre The command must be a Marshallable.
   * @post The command will be executed.
   *
   * @note Pranav Dani is acknowledged for providing me with the idea of storing committed  operation IDs in a map to increase the efficiency of the implementation.
   */
  void KvServer::OnNextCommand(Marshallable &m)
  {
    RaftServer &rs = GetRaftServer();
    auto v = static_cast<MultiStringMarshallable *>(&m);
    auto lambda = [this, &v, &rs]()
    {
      std::lock_guard<std::recursive_mutex> lock(rs.mtx_);
      std::string oid_str = v->data_[0];

      getStore()[oid_str] = true;
    };
    Coroutine::CreateRun(lambda);
  }

  shared_ptr<KvClient> KvServer::CreateClient()
  {
    /* don't change this function */

    cli->commo_ = commo_;

    id = id << 16;
    cli->cli_id_ = id + cli_cnt_++;
    return cli;
  }
