int64_t KvServer::GetNextOpId() {

  int64_t ret = sp_log_svr_->site_id_;

  bool isLeader = GetRaftServer().Start(cmdToSend, &index, &term);
  if (!isLeader) {
    return KV_NOTLEADER;
  }
  Log_debug("Request with operation id: %d started processing", oid);

  Timer *timer = new Timer();
  timer->start();
  while (timer->elapsed() <= PROCESSING_TIMEOUT && cmdResponse.find(oid) == cmdResponse.end())
  {
    Coroutine::Sleep(50000);
  }

  int status = KV_TIMEOUT;
  if (cmdResponse.find(oid) != cmdResponse.end()) {
    status = KV_SUCCESS;
    Log_debug("Request with operation id: %d succeeded", oid);
  } else {
    Log_debug("Request with operation id: %d timed out", oid);
  }
  cmdResponse.erase(oid);
  return status;
}

void KvServer::OnNextCommand(Marshallable& m) {
  /* your code here */
  auto cmd = (MultiStringMarshallable*)(&m);
  string operationId = cmd->data_.at(0);
  string operationType = cmd->data_.at(1);
  string key = cmd->data_.at(2);
  
  stringstream operationIdStream(operationId);
  uint64_t oid;
  operationIdStream >> oid;
  Log_debug("Request with operation id: %d to be applied", oid);

  if (operationType.compare("put") == 0) {
    string value = cmd->data_.at(3);
    kvStore[key] = value;
    cmdResponse[oid] = KV_EMPTY_RESPONSE;
    Log_debug("Put request with operation id: %d is applied", oid);
  } 
  
  else if (operationType.compare("append") == 0) {
    string value = cmd->data_.at(3);
    if (kvStore.find(key) == kvStore.end()) {
      kvStore[key] = value;
    } else {
      kvStore[key] += value;
    }
    cmdResponse[oid] = KV_EMPTY_RESPONSE;
    Log_debug("Append request with operation id: %d is applied", oid);
  } 
  
  else if (operationType.compare("get") == 0) {
    cmdResponse[oid] = kvStore[key];
    Log_debug("Get request with operation id: %d is applied", oid);
  }

  Log_debug("Request with operation id: %d is applied", oid);
}
