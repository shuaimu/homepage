int KvServer::Append(const uint64_t& oid, const string& k, const string& v) {
  /* your code here */
  RaftServer& server = GetRaftServer();

  appendEvent = Reactor::CreateSpEvent<IntEvent>();


  shared_ptr<MultiStringMarshallable> s =
      make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("append");
  s->data_.push_back(k);
  s->data_.push_back(v);

  auto cmdptr_m = dynamic_pointer_cast<Marshallable>(s);

  Log_info("{APPEND CALLED} SERVERID = %d SERVER_STATE = %d", server.loc_id_,
           server.serverState);

  if (server.serverState != LEADER) {
    return KV_NOTLEADER;
  }

  is_leader = server.Start(cmdptr_m, &index, &term);
  // Coroutine::Sleep(6000000);
  appendEvent->Wait(900000);
  if (appendEvent->status_ == Event::TIMEOUT) {
    if (server.serverState == LEADER) return KV_NOTLEADER;
    return KV_TIMEOUT;
  } else if (appendEvent->get() == 1) {
    Log_info("{APPEND SUCCESS} SERVERID = %d", server.loc_id_);
    return KV_SUCCESS;
  }
  return KV_TIMEOUT;

  Log_info("{APPEND} SERVERID = %d oid = %d key = %s value = %s",
           server.loc_id_, oid, k.c_str(), v.c_str());
}

int KvServer::Get(const uint64_t& oid, const string& k, string* v) {
  /* your code here */
  RaftServer& server = GetRaftServer();

  getEvent = Reactor::CreateSpEvent<IntEvent>();


  shared_ptr<MultiStringMarshallable> s =
      make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);

  auto cmdptr_m = dynamic_pointer_cast<Marshallable>(s);

  if (server.serverState != LEADER) {
    return KV_NOTLEADER;
  }
  is_leader = server.Start(cmdptr_m, &index, &term);
  getEvent->Wait(900000);
  if (getEvent->status_ == Event::TIMEOUT) {
    return KV_TIMEOUT;
  } else if (getEvent->get() == 1) {
    *v = kv[k];
    Log_info("{GET SUCCESS} SERVERID = %d key = %s value = %d", server.loc_id_,
             k.c_str(), v->c_str());

    return KV_SUCCESS;
  }
  Log_info("{GET} SERVERID = %d oid = %d key = %s", server.loc_id_, oid,
           k.c_str());

  return KV_TIMEOUT;
}

void KvServer::OnNextCommand(Marshallable& m) {
  /* your code here */
  Coroutine::CreateRun([this, &m]() {
    RaftServer& server = GetRaftServer();


    server.mtx_.lock();
    auto v = (MultiStringMarshallable*)(&m);
    uint64_t oid = stoul(v->data_[0]);
    string operation = v->data_[1];
    if (operation == "put") {
      string key = v->data_[2];
      string value = v->data_[3];
      kv[key] = value;
      Log_info(
          "{PUT OnNextCommand} SERVERID = %d key = %s value = %s is_leader = "
          "%d",
          server.loc_id_, key.c_str(), kv[key].c_str(), is_leader);

      if (server.serverState == LEADER) {
        server.mtx_.unlock();
        putEvent->Set(1);
      }
    } else if (operation == "append") {
      string key = v->data_[2];
      string value = v->data_[3];
      if (kv.find(key) == kv.end()) {
        kv[key] = value;
        Log_info(
            "{APPEND KEY NOT FOUND} SERVERID = %d key = %s value = %s "
            "is_leader = %d",
            server.loc_id_, key.c_str(), kv[key].c_str(), is_leader);
      } else {
        kv[key] += value;
        Log_info(
            "{APPEND KEY FOUND} SERVERID = %d key = %s value = %s is_leader = "
            "%d",
            server.loc_id_, key.c_str(), kv[key].c_str(), is_leader);
      }
      if (server.serverState == LEADER) {
        server.mtx_.unlock();
        appendEvent->Set(1);
      }
    } else if (operation == "get") {
      if (server.serverState == LEADER) {
        server.mtx_.unlock();
        getEvent->Set(1);
      }
    }
    server.mtx_.unlock();
  });
}
