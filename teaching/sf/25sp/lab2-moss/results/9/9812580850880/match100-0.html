int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  Log_info("KvServer::Inside Get, KvServer, k %s, v %s, oid %lu", k.c_str(), v, oid);
  RaftServer& raft = GetRaftServer(); 


  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  s->data_.push_back(*v);

  bool is_leader = false;
  uint64_t term = 0;
  uint64_t index = 0;

  (&raft)->GetState(&is_leader, &term);
  
  if (is_leader == false)
    return KV_NOTLEADER;
  else { 
      shared_ptr<Marshallable> cmd = s;
      (&raft)->Start(cmd, &index, &term);

      //let's wait for 300ms
      int k = 0;
      while (k++ < 6) {
        Coroutine::Sleep(50000);
        mtx.lock();
        if (commit_set.find(s->data_) != commit_set.end()) {
          *v = kv_map[s->data_[2]];
          Log_info("KvServer::Value is *v %s , k %s , op %s", (*v).c_str(), s->data_[2].c_str(), s->data_[1].c_str());
          mtx.unlock();
          return KV_SUCCESS;
        }
        mtx.unlock();
      }
  };
    
  return KV_NOTLEADER;
}

void KvServer::OnNextCommand(Marshallable& m) {
  Log_info("KvServer::Inside OnNextCommand");
  auto v = (MultiStringMarshallable*)(&m);

  mtx.lock();
  if (v->data_[1] == "get") {
    v->data_[3] = (kv_map[v->data_[2]]);
  }
  else 
  if (v->data_[1] == "put") {
    kv_map[v->data_[2]] = v->data_[3];
  }
  else 
  if (v->data_[1] == "append") {

    if (kv_map.find(v->data_[2]) == kv_map.end()) {
      kv_map[v->data_[2]] = v->data_[3];
    } 
    else
      kv_map[v->data_[2]] = kv_map[v->data_[2]] + v->data_[3];
  
  }
  commit_set.insert(v->data_);
  mtx.unlock();

}
