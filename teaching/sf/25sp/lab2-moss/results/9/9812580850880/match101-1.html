int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  


  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  shared_ptr<Marshallable> cmd = s;
  uint64_t index = 0;
  uint64_t term = 0;
  bool isLeader = 0;
  *v = "";
  this->GetRaftServer().GetState(&isLeader,&term);
  if(!isLeader)
  	return KV_NOTLEADER;
  
  bool appended = this->GetRaftServer().Start(cmd,&index,&term);
  operationStatus[to_string(oid)] = 0; //Pending
  Coroutine::Sleep(800000);
  //Log_info("Timeout?%d",operationStatus[to_string(oid)]);
  if(operationStatus[to_string(oid)]!=1)
  	return KV_TIMEOUT;
  if(operationStatus[to_string(oid)]==1 && map.find(k)!=map.end())
  {
  	//Log_info("get%s",k);
  	*v = map[k];
  	return KV_SUCCESS;
  }
  
  return KV_SUCCESS;
}

void KvServer::OnNextCommand(Marshallable& m) {
  
  /* your code here */


  auto s = (MultiStringMarshallable*)(&m);
  string oid = s->data_[0];
  string operation = s->data_[1];
  string key = s->data_[2];
  operationStatus[oid] = 1; //Completed
  if(operation.compare("put")==0 || (operation.compare("append")==0 && map.find(key)==map.end()))
  {
  	string value = s->data_[3];
  	//Log_info("key %s value %s",key,value);
  	map[key] = value;
  }
  else if(operation.compare("append")==0){
  	//Log_info("I am in append");
  	string value = s->data_[3];
  	map[key] = map[key]+value;
  }
  else if(operation.compare("get")==0){
  	//Log_info("I am in get");
  }
}
