int64_t KvServer::GetNextOpId() {

shared_ptr<IntEvent> 
KvServer::callToStart(shared_ptr<Marshallable> cmdptr_m, uint64_t &ind, uint64_t &term) {
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  RaftServer& rs = GetRaftServer();
  bool success = rs.Start(cmdptr_m, &ind, &term);
  return ev;
}



int KvServer::Put(const uint64_t& oid, const string& k, const string& v) {
  RaftServer& rs = GetRaftServer();
  if (rs.isLeader) {
    Log_info("Executing Put Operation for %d with key:value %s:%s", sp_log_svr_->site_id_, k.c_str(), v.c_str());
    //auto future = async(launch::async, [oid, k, v, this]() {
    auto s = make_shared<MultiStringMarshallable>();


    s->data_.push_back(to_string(oid));
    s->data_.push_back("put");
    s->data_.push_back(k);
    s->data_.push_back(v);
    uint64_t ind;
    uint64_t term;
    auto cmdptr_m = dynamic_pointer_cast<Marshallable>(s);
    auto event = callToStart(cmdptr_m, ind, term);
    event->Wait(500000);
    if (data[k]==v) {
        return KV_SUCCESS;
    } else {
      return KV_TIMEOUT;
    }
  } else {
    return KV_NOTLEADER;
  }
  //});
  //return future.get();
}




int KvServer::Append(const uint64_t& oid, 
                     const string& k,
                     const string& v) {
  RaftServer& rs = GetRaftServer();
  if (rs.isLeader) {
    Log_info("Executing Append Operation for %d with key:value %s:%s", sp_log_svr_->site_id_, k.c_str(), v.c_str());
  //auto future = async(launch::async, [oid, k, v, this]() {
    auto s = make_shared<MultiStringMarshallable>();


    s->data_.push_back(to_string(oid));
    s->data_.push_back("append");
    s->data_.push_back(k);
    s->data_.push_back(v);
    uint64_t ind;
    uint64_t term;
    auto cmdptr_m = dynamic_pointer_cast<Marshallable>(s);
    string oldValue = data[k].c_str();
    auto event = callToStart(cmdptr_m, ind, term);
    event->Wait(500000);
    Log_info("%s, %s - printing appended value", oldValue.append(v).c_str(), data[k].c_str());
    if (oldValue.append(v).compare(data[k])) {
        return KV_SUCCESS;
    } else {
        return KV_NOTLEADER;
    }
  } else {


    return KV_NOTLEADER;
  }
  //});
  //return future.get();
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) { 
  RaftServer& rs = GetRaftServer();
  if (rs.isLeader) {
    Log_info("Executing Get Operation for %d with key %s", sp_log_svr_->site_id_, k.c_str());  
  //auto future = async(launch::async, [oid, k, v, this]() {
    timeout = true;
    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(to_string(oid));
    s->data_.push_back("get");
    s->data_.push_back(k);
    RaftServer& rs = GetRaftServer();
    uint64_t ind;
    uint64_t term;
    auto cmdptr_m = dynamic_pointer_cast<Marshallable>(s);
    auto event = callToStart(cmdptr_m, ind, term);
    event->Wait(500000);
    if (!timeout && data.find(k)!=data.end()) {
        *v = data[k];
        return KV_SUCCESS;
    } else {
        return KV_TIMEOUT;
    }
  } else {
    return KV_NOTLEADER;
  }
  //});
  //return future.get();
}

void KvServer::OnNextCommand(Marshallable& m) {
  Coroutine::CreateRun([&m, this](){
    RaftServer& rs = GetRaftServer();
    auto command = (MultiStringMarshallable*)(&m);
    if (command->data_.empty()) {
      return;
    }
    string operationType = command->data_[1];
    if (operationType == "put") {
      string key = command->data_[2];
      string value = command->data_[3];
      data[key] = value;
      Log_info("Applied Put operation for key:value %s:%s for %d", key.c_str(), value.c_str(), rs.site_id_);
    } else if (operationType == "append") {
      string key = command->data_[2];
      string value = command->data_[3];
      data[key].append(value);
      Log_info("Applied Append operation for value %s:%s for %d", key.c_str(), value.c_str(), rs.site_id_);
    } else {
      Log_info("Applied Get operation"); 
      timeout = false;
    }
  });
}
