void KvServer::OnNextCommand(Marshallable& m) {

  Coroutine::CreateRun([&m, this](){ 
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  Log_info("We in OnNextCommand KV Server ----> <---- CMD : %s for Server State : %d and OPD : %s Key : %s", v->data_.at(1).c_str(),sp_log_svr_->site_id_, v->data_.at(0).c_str(), v->data_.at(2).c_str());

  if(v->data_.at(1) == "put"){
    Log_info("We in OnNextCommand PUT KV Server ----> <---- for Server State : %d and OPD : %s Key : %s Value : %s", sp_log_svr_->site_id_, v->data_.at(0).c_str(), v->data_.at(2).c_str(), v->data_.at(3).c_str());
    Log_info("KV STORE SIZE : %d ", KvDbStore.size());
    KvDbStore[v->data_.at(2)] = v->data_.at(3);
    Log_info("AFTER KV STORE SIZE : %d ", KvDbStore.size());
  } 
  else if(v->data_.at(1) == "append"){
    auto it = KvDbStore.find(v->data_.at(2));
    if (it != KvDbStore.end()) {
        KvDbStore[v->data_.at(2)] = KvDbStore[v->data_.at(2)] + v->data_.at(3);
    } else {
        KvDbStore[v->data_.at(2)] = v->data_.at(3);
    }
  } 

  if(GetRaftServer().current_state_ == 2 && op_success.find(v->data_.at(0)) != op_success.end()){
    Log_info("OnNextCommand Event OPD: %s Value is : %d and status : %d", v->data_.at(0).c_str(), op_success.at(v->data_.at(0))->cmit_evt.value_, op_success.at(v->data_.at(0))->status);
    op_success.at(v->data_.at(0))->status = KV_SUCCESS;
    op_success.at(v->data_.at(0))->cmit_evt.Set(1);  
    Log_info("OnNextCommand Set's Event OPD : %s Value to : %d and status : %d", v->data_.at(0).c_str(), op_success.at(v->data_.at(0))->cmit_evt.value_, op_success.at(v->data_.at(0))->status);
  }
  });
}
