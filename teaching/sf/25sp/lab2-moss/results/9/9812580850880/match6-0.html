void KvServer::eraseKV(const string k, const string v){
  std::unique_lock<std::recursive_mutex> lock(mtx_);

  for (std::map<std::string, putKV>::iterator it = putKVMap.begin(); it != putKVMap.end(); ++it) {
        const std::string& key = it->first;
        putKV tmpKV = it->second;
        if(tmpKV.key == k && tmpKV.value == v && tmpKV.status == 2){
         putKVMap.erase(it);
         break;
        }
    }
}

void KvServer::updateMap(const string k, const string v, int status){
  for (std::map<std::string, putKV>::iterator it = putKVMap.begin(); it != putKVMap.end(); ++it) {
        const std::string& key = it->first;
        putKV tmpKV = it->second;
        if(tmpKV.key == k && tmpKV.value == v){
           it->second.status = status;
           break;
        }
    }
}

int KvServer::checkKVSent(const string k, const string v){
  std::unique_lock<std::recursive_mutex> lock(mtx_);

  for (std::map<std::string, putKV>::iterator it = putKVMap.begin(); it != putKVMap.end(); ++it) {
        const std::string& key = it->first;
        putKV tmpKV = it->second;
        if(tmpKV.key == k && tmpKV.value == v ){
          return tmpKV.status;
        }
    }
    return 55;//random
}

void KvServer::applyToKVDB(Marshallable& m){
  
  std::unique_lock<std::recursive_mutex> lock(mtx_);

  auto v = (MultiStringMarshallable*)(&m);

  const string clientId = v->data_[0];
  const string requestType = v->data_[1];
  const string comitKey = v->data_[2];
  const string comitValue = v->data_[3];
  const string reqId = v->data_[4];



  cout<<"[OnNextCommand] Request Id: "<< reqId<<" Client Id: "<<clientId<<" Request type: "<< requestType << " Key: "<<comitKey<<" Value: "<<comitValue<<endl;



  // std::unique_lock<std::mutex> lock(mu);
  // operationApplied = true;
  applyCondition.notify_one(); // Signal that the operation is applied.

  if(!applied[reqId]){
      applied[reqId] = true;
  }

  updateMap(comitKey,comitValue,2);//2: sucess, ready to be sent

  if(requestType == "get"){
    //operationApplied = true;
    Log_info("Get Commited!!");
  }else{
    if(requestType == "put"){
      store[comitKey] = comitValue;
    }else{
      if(store.count(comitKey) == 0){
        store[comitKey] = comitValue;
      }else{
        store[comitKey] = store[comitKey] + comitValue;
      }
    }

  }
  
}
