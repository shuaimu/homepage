        eventStore[oid_s] = event;

        event->Wait(1000 * 1000);

        if (event->status_ == Event::TIMEOUT)
        {
            Log_error("Leader %d: Raft's Start call timed out for oid %s", raftServer.site_id_, oid_s.c_str());
            return KV_TIMEOUT;
        }
        else
        {


            Log_info("Leader %d: Applied key-value op %s successfully", raftServer.site_id_, oid_s.c_str());

        }

    int KvServer::Get(const uint64_t &oid,
                      const string &k,
                      string *v)
    {

        RaftServer &raftServer = GetRaftServer();

        string oid_s = to_string(oid);

        auto newCmd = make_shared<MultiStringMarshallable>();
        newCmd->data_.push_back(oid_s);
        newCmd->data_.push_back("get");
        newCmd->data_.push_back(k);
        newCmd->data_.push_back("");

        uint64_t *term = new uint64_t;
        uint64_t *index = new uint64_t;

        auto cmdptr_m = dynamic_pointer_cast<Marshallable>(newCmd);

        bool isLeader = raftServer.Start(cmdptr_m, index, term);

        if (isLeader == false)
        {
            return KV_NOTLEADER;
        }

        Log_info("Leader %d: Sent GET command %s to Raft cluster", raftServer.site_id_, oid_s.c_str());

        auto event = Reactor::CreateSpEvent<IntEvent>();

        eventStore[oid_s] = event;

        event->Wait(1000 * 1000);

        if (event->status_ == Event::TIMEOUT)
        {
            Log_error("Leader %d: Raft's Start call timed out for oid %s", raftServer.site_id_, oid_s.c_str());
            return KV_TIMEOUT;
        }
        else
        {
            Log_info("Leader %d: Applied key-value op %s successfully", raftServer.site_id_, oid_s.c_str());
            *v = kvStore[k];
            return KV_SUCCESS;
        }

        return KV_SUCCESS;
    }

    void KvServer::OnNextCommand(Marshallable &m)
    {
        auto v = (MultiStringMarshallable *)(&m);
        RaftServer &raftServer = GetRaftServer();

        string oid_s = v->data_[0];
        string opType = v->data_[1];
        string key = v->data_[2];
        string value = v->data_[3];

        Log_info("Server %d: Applied %s command in raft: %s", raftServer.site_id_, opType.c_str(), oid_s.c_str());

        if (opType == "put")
        {
            kvStore[key] = value;
        }
        else if (opType == "append")
        {
            auto initialEntry = kvStore.find(key);
            if (initialEntry == kvStore.end())
            {
                kvStore[key] = value;
            }
            else
            {
                kvStore[key] = initialEntry->second + value;
            }
        }

        auto ev = eventStore.find(oid_s);
        if (ev != eventStore.end())
        {
            Log_info("Leader %d: Setting event for %s", raftServer.site_id_, oid_s.c_str());
            ev->second->Set(1);
        }
    }

    {

        cli->commo_ = commo_;

        cli->cli_id_ = id + cli_cnt_++;
        return cli;
    }
