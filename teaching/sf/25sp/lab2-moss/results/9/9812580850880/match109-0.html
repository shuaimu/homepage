int64_t KvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;
  ret = ret << 32;
  ret = ret + op_id_cnt_++; 
  return ret;
}

                  const string& k,
                  const string& v) {
  
  /*

  s->data_.push_back(to_string(op_id));

  */
  /* your code here */
  RaftServer& svr = GetRaftServer();
  
  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  auto cmd = dynamic_pointer_cast<Marshallable>(s);

  uint64_t index;
  uint64_t term=0;

  auto leader = svr.Start(cmd, &index, &term);
  Log_info("%d started the wait for cosleep for PUT",svr.loc_id_);
 
  if(leader){
  Log_info("%d is the leader",svr.loc_id_);
  }

  else if(leader==false){
  Log_info("trying for the wrong server, leader isn't %d",svr.loc_id_);
  return KV_NOTLEADER;
}
  
  Coroutine::Sleep(500000);
  Log_info("%d done with wait for cosleep for PUT",svr.loc_id_);

if(ops.find(to_string(oid)) != ops.end() && leader==true) {


  Log_info("%d: KV PAIR SUCCESSFULLY INSERTED",svr.loc_id_);
  return KV_SUCCESS;
}
  Log_info("log not inserted in leader so timing out %d",svr.loc_id_);
  return KV_TIMEOUT;
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  string op = v->data_[1];
  if(op=="put"){
    ops.insert(v->data_[0]);
    kv[v->data_[2]]=v->data_[3];
    Log_info("Putting value %s for %s",v->data_[2],v->data_[3]);
  }
  else if(op=="append"){
    ops.insert(v->data_[0]);
    kv[v->data_[2]]=kv[v->data_[2]]+(v->data_[3]);
    Log_info("appending value %s for %s",v->data_[2],v->data_[3]);
    Log_info("%s now has value %s",v->data_[2],kv[v->data_[2]]);
  }
  else{
    Log_info("going to attempt to fetch value");
    ops.insert(v->data_[0]);
    Log_info("Succeeded to fetch value");
  }
}

  cli->commo_ = commo_;
