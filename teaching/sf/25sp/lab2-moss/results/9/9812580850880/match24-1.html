int KvServer::HandleGetReturn(uint64_t oid, uint64_t index, uint64_t term, string* v)
{
  CANARY;
  auto start = getTime();
  while(getTime() - start < messageTimeout)
  {
    Coroutine::Sleep(messagePoll.count());
  CANARY;
    auto it = GetTable.find(oid);
    if (it == GetTable.end())
      continue;
    *v = it->second;
    return KV_SUCCESS;
  }
  return KV_TIMEOUT;
}

int KvServer::HandleBoolReturn(uint64_t oid, uint64_t index, uint64_t term)
{
  CANARY;
  auto start = getTime();
  while(getTime() - start < messageTimeout)
  {
  CANARY;
    Coroutine::Sleep(messagePoll.count());
    auto it = BoolTable.find(oid);
    if (it == BoolTable.end())
      continue;
    return KV_SUCCESS;
  }
  return KV_TIMEOUT;
}

void KvServer::OnNextCommand(Marshallable& m) {
  CANARY;
  auto cmd = (MultiStringMarshallable*)(&m);
  /* your code here */
  uint64_t oid = stoull(cmd->data_[0]);
  string cmd_s = cmd->data_[1];
  string key = cmd->data_[2];

  auto state = StateTable.find(key);

  if (cmd_s == "get")
  {
    CANARY;
    Log_info("SIZE: %d", StateTable.size());
    verify(state != StateTable.end());
    string s(state->second);
    GetTable[oid] = s;
  }
  else if (cmd_s == "append")
  {
    CANARY;
    // Append acts as PUT if there's nothing to append to!
    if (state == StateTable.end())
      StateTable.insert({key, cmd->data_[3]});
    else
      StateTable[key] = state->second + cmd->data_[3];
    BoolTable[oid] = true;
  }
  else if (cmd_s == "put")
  {
    CANARY;
    if (state == StateTable.end())
      StateTable.insert({key, cmd->data_[3]});
    else
      StateTable[key] = cmd->data_[3];
    BoolTable[oid] = true;
  }
  else
  {
    CANARY;
    Log_info("BAD! WRONG COMMAND! %s", cmd_s);
    exit(EXIT_FAILURE);
  }

}
