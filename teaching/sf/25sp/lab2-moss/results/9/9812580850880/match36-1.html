int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {

  std::lock_guard<std::recursive_mutex> lock(mtx_);


  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  uint64_t index = -1;
  uint64_t term = -1;
  RaftServer &rs= GetRaftServer();
  Log_info("Server ID %d Is leader: %s", rs.loc_id_, rs.currentState.c_str());
  shared_ptr<Marshallable> cmd = s;
  if (rs.currentState!="leader"){
    return KV_NOTLEADER;
  }
  getCommit=false;
  rs.Start(cmd,&index, &term);
  Coroutine::Sleep(550000);
 
  *v=kvmap[k];
  if(getCommit==false){
   return KV_TIMEOUT;
  }
  return KV_SUCCESS;
}

void KvServer::OnNextCommand(Marshallable& m) {
  std::lock_guard<std::recursive_mutex> lock(mtx_);
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  if(v->data_[1]=="put"){
    putCommit=true;
    kvmap[v->data_[2]]=v->data_[3];
  }
  if(v->data_[1]=="append"){
    appendCommit=true;
    kvmap[v->data_[2]]+=v->data_[3];
  }
  if(v->data_[1]=="get"){
    getCommit=true;
  }
  Log_info("On next command called");
}
