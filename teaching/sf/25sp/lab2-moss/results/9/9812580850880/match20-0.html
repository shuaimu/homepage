                  const string& k,

  uint64_t term = 0;
  uint64_t index = 0;

  if(GetRaftServer().serverType != LEADER){
    return KV_NOTLEADER;
  }

  std::recursive_mutex mtx;
  auto s = make_shared<MultiStringMarshallable>();

  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);

  auto cmd = shared_ptr<Marshallable>(s);

  auto ev = Reactor::CreateSpEvent<IntEvent>();

  Log_info("leader %d inserting oid %lu to the queue", GetRaftServer().loc_id_, oid);

  queue[to_string(oid)] = ev;
  mtx.unlock();

  if(!GetRaftServer().Start(cmd, &index, &term)){

  }
  else{
    ev->Wait(1200000);

    if(ev->status_ == Event::TIMEOUT){
      Log_info("leader %d sending timeout event for put for oid %lu", GetRaftServer().loc_id_, oid);
      return KV_TIMEOUT;
    }
    else{
      return KV_SUCCESS;
    } 
  } 
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */

  uint64_t term = 0;
  uint64_t index = 0;

  Log_info("leader %d received get request for oid %lu", GetRaftServer().loc_id_, oid);
  if(GetRaftServer().serverType != LEADER){
    return KV_NOTLEADER;
  }
  


  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  s->data_.push_back("value");

  auto cmd = shared_ptr<Marshallable>(s);

  std::recursive_mutex mtx;
  auto ev = Reactor::CreateSpEvent<IntEvent>();

  Log_info("leader %d inserting oid %lu to the queue", GetRaftServer().loc_id_, oid);
  
  mtx.lock();
  queue[to_string(oid)] = ev;
  mtx.unlock();

  if(!GetRaftServer().Start(cmd, &index, &term)){
    return KV_NOTLEADER;
  }
  else{
    ev->Wait(1200000);

    if(ev->status_ == Event::TIMEOUT){
      Log_info("leader %d sending timeout event for get for oid %lu", GetRaftServer().loc_id_, oid);
      return KV_TIMEOUT;
    }
    else{
      mtx.lock();
      *v = datamap[k];
      mtx.unlock();
    }

    Log_info("leader %d sending success event for oid %lu", GetRaftServer().loc_id_, oid);
    return KV_SUCCESS;
  }
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  string oid = v->data_[0];


  string type = v->data_[1];
  string key = v->data_[2];
  string value = v->data_[3];

  Log_info("server %d on next command oid %lu", GetRaftServer().loc_id_, stoull(oid));

  std::recursive_mutex mtx;
  if(type == "append"){
    Log_info("server %d, entering append on next", GetRaftServer().loc_id_);
    mtx.lock();
    if(datamap.find(key) != datamap.end()){
      datamap[key] += value;
    }
    else{
      datamap[key] = value;
    }
    mtx.unlock();
  }
  if(type == "put"){
    Log_info("server %d, entering put onnext", GetRaftServer().loc_id_);
    mtx.lock();
    datamap[key] = value;
    mtx.unlock();
  }
  if(GetRaftServer().serverType == LEADER){
    Log_info("leader %d oid %lu inside on next for leader", GetRaftServer().loc_id_, stoull(oid));
    Log_info("leader %d going to set event in on next", GetRaftServer().loc_id_);
    if(queue.find(oid) != queue.end()){
      Log_info("leader %d setting event in on next", GetRaftServer().loc_id_);
      queue[oid]->Set(1);
    }
  }
}
