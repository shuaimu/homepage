int KvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
  uint64_t myLocId = GetRaftServer().loc_id_;                   
  Log_info("Site %d: Put started with %s:%s",myLocId,k.c_str(),v.c_str());
  shared_ptr<MultiStringMarshallable> s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  


  shared_ptr<Marshallable> sMar = dynamic_pointer_cast<Marshallable>(s);

  uint64_t raftLeaderIndex,raftLeaderTerm;

  bool isLeader = GetRaftServer().Start(sMar,&raftLeaderIndex,&raftLeaderTerm);

  if(!isLeader)
  {
    return KV_NOTLEADER;
  }

  if(opEvents.find(to_string(oid)) == opEvents.end())
  {
    opEvents[to_string(oid)] = nullptr;
  }

  if(opEvents[to_string(oid)] == nullptr)
  {
    opEvents[to_string(oid)] = Reactor::CreateSpEvent<IntEvent>();
  }

  opEvents[to_string(oid)]->Wait(2000000);

  if(opEvents[to_string(oid)]->status_ == Event::TIMEOUT)
  {
    //Should we store this anyway for "record"?
    opEvents[to_string(oid)] = nullptr;
    return KV_TIMEOUT;
  }

  opEvents[to_string(oid)] = nullptr;
  return KV_SUCCESS;
  

}

int KvServer::Append(const uint64_t& oid, 
                     const string& k,
                     const string& v) {

  uint64_t myLocId = GetRaftServer().loc_id_;                   
  Log_info("Site %d:Append started with %s:%s",myLocId,k.c_str(),v.c_str());
  shared_ptr<MultiStringMarshallable> s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("append");
  s->data_.push_back(k);
  s->data_.push_back(v);
  


  shared_ptr<Marshallable> sMar = dynamic_pointer_cast<Marshallable>(s);

  uint64_t raftLeaderIndex,raftLeaderTerm;

  bool isLeader = GetRaftServer().Start(sMar,&raftLeaderIndex,&raftLeaderTerm);

  if(!isLeader)
  {
    Log_info("Site %d:Returning not leader with %s:%s",myLocId,k.c_str(),v.c_str());
    return KV_NOTLEADER;
  }

  if(opEvents.find(to_string(oid)) == opEvents.end())
  {
    opEvents[to_string(oid)] = nullptr;
  }

  if(opEvents[to_string(oid)] == nullptr)
  {
    opEvents[to_string(oid)] = Reactor::CreateSpEvent<IntEvent>();
  }

  opEvents[to_string(oid)]->Wait(2000000);

  if(opEvents[to_string(oid)]->status_ == Event::TIMEOUT)
  {
    Log_info("Site %d:Timeout with %s:%s",myLocId,k.c_str(),v.c_str());
    opEvents[to_string(oid)] = nullptr;
    return KV_TIMEOUT;
  }
  
  opEvents[to_string(oid)] = nullptr;
  Log_info("Site %d: Appended into DB values %s:%s",myLocId,k.c_str(),v.c_str());

  return KV_SUCCESS;
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  uint64_t myLocId = GetRaftServer().loc_id_;    
  Log_info("Site %d: Get started with key %s",myLocId,k.c_str());
  shared_ptr<MultiStringMarshallable> s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);  


  shared_ptr<Marshallable> sMar = dynamic_pointer_cast<Marshallable>(s);

  uint64_t raftLeaderIndex,raftLeaderTerm;

  bool isLeader = GetRaftServer().Start(sMar,&raftLeaderIndex,&raftLeaderTerm);

  if(!isLeader)
  {
    return KV_NOTLEADER;
  }

  if(opEvents.find(to_string(oid)) == opEvents.end())
  {
    opEvents[to_string(oid)] = nullptr;
  }

  if(opEvents[to_string(oid)] == nullptr)
  {
    opEvents[to_string(oid)] = Reactor::CreateSpEvent<IntEvent>();
  }

  opEvents[to_string(oid)]->Wait(2000000);

  if(opEvents[to_string(oid)]->status_ == Event::TIMEOUT)
  {
    opEvents[to_string(oid)] = nullptr;
    return KV_TIMEOUT;
  }

  if(actualDb.find(k) == actualDb.end())
  {
    v = nullptr;
  }
  else
  {


    *v = actualDb[k];
  }

  return KV_SUCCESS;
}

    if(optype == "put")
    {
        string opval = v->data_[3];
        actualDb[opkey] = opval;

    }
    if(optype == "append")
    {
        string opval = v->data_[3];
        if(actualDb.find(opkey) == actualDb.end())
