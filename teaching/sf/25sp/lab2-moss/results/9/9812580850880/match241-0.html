  void KvServer::OnNextCommand(Marshallable &m)
  {
    auto v = (MultiStringMarshallable *)(&m);
    /* your code here */
    // Log_info("GOT COMMAND FROM RAFT");

    // for (auto x : v->data_)
    // {
    //   cout << x << " ";
    // }
    // cout << endl;

    response = true;

    string oid = v->data_[0];
    string op = v->data_[1];
    string key = v->data_[2];

    if (op == GET)
      return;

    string value = v->data_[3];

    if (op == PUT)
    {
      db[key] = value;


      return;
    }

    if (db.find(key) == db.end())
      db[key] = value;
    else
      db[key] += value;
  }
