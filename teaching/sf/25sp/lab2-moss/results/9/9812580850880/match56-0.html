int64_t KvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;
  ret = ret << 32;
  ret = ret + op_id_cnt_++; 
  return ret;
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  auto data = v->data_;
  if(data[1] == "put"){
    Log_info("PUTT %d [%d:%d]!", GetRaftServer().loc_id_, data[2], data[3]);
    kvStore[data[2]] = data[3];
    put_f=true;
  }
  else if(data[1] == "append"){
    Log_info("APPEND %d [%d:%d]!", GetRaftServer().loc_id_, data[2], data[3]);
    // kvStore[data[2]] += data[3];
    if (kvStore.find(data[2]) != kvStore.end()) {
      kvStore[data[2]] += data[3];
    } 
    else {
      kvStore[data[2]] = data[3];
    }
    app_f=true;
  }
  else if(data[1] == "get"){
    Log_info("GET %d [%d]!", GetRaftServer().loc_id_, data[2]);
    // data[3] = kvStore[data[2]];
    get_f=true;
    // Log_info("I GOT %s %s %s", data[1].c_str(),data[2].c_str(),data[3].c_str());
  }
}

shared_ptr<KvClient> KvServer::CreateClient() {
  /* don't change this function */
  auto cli = make_shared<KvClient>();
  verify(commo_ != nullptr);
  cli->commo_ = commo_;
  uint32_t id = sp_log_svr_->site_id_;

  cli->cli_id_ = id+cli_cnt_++;
