  int KvServer::Get(const uint64_t &oid,
                    const string &k,
                    string *v)
  {
    /* your code here */
    //lock_guard<std::mutex> lock(myMutex);


    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(to_string(oid));
    s->data_.push_back("get");
    s->data_.push_back(k);
    RaftServer &raftServ = GetRaftServer();
    
    // lock_guard<std::mutex> lock(myMutex);
    uint64_t commitIndex;
    uint64_t term;
    bool res;
    auto m_obj = dynamic_pointer_cast<Marshallable>(s);
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    Log_info("calling start for get");
    res = raftServ.Start(m_obj,&commitIndex,&term);
    ev->Wait(350000);
    if(!res){
      return KV_NOTLEADER;
    }
    else if(res){
      if(raftServ.lastApplied>=commitIndex){
        *v = data[k];
        return KV_SUCCESS;
      }
      else{
        return KV_TIMEOUT;
      }
    }
    else{return KV_TIMEOUT;}
  }



  void KvServer::OnNextCommand(Marshallable &m)
  {
    auto v = (MultiStringMarshallable *)(&m);
    /* your code here */
    
    Coroutine::CreateRun([this,v](){
      
      auto opid = v->data_[0];
      auto operation = v->data_[1];
      Log_info("in on next for op id %s",opid);
      Log_info("some infot in next command %s operation",operation);
      if(operation=="put"){
        data[v->data_[2]] = v->data_[3];
      }
      else if(operation=="append"){
        data[v->data_[2]] += v->data_[3];
      }
      Log_info("trying to apply lastapplied?");
      GetRaftServer().lastApplied+=1;

      Log_info("applied lastapplied?");
    });

  }
