int64_t KvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;

  ret = ret + op_id_cnt_++;

}

shared_ptr<KvClient> KvServer::CreateClient() {
  /* don't change this function */
  auto cli = make_shared<KvClient>();
  verify(commo_ != nullptr);
  cli->commo_ = commo_;
