void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  Log_info(" In on next command size of v->data_ array : %d",v->data_.size());
  string comm = v->data_[1].c_str();
  string key = v->data_[2].c_str();
  string val = v->data_[3].c_str();
  Log_info("no error yet %s ",v->data_[1].c_str());
  if(comm == "put")
  {
    Log_info("--PUT--");
    evp.Set(1);
    if(kv_map.find(key)==kv_map.end())
    {
      Log_info("not found");
      kv_map[key] = val;
    }
    else{
      Log_info("found");
      kv_map[key] = val;
      Log_info("In Append, String is %s",kv_map[key].c_str());
    }
  }
  else if(comm == "append")
  {
    Log_info("--APPEND--");
    //eva->Set(1);
    eva.Set(1);
    if(kv_map.find(key)==kv_map.end())
    {
      Log_info("not found");
      kv_map[key] = val;
    }
    else{
      Log_info("found");
      kv_map[key] += val;
      Log_info("In Append, String is %s",kv_map[key].c_str());
    }
  }
  else{
    //evg->Set(1);
    Log_info("--GET-- for key: %s",key.c_str());
    evg.Set(1);

  }
}
