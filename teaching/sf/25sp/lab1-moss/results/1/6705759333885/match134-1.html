shared_ptr<IntEvent> RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t term,
                                uint64_t candidateId,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                uint64_t* voterTerm,
                                bool_t* vote_granted) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  //TODO: gautam add timeout for events or keep waiting??
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {


      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [candidateId, site_id, term, voterTerm, vote_granted, ev](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        fu->get_reply() >> *voterTerm;
        fu->get_reply() >> *vote_granted;
        ev->Set(1);
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */

      Call_Async(proxy, RequestVote, term, candidateId, lastLogIndex, lastLogTerm, fuattr);
    }
  }
  return ev;
}


shared_ptr<IntEvent> RaftCommo::SendEmptyAppendEntry(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t leaderTerm,
                                  uint64_t leaderId,
                                  uint64_t prevLogIndex,
                                  uint64_t prevLogTerm,
                                  uint64_t leaderCommitIndex,
                                  uint64_t* replyTerm) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
   auto ev = Reactor::CreateSpEvent<IntEvent>();
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev, replyTerm, leaderId, site_id](Future* fu) {
        fu->get_reply() >> *replyTerm;
        ev->Set(2);
      };
      Call_Async(proxy, EmptyAppendEntry, leaderTerm, leaderId, prevLogIndex, prevLogTerm, leaderCommitIndex, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent> RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t leaderTerm,
                                  uint64_t leaderId,
                                  uint64_t prevLogIndex,
                                  uint64_t prevLogTerm,
                                  vector<LogEntry> entries,
                                  uint64_t leaderCommitIndex,
                                  uint64_t* replyTerm,
                                  bool_t* followerAppendOK
                                  ) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
   //TODO: gautam Paper says its possible to send multiple entries over single RPC to improve efficiency
    auto ev = Reactor::CreateSpEvent<IntEvent>();
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;


      fuattr.callback = [ev, leaderTerm, leaderId, prevLogIndex, prevLogTerm, entries, leaderCommitIndex, replyTerm, followerAppendOK](Future* fu) {
        fu->get_reply() >> *replyTerm;
        fu->get_reply() >> *followerAppendOK;
        ev->Set(3);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      vector<janus::RPCLogEntry> vec;
      for(auto& logEntry: entries) {
        MarshallDeputy md(logEntry.cmd);
        vec.emplace_back(RPCLogEntry{logEntry.term, md});
      }
      //MarshallDeputy md(cmd);
      Call_Async(proxy, AppendEntries, leaderTerm, leaderId, prevLogIndex, prevLogTerm, vec, leaderCommitIndex, fuattr);
    }
  }
  return ev;
}


shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();

  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
