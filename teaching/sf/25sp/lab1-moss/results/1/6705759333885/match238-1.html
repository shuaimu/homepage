shared_ptr<IntEvent> 
RaftCommo::SendRequestVote(parid_t par_id,
                           ballot_t term, 
                           siteid_t site_id,
                           locid_t candidate_loc_id_,
                           slotid_t last_log_index,
                           ballot_t last_log_term, 
                           ballot_t* current_term_,
                           bool_t* vote_granted) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */

  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [current_term_, vote_granted, ev](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        Log_info("RPC callback....!");
        fu->get_reply() >> *current_term_;
        fu->get_reply() >> *vote_granted;
        ev->Set(1);
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Log_info("Calling Async for %d :",site_id);
      Call_Async(proxy, RequestVote, term, candidate_loc_id_, last_log_index, last_log_term, fuattr);
      Log_info("DoneCalling Async for %d :",site_id);
    }
  }
  Log_info("Return Event from Commo for %d :",site_id);
  return ev;
}

void RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  shared_ptr<Marshallable> cmd) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [](Future* fu) {
        bool_t followerAppendOK;
        fu->get_reply() >> followerAppendOK;
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      MarshallDeputy md(cmd);
      Call_Async(proxy, AppendEntries, md, fuattr);
    }
  }
}



shared_ptr<IntEvent> 
RaftCommo::SendEmptyAppendEntries(parid_t par_id,
                                  ballot_t term,
                                  siteid_t site_id,
                                  locid_t leader_loc_id_,
                                  slotid_t prev_Log_Index_,
                                  ballot_t prev_Log_Term_,
                                  std::vector<MarshallDeputy> suffix_logs_,


                                  std::vector<ballot_t> suffix_logs_term_,
                                  slotid_t leader_commit_,                              
                                  ballot_t* current_term_,
                                  slotid_t* res_ack_idx_,
                                  bool_t* followerAppendOK) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [current_term_, res_ack_idx_, followerAppendOK, ev](Future* fu) {
        fu->get_reply() >> *current_term_;
        fu->get_reply() >> *res_ack_idx_;
        fu->get_reply() >> *followerAppendOK;
        ev->Set(1);
        Log_info("RPC callback....!");
      };
      Log_info("Calling Async for %d :",site_id);
      Call_Async(proxy, EmptyAppendEntries, term, leader_loc_id_, prev_Log_Index_, prev_Log_Term_, suffix_logs_, suffix_logs_term_, leader_commit_, fuattr);
      Log_info("DoneCalling Async for %d :",site_id);
    }
  }
  Log_info("Return Event from Commo for %d :",site_id);
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
