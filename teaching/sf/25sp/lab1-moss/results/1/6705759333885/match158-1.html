shared_ptr<IntEvent>  RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,                                
                                const uint64_t& candidate_id,
                                const uint64_t& term,
                                const uint64_t& last_log_term,
                                const uint64_t& last_log_index,
                                uint64_t* ret1,
                                bool_t* vote_granted) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ret1, vote_granted, ev](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        // uint64_t ret1;
        // bool_t vote_granted;
        /* retrieve RPC return values in order */
        fu->get_reply() >> *ret1;
        fu->get_reply() >> *vote_granted;
        ev->Set(1);
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, candidate_id, term, last_log_term, last_log_index, fuattr);
    }
  }

  return ev;
}

   */

  auto ev = Reactor::CreateSpEvent<IntEvent>();

shared_ptr<IntEvent> RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  const uint64_t& leader_term,
                                  const uint64_t& leader_id,
                                  const uint64_t& prev_log_index,
                                  const uint64_t& prev_log_term,
                                  const uint64_t& leader_commit,


                                  vector<uint64_t> terms,
                                  vector<shared_ptr<Marshallable>> cmds,
                                  bool_t* followerAppendOK,
                                  uint64_t* follower_term,
                                  uint64_t* set_next_index
                                 ) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */

  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [followerAppendOK, follower_term, set_next_index, ev](Future* fu) {


        fu->get_reply() >> *followerAppendOK;
        fu->get_reply() >> *follower_term;
        fu->get_reply() >> *set_next_index;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      vector<MarshallDeputy> mds;
      for(shared_ptr<Marshallable> cmd:cmds){
        mds.push_back(MarshallDeputy(cmd));
      }

      Call_Async(proxy, 
                 AppendEntries,
                 leader_term, 
                 leader_id, 
                 prev_log_index, 
                 prev_log_term, 
                 leader_commit,
                 terms, 
                 mds, 
                 fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
