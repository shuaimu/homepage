shared_ptr<IntEvent> RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                siteid_t receiver,
                                uint64_t candidateTerm,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                bool_t* voteGranted,
                                uint64_t* updatedTerm) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  Log_info("server %d Inside send requestVote commmo", site_id);
  
  auto globalEv = Reactor::CreateSpEvent<IntEvent>();
  auto proxies = rpc_par_proxies_[par_id];
  uint64_t noServers = proxies.size();
  Log_info("server %d majority needed: %d", site_id, noServers);
  
  for (auto& p : proxies) {
    if (p.first == receiver) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;

      fuattr.callback = [updatedTerm, voteGranted, site_id, receiver, globalEv](Future* fu) {
        Log_info("server %d Returning to the callback for requestvote to %d", site_id, receiver);

        fu->get_reply() >> *updatedTerm;
        fu->get_reply() >> *voteGranted;
        /* process the RPC response here */
        Log_info("server %d Term %d and vote granted %d", site_id, *updatedTerm, *voteGranted);
        Log_info("server %d In the callback", site_id);

        globalEv->Set(1);
        };

      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Log_info("server %d Sending async requestVote call to server %d", site_id, p.first);
      Call_Async(proxy, RequestVote, site_id, candidateTerm, lastLogIndex, lastLogTerm, fuattr);
    }
  }
  return globalEv;
}



shared_ptr<IntEvent> RaftCommo::SendEmptyAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t leaderTerm,
                                  uint64_t leaderCommitIndex,
                                  uint64_t* updatedTerm,
                                  bool_t* appendOK) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto globalEv = Reactor::CreateSpEvent<IntEvent>();

  for (auto& p : proxies) {
    if (p.first != site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;

      fuattr.callback = [updatedTerm, appendOK, globalEv](Future* fu) {
        uint64_t responseTerm = 0;
        bool_t electionTimeoutReset = false;
        fu->get_reply() >> electionTimeoutReset;
        fu->get_reply() >> responseTerm;
        
        std::mutex sharedUpdateMutex;
        std::lock_guard<std::mutex> lock_guard(sharedUpdateMutex);
        
        if(responseTerm > 0 && responseTerm > *updatedTerm){
          *updatedTerm = responseTerm;
        }
        if(electionTimeoutReset){


          *appendOK = electionTimeoutReset;
        }
        globalEv->Set(1);
      };
      Call_Async(proxy, EmptyAppendEntries, site_id, leaderTerm, leaderCommitIndex, fuattr);
    }
  }
  return globalEv;
}

shared_ptr<IntEvent> RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  siteid_t receiver,
                                  uint64_t leaderTerm,
                                  uint64_t prevLogIndex,
                                  uint64_t prevLogTerm,
                                  shared_ptr<Marshallable> cmd,
                                  uint64_t leaderCommitIndex,
                                  uint64_t elementTerm,
                                  bool_t isHeartbeat,
                                  bool_t* followerAppendOK,
                                  uint64_t* updatedTerm,
                                  uint64_t* followerNextIndex) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();

  for (auto& p : proxies) {
    if (p.first == receiver) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev, followerAppendOK, updatedTerm, receiver, site_id, followerNextIndex](Future* fu) {
        std::mutex sharedUpdateMutex;
        std::lock_guard<std::mutex> lock_guard(sharedUpdateMutex);
        
        bool_t appendOK = false;
        uint64_t responseTerm = 0;
        fu->get_reply() >> responseTerm;
        fu->get_reply() >> appendOK;
        fu->get_reply() >> *followerNextIndex;

        Log_info("leader %d append entry callback from server %d", site_id, receiver);

        if(responseTerm > 0 && responseTerm > *updatedTerm){
          *updatedTerm = responseTerm;
        }
        *followerAppendOK = appendOK;
        Log_info("leader %d callback from the server %d with appendok: %d %d", site_id, receiver, appendOK, *followerAppendOK);
        Log_info("leader %d setting the event status", site_id);
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      MarshallDeputy md(cmd);
      Log_info("leader %d sending async call append entries to server %d", site_id, receiver);
      Call_Async(proxy, AppendEntries, site_id, leaderTerm, prevLogIndex, prevLogTerm, leaderCommitIndex, md, elementTerm, isHeartbeat, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res, uint64_t* count) {
  Log_info("server %d enter the sendstring commo", loc_id_);
  siteid_t id = loc_id_;
  auto proxies = rpc_par_proxies_[par_id];
  
  auto globalEv = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first != site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,id,globalEv,count](Future* fu) {
        fu->get_reply() >> *res;
        Log_info("server %d control is in the call back", id);
        if(*res == "world"){
        *count = *count + 1;
        Log_info("server %d updating count to: %d", id, *count);
        if(*count >= 5){
          Log_info("server %d setting global status based on count value %d", id, *count);
          globalEv->Set(1);
          Log_info("server %d after global event set", id);
        }
      }
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Log_info("server %d call RPC async", loc_id_);
      Call_Async(proxy, HelloRpc, msg, fuattr);
      Log_info("server %d After the call async method", loc_id_);
    }
  }

  return globalEv;
}
