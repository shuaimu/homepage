void 
RaftCommo::SendEmptyAppendEntries(parid_t par_id,
                      siteid_t site_id,
                      uint64_t leaderCommitIndex,
                      uint64_t term,
                      uint64_t prevLogIndex,
                      uint64_t prevLogTerm,
                      uint64_t leaderId,
                      shared_ptr<IntEvent> evToTrigger,
                      uint64_t* termReturned,
                      bool_t* successReturned,
                      uint64_t* conflictTermReturned,
                      uint64_t* conflictIdxReturned)
{
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [termReturned, successReturned,conflictTermReturned, conflictIdxReturned, evToTrigger](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        bool_t followerAppendOkReceived;
        uint64_t lclTermReturned;
        uint64_t lclConflictTermReturned;
        uint64_t lclConflictIdxReturned;

        fu->get_reply() >> lclTermReturned;
        fu->get_reply() >> followerAppendOkReceived;
        fu->get_reply() >> lclConflictTermReturned;
        fu->get_reply() >> lclConflictIdxReturned;

        *termReturned = lclTermReturned;
        *successReturned = followerAppendOkReceived;
        *conflictTermReturned = lclConflictIdxReturned;


        *conflictIdxReturned = lclConflictIdxReturned;
        if(evToTrigger->status_ != Event::TIMEOUT)
        {
          evToTrigger->Set(1);
        }
        

        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, EmptyAppendEntries, leaderCommitIndex, term, prevLogIndex,prevLogTerm, leaderId, fuattr);
    }
  }
}



void 


RaftCommo::SendFullAppendEntries(parid_t par_id,
                      siteid_t site_id,
                      uint64_t leaderCommitIndex,
                      uint64_t term,
                      uint64_t prevLogIndex,
                      uint64_t prevLogTerm,
                      std::vector<shared_ptr<Marshallable>> cmds,

    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [termReturned, successReturned, conflictTermReturned, conflictIdxReturned, evToTrigger](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        bool_t followerAppendOkReceived;
        uint64_t lclTermReturned;
        uint64_t lclConflictTermReturned;
        uint64_t lclConflictIdxReturned;
        fu->get_reply() >> lclTermReturned;
        fu->get_reply() >> followerAppendOkReceived;
        fu->get_reply() >> lclConflictTermReturned;
        fu->get_reply() >> lclConflictIdxReturned;

        *successReturned = followerAppendOkReceived;
        *conflictTermReturned = lclConflictIdxReturned;
        *conflictIdxReturned = lclConflictIdxReturned;
        if(evToTrigger->status_ != Event::TIMEOUT)

shared_ptr<QuorumEvent>  
RaftCommo::SendRequestVote(     parid_t par_id,
                                siteid_t mySiteId,
                                uint64_t term,
                                uint64_t candidateId,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                shared_ptr<uint64_t>  currentTermReturned) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto qev = Reactor::CreateSpEvent<QuorumEvent>(4,2);
  for (auto& p : proxies) {
    if (p.first != mySiteId) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [qev, currentTermReturned, mySiteId](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        uint64_t termRecv;
        uint64_t lclVoteGranted;
        fu->get_reply() >> termRecv;
                

        if (currentTermReturned == NULL)
        {
              Log_info("Site %d :Null detected", mySiteId);
        }

        if (termRecv > *currentTermReturned)
        {
          *currentTermReturned = termRecv;
        }
        
        fu->get_reply() >> lclVoteGranted;


        if(lclVoteGranted)
        {
          Log_info("Site %d :Voting yes", mySiteId);
          qev->VoteYes();
        }
        else
        {
          Log_info("Site %d :Voting no", mySiteId);
          qev->VoteNo();
        }
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, term, candidateId, lastLogIndex,lastLogTerm, fuattr);
    }
  }
  return qev;
}

void 
RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t leaderCommitIndex,
                                  shared_ptr<Marshallable> cmd,
                                  bool_t* didVote,
                                  shared_ptr<QuorumEvent> aerev) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [didVote,aerev](Future* fu) {
        fu->get_reply() >> *didVote;
        if(*didVote)
        {
          aerev->VoteYes();
        }
        else
        {
          aerev->VoteNo();
        }
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      MarshallDeputy md(cmd);
      Call_Async(proxy, AppendEntries, md, leaderCommitIndex, fuattr);

    }
  }
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
