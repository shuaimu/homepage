shared_ptr<IntEvent> 
RaftCommo::SendRequestVote(parid_t par_id,
                            siteid_t site_id,
                       locid_t candidateId,
                       uint64_t lastLogIndex,
                       uint64_t lastLogTerm,
                       uint64_t candidateTerm,
                       uint64_t* resTerm,
                       bool_t* voteGranted,
                       bool_t* revertServer) {
  
  Log_info("RPC Call SendRequestVote for candidate %lli ",candidateId);
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [resTerm,voteGranted,revertServer,ev](Future* fu) {


        fu->get_reply() >> *resTerm;
        fu->get_reply() >> *voteGranted;
        fu->get_reply() >> *revertServer;
        ev->Set(1);
      };
      Call_Async(proxy, RequestVote, candidateId, lastLogIndex, lastLogTerm, candidateTerm, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendAppendEntries(parid_t par_id,
                                siteid_t site_id,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                uint64_t leaderTerm,
                                locid_t leaderId,
                                uint64_t leaderCommit,
                                uint64_t appendTerm,
                                shared_ptr<Marshallable> cmd,
                                uint64_t* resTerm,
                                bool_t* revertServer,
                                bool_t* followerAppendOK,
                                bool_t* decrement,
                                uint64_t* followerLogSize) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  Log_info("RPC Call AppendEntries for leader %lli to server %lli",leaderId,site_id); 
  auto proxies = rpc_par_proxies_[par_id];
  //Log_info("130 running in leader %lli",loc_id_);
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  //Log_info("132 running in leader %lli",loc_id_);
  for (auto& p : proxies) {
    if (p.first == site_id) {
      //Log_info("135 running in leader %lli",loc_id_);
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      //Log_info("138 running in leader %lli",loc_id_);
      fuattr.callback = [resTerm,revertServer,followerAppendOK,followerLogSize,decrement,ev](Future* fu) {
        fu->get_reply() >> *resTerm;
        fu->get_reply() >> *revertServer;
        fu->get_reply() >> *followerAppendOK;
        fu->get_reply() >> *decrement;
        fu->get_reply() >> *followerLogSize;
        ev->Set(1);
      };
      //Log_info("148 running in leader %lli",loc_id_);
      MarshallDeputy md(cmd);
      //Log_info("150 running in leader %lli",loc_id_);
      //Log_info("Before async call of leader %lli to server %lli",leaderId,site_id);
      Call_Async(proxy, AppendEntries, lastLogIndex, lastLogTerm, leaderTerm, leaderId, leaderCommit,appendTerm, md, fuattr);
      //Log_info("After async call of leader %lli to server %lli",leaderId,site_id);
    }
  }
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendHeartbeat(parid_t par_id,
                                siteid_t site_id,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                uint64_t leaderTerm,
                                locid_t leaderId,
                                uint64_t leaderCommit,
                                uint64_t* resTerm,
                                bool_t* revertServer,
                                bool_t* followerAppendOK,
                                bool_t* decrement) {
  
  Log_info("RPC Call SendHeartbeat for leader %lli to server %lli",leaderId,site_id); 
  auto proxies = rpc_par_proxies_[par_id];
  //Log_info("130 running in leader %lli",loc_id_);
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  //Log_info("132 running in leader %lli",loc_id_);
  for (auto& p : proxies) {
    if (p.first == site_id) {
      //Log_info("135 running in leader %lli",loc_id_);
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      //Log_info("138 running in leader %lli",loc_id_);
      fuattr.callback = [resTerm,revertServer,followerAppendOK,decrement,leaderId,ev,p](Future* fu) {
        fu->get_reply() >> *resTerm;
        fu->get_reply() >> *revertServer;
        fu->get_reply() >> *followerAppendOK;
        fu->get_reply() >> *decrement;
        ev->Set(1);
      };
      //Log_info("148 running in leader %lli",loc_id_);
      //Log_info("150 running in leader %lli",loc_id_);
      //Log_info("Before async call of leader %lli to server %lli",leaderId,site_id);
      Call_Async(proxy, Heartbeat, lastLogIndex, lastLogTerm, leaderTerm, leaderId, leaderCommit, fuattr);
      //Log_info("After async call of leader %lli to server %lli",leaderId,site_id);
    }
  }
  return ev;
}

shared_ptr<IntEvent> 


RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  Log_info("RPC Call SendString"); 
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
