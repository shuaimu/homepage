RaftCommo::RaftCommo(PollMgr *poll) : Communicator(poll) {
}

RaftProxy *RaftCommo::get_proxy_for_site(parid_t par_id, siteid_t site_id) {
    for (auto &p: rpc_par_proxies_[par_id]) {
        if (p.first == site_id) return (RaftProxy *) p.second;
    }

    return nullptr; // Should never happen
}

void RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                RequestVoteInput input,
                                const function<void(siteid_t, term_t, uint16_t)> &request_vote_callback) {
    RaftProxy *proxy = get_proxy_for_site(par_id, site_id);
    if (proxy == nullptr) return;



    FutureAttr fuattr;
    fuattr.callback = [site_id, request_vote_callback](Future *fu) {
        /* this is a handler that will be invoked when the RPC returns */
        term_t term;
        uint16_t vote_status;
        /* retrieve RPC return values in order */
        fu->get_reply() >> term;
        fu->get_reply() >> vote_status;
        /* process the RPC response here */

        request_vote_callback(site_id, term, vote_status);
    };
    /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
    * to asynchronously invoke RPCs */
    Call_Async(proxy, RequestVote,
               input.sender_siteid, input.candidate_term, input.last_log_index, input.last_log_term, // params
               fuattr);

    Log_debug("Sent RequestVote to %d from %d. Rpc count: %d", site_id, input.sender_siteid, rpc_count_);
}

void RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  AppendEntriesInput input,
                                  const function<void(siteid_t, term_t, index_t, uint16_t)> &append_entries_callback) {
    RaftProxy *proxy = get_proxy_for_site(par_id, site_id);
    if (proxy == nullptr) return;

    FutureAttr fuattr;
    fuattr.callback = [site_id, append_entries_callback](Future *fu) {
        term_t term;
        index_t last_log_index;
        uint16_t append_status;

        fu->get_reply() >> term;
        fu->get_reply() >> last_log_index;
        fu->get_reply() >> append_status;

        append_entries_callback(site_id, term, last_log_index, append_status);
    };

    /* wrap Marshallable in a MarshallDeputy to send over RPC */

    Call_Async(proxy, AppendEntries,
               input.sender_siteid, input.leader_term, input.prev_log_index, input.prev_log_term, // params
               input.leader_commit, input.entry_terms, input.commands, // params
               fuattr);

    Log_debug("Sent AppendEntries to %d from %d for index %d. Rpc count: %d", site_id, input.sender_siteid,
             input.prev_log_index, rpc_count_);
}

void RaftCommo::SendEmptyAppendEntries(parid_t par_id,
                                       siteid_t site_id,
                                       EmptyAppendEntriesInput input,
                                       const function<void(siteid_t, bool_t, index_t)> &heartbeat_callback) {
    RaftProxy *proxy = get_proxy_for_site(par_id, site_id);
    if (proxy == nullptr) return;



    FutureAttr fuattr;
    fuattr.callback = [site_id, heartbeat_callback](Future *fu) {
        bool_t ack;
        index_t last_log_index;

        fu->get_reply() >> ack;
        fu->get_reply() >> last_log_index;

        heartbeat_callback(site_id, ack, last_log_index);
    };

    Call_Async(proxy, EmptyAppendEntries,
               input.sender_siteid, input.leader_term, input.leader_commit, // params
               fuattr);

    Log_debug("Sent EmptyAppendEntries to %d from %d. Rpc count: %d", site_id, input.sender_siteid, rpc_count_);
}

shared_ptr <IntEvent>
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string &msg, string *res) {
    auto proxies = rpc_par_proxies_[par_id];
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    for (auto &p: proxies) {
        if (p.first == site_id) {
            RaftProxy *proxy = (RaftProxy *) p.second;
            FutureAttr fuattr;
            fuattr.callback = [res, ev](Future *fu) {
                fu->get_reply() >> *res;
                ev->Set(1);
            };
            /* wrap Marshallable in a MarshallDeputy to send over RPC */
            Call_Async(proxy, HelloRpc, msg, fuattr);
        }
    }
    return ev;
}
