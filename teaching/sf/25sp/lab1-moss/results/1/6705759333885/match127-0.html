void RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t arg1,
                                uint64_t arg2,
                                uint64_t arg3,
                                uint64_t arg4,
                                uint64_t *ret1,
                                bool_t *voteGranted) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ret1,voteGranted](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */

        /* retrieve RPC return values in order */
        fu->get_reply() >> *ret1;
        fu->get_reply() >> *voteGranted;
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, arg1, arg2, arg3, arg4, fuattr);
    }
  }
}

void
RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id, // site_id of the server which needs to recieve rpc
                                  uint64_t arg1,
                                  uint64_t arg2, // site_id of the server sending the RPC
                                  uint64_t arg3,
                                  uint64_t arg4,
                                  vector<shared_ptr<Marshallable>> arg5,
                                  uint64_t arg6,
                                  shared_ptr<Marshallable> cmd,
                                  bool_t* followerAppendOk,
                                  uint64_t* copiedTill) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  // PENDING: Check if this needs to be idempotent
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [followerAppendOk,copiedTill](Future* fu) {
        fu->get_reply() >> *followerAppendOk;
        fu->get_reply() >> *copiedTill;
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      MarshallDeputy md(cmd);
      vector<MarshallDeputy> arg5Marsh;
      for(int i=0;i<arg5.size();i++) {
        arg5Marsh.push_back(MarshallDeputy (arg5[i]));
      }
      Call_Async(proxy, AppendEntries,par_id,site_id,arg1,arg2,arg3,arg4,arg5Marsh,arg6,md,fuattr);
    }
  }
  return;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;


      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}

void RaftCommo::SendEmptyAppendEntries(parid_t par_id, siteid_t site_id, 
      uint64_t arg1, uint64_t arg2,uint64_t arg3, 
      const vector<MarshallDeputy>& arg4, string* res, bool_t* success, uint64_t* copiedTill) {
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res, success, copiedTill](Future* fu) {
        fu->get_reply() >> *res;
        fu->get_reply() >> *success;
        fu->get_reply() >> *copiedTill;
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, EmptyAppendEntries, arg1, arg2, arg3, arg4,fuattr);
    }
  }
}
