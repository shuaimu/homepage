shared_ptr<IntEvent>
RaftCommo::SendRequestVote(parid_t par_id,
                                uint64_t currentTerm,
                                uint64_t candidateId,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                uint64_t* maxTerm,
                                uint64_t* votes) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  *maxTerm = currentTerm;
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first != candidateId){
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [maxTerm, votes, ev](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        uint64_t term;
        bool_t vote;
        fu->get_reply() >> term;
        fu->get_reply() >> vote;
        if(term > *maxTerm){
          *maxTerm = term;
          // ev->Set(1);
        } else if(vote){
          *votes += 1;
          if(*votes >= 3 && ev->status_ != Event::TIMEOUT){
            ev->Set(1);
          }
        }
        /* process the RPC response here */
      };
      
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, currentTerm, candidateId, lastLogIndex, lastLogTerm, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent>
RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t leaderTerm,
                                  uint64_t leaderId,
                                  uint64_t prevLogIndex,
                                  uint64_t prevLogTerm,
                                  vector<MarshallDeputy> entries,


                                  vector<uint64_t> logTerms,
                                  uint64_t leaderCommit,
                                  uint64_t* followerLogTerm,
                                  uint64_t* resTerm,
                                  bool_t* followerAppendOK) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev, resTerm, followerAppendOK, leaderId, followerLogTerm, site_id](Future* fu) {
        fu->get_reply() >> *followerLogTerm;


        fu->get_reply() >> *resTerm;
        fu->get_reply() >> *followerAppendOK;
        if(ev->status_!= Event::TIMEOUT){
          ev->Set(1);
        }
      };
      Call_Async(proxy, AppendEntries, leaderTerm, leaderId, prevLogIndex, prevLogTerm, entries, logTerms, leaderCommit, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent>
RaftCommo::EmptyAppendEntries(parid_t par_id,
                              uint64_t leaderTerm,
                              siteid_t leaderId,
                              uint64_t prevLogIndex,
                              uint64_t prevLogTerm,
                              uint64_t leaderCommit,
                              set<uint64_t> backlog,
                              uint64_t* resTerm,
                              bool_t* resOK) {
  /*
   * More example code for sending a single RPC to server at leaderId
   * You may modify and use this function or just use it as a reference
   */
  *resTerm = leaderTerm;
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first != leaderId && backlog.count(p.first) == 0 ) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev, resTerm, resOK](Future* fu) {
        uint64_t term;
        bool_t followerOK;
        fu->get_reply() >> term;
        fu->get_reply() >> followerOK;
        if(ev->status_!= Event::TIMEOUT){
          if(term > *resTerm){
            *resTerm = term;
            ev->Set(1);
          }
          if(*resOK && !followerOK){
            *resOK = followerOK;
            ev->Set(1);
          }
        }
      };
      Call_Async(proxy, EmptyAppendEntries, leaderTerm, leaderId, prevLogIndex, prevLogTerm, leaderCommit, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
