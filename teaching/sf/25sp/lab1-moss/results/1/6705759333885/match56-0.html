void RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t arg1,
                                uint64_t arg2,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                bool* voted_for_glo) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  //   auto ev = Reactor::CreateSpEvent<IntEvent>();
//   cout << "RPC CALL SEND: Server #" << arg2 << " requesting server #" << site_id << " for vote. Term of request: " << arg1 << endl;
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [voted_for_glo, site_id](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        uint64_t ret1;
        uint64_t vote_granted;
        /* retrieve RPC return values in order */
        fu->get_reply() >> ret1;
        fu->get_reply() >> vote_granted;

        // cout << "RETURNED VOTE!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!" << site_id << " VOTE GRANTED: " << vote_granted << " RET1: " << ret1 << " ERROR CODE: " << fu->get_error_code() << endl;

        // cout << "COUT VOTE GRANTED: vote granted: " << vote_granted << endl;
        *voted_for_glo = (vote_granted == 1) ? true : false;

        // *vote_granted_glo = vg;
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
    //   cout << "RPC CALL SEND22: Server #" << arg2 << " requesting server #" << site_id << " for vote. Term of request: " << arg1 << endl;
      Call_Async(proxy, RequestVote, arg1, arg2, lastLogIndex, lastLogTerm, fuattr);

shared_ptr<IntEvent>
RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  shared_ptr<Marshallable> cmd,
                                  uint64_t term,
                                  uint64_t leaderId,
                                  uint64_t prevLogIndex,
                                  uint64_t prevLogTerm,
                                  int64_t logTerm,
                                  uint64_t leaderCommit,
                                  uint64_t* cur_term_ptr,
                                  volatile uint64_t* nextIndexPtr,
                                  std::recursive_mutex* mtx,
                                  roles* role,
                                  std::chrono::high_resolution_clock::time_point* time_since_rec) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;

      fuattr.callback = [cur_term_ptr, ev, nextIndexPtr, site_id, leaderId, mtx, role, time_since_rec](Future* fu) {
        int64_t temp_success;
        uint64_t rtt;
        bool_t success = false;
        fu->get_reply() >> temp_success;
        fu->get_reply() >> rtt;

        /*
            If temp_success = 2 :: true, appended
            If temp_success = 1 :: false, need to send prev request (or check if terms are off)
            If temp_success = 0 :: request failed
            If temp_success = -1 :: heartbeat, ignore
        */
       if(temp_success == 0){
        // cout << "request ignored" << endl;
        return;
       }

       if(temp_success == -1){
            // Heartbeat, ignore
            ev->Set(1);
            if(rtt > *cur_term_ptr){
                *cur_term_ptr = rtt; // Maybe?
                *role = follower; // follower
                *time_since_rec = std::chrono::high_resolution_clock::now();
            }
       }else{
            success = (temp_success == 1) ? false : true;
            ev->Set(1);
            *time_since_rec = std::chrono::high_resolution_clock::now();

            // If term is higher than leader, incremnet and "step down?"
            if(rtt > *cur_term_ptr){
                *cur_term_ptr = rtt; // Maybe?
                *role = follower; // follower
            }

            // If false, decrement nextIndex of this server
            mtx->lock();
            if(!success && *nextIndexPtr > 1){
                // cout << "decrement index " << *nextIndexPtr << " temp_success: " << temp_success << "site_ids " << site_id << " " << leaderId << endl;
                (*nextIndexPtr)--;
            }

            // If true, increment nextIndex of this server
            if(success){
                // cout << "increment index " << *nextIndexPtr << " temp_success: " << temp_success << endl;
                (*nextIndexPtr)++;
            }
            mtx->unlock();
            // cout << "after decrement index " << *nextIndexPtr << endl;
            // cout << "Svr " << site_id << " updated leaders nextIndex of it to " << *nextIndexPtr << ", Got temp success: " << temp_success << ", and received term: " << rtt << endl;
       }        
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      MarshallDeputy md(cmd);
      Call_Async(proxy, AppendEntries, md, term, leaderId, prevLogIndex, prevLogTerm, logTerm, leaderCommit, fuattr);
    }
  }
  return ev;
}
