shared_ptr<IntEvent> RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t candidateTerm,
                                uint64_t candidateId,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                uint64_t* voterTerm,
                                bool_t* voterHealthy,
                                bool_t* voteGranted) {
  // TODO
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [voterTerm, voterHealthy, voteGranted, ev](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        fu->get_reply() >> *voterTerm;
        fu->get_reply() >> *voterHealthy;
        fu->get_reply() >> *voteGranted;
        ev->Set(1);
        /* process the RPC response here */
        // TODO
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, candidateTerm, candidateId, lastLogIndex, lastLogTerm, fuattr);
    }
  }
  return ev;
}



shared_ptr<IntEvent> RaftCommo::SendAppendEntries(parid_t par_id,
                                                  siteid_t site_id,
                                                  // args
                                                  uint64_t leaderTerm,
                                                  uint64_t leaderID,
                                                  uint64_t prevLogIndex,
                                                  uint64_t prevLogTerm,
                                                  std::vector<shared_ptr<Marshallable> > entries,
                                                  std::vector<uint64_t> entryTerms,
                                                  uint64_t leaderCommitIndex,
                                                  // ret_vals
                                                  bool_t* followerHealthy,
                                                  bool_t* followerAppendOK,
                                                  uint64_t* followerTerm) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [followerHealthy, followerAppendOK, followerTerm, ev](Future* fu) {
        fu->get_reply() >> *followerHealthy;
        fu->get_reply() >> *followerAppendOK;
        fu->get_reply() >> *followerTerm;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      std::vector<MarshallDeputy> entries_md;
      for (shared_ptr<Marshallable> entry_ptr : entries) {
        MarshallDeputy entry_md(entry_ptr);
        entries_md.push_back(entry_md);
      }
      Call_Async(proxy, AppendEntries, leaderTerm, leaderID, prevLogIndex, prevLogTerm, entries_md, entryTerms, leaderCommitIndex, fuattr);
    }
  }
  return ev;
}
