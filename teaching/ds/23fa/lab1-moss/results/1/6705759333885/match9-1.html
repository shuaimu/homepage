void RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t term,
                                uint64_t candidate_id,
                                uint64_t prev_log_index,
                                uint64_t prev_log_term,
                                RaftServer* svr) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
        FutureAttr fuattr;
        fuattr.callback = [svr, site_id, term](Future* fu) {
            auto err = fu->get_error_code();
            if(err != 0) {
                Log_info("klog: Error getting vote");
                return;
            }
            uint64_t resp_term;
            bool_t append_ok;
            fu->get_reply() >> resp_term;
            fu->get_reply() >> append_ok;


            svr->on_request_vote_response(term, site_id, resp_term, append_ok);
        };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, term, candidate_id, prev_log_index, prev_log_term, fuattr);
    }
  }
}

shared_ptr<IntEvent>
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}

    void RaftCommo::SendAppendEntries(uint32_t par_id, uint16_t site_id, uint64_t req_term, uint64_t leader_id,
                                      uint64_t prev_log_index, uint64_t prev_log_term, shared_ptr<Marshallable> cmd, uint64_t cmd_term,
                                      uint64_t leader_commit, RaftServer* svr) {

        auto proxies = rpc_par_proxies_[par_id];
        for (auto& p : proxies) {
            if (p.first == site_id) {
                RaftProxy *proxy = (RaftProxy*) p.second;
                FutureAttr fuattr;
                fuattr.callback = [site_id, prev_log_index, svr](Future* fu) {
                    uint64_t resp_term;
                    bool_t success;
                    fu->get_reply() >> resp_term;
                    fu->get_reply() >> success;
                    svr->on_append_entries_response(site_id, prev_log_index, resp_term, success);
                };
                /* wrap Marshallable in a MarshallDeputy to send over RPC */
                MarshallDeputy md{cmd};
                Call_Async(proxy, AppendEntries, req_term, leader_id, prev_log_index, prev_log_term, md, cmd_term, leader_commit, fuattr);
            }
        }
    }
