  shared_ptr<IntEvent>
  RaftCommo::SendRequestVote(parid_t par_id,
                             siteid_t site_id,
                             uint64_t term,
                             uint32_t candidateId,
                             uint64_t lastLogIndex,
                             uint64_t lastLogTerm,
                             uint64_t *curr_Term,
                             bool_t *vote_granted)
  {
    /*
     * Example code for sending a single RPC to server at site_id
     * You may modify and use this function or just use it as a reference
     */
    auto proxies = rpc_par_proxies_[par_id];
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    for (auto &p : proxies)
    {
      if (p.first == site_id)
      {
        RaftProxy *proxy = (RaftProxy *)p.second;
        FutureAttr fuattr;

        fuattr.callback = [curr_Term, vote_granted,ev](Future *fu)
        {
          /* this is a handler that will be invoked when the RPC returns */

          fu->get_reply() >> *curr_Term;
          fu->get_reply() >> *vote_granted;

          // *vote_granted = votesss + *vote_granted;
          // cout << "vote granted " << votesss << endl;
          // cout << "termI: " << termI <<endl;
           Log_info("Is commo getting response yes");
          ev->Set(1);
          /* process the RPC response here */
        };
        /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
         * to asynchronously invoke RPCs */
        Log_info("Is commo getting called yes");
        Call_Async(proxy, RequestVote, term, candidateId, lastLogIndex, lastLogTerm, fuattr);
      }
    }
    return ev;
  }

  shared_ptr<IntEvent>
  //void 
  RaftCommo::SendAppendEntries(parid_t par_id,
                                    siteid_t site_id,
                                    uint64_t term,


                                    uint32_t leaderId,
                                    uint64_t prevLogIndex,
                                    uint64_t prevLogTerm,
                                    uint64_t leaderCommit,
                                    vector<uint64_t> pendingTerms,
                                    vector<shared_ptr<Marshallable>> cmds,
                                    uint64_t* curr_Term,
                                    uint64_t* ack_len,
                                    bool_t* followerAppendOK)
  {
    
    auto proxies = rpc_par_proxies_[par_id];
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    for (auto &p : proxies)
    {
      if (p.first == site_id)
      {
        RaftProxy *proxy = (RaftProxy *)p.second;
        FutureAttr fuattr;
        fuattr.callback = [curr_Term,ack_len ,followerAppendOK,ev](Future *fu)
        {
          // bool_t followerAppendOK;


          fu->get_reply() >> *curr_Term;
          fu->get_reply() >> *ack_len;
          fu->get_reply() >> *followerAppendOK;
          
          ev->Set(1);
        };
        /* wrap Marshallable in a MarshallDeputy to send over RPC */

        // for(auto cmd:cmds){
          
        // }
        vector<MarshallDeputy> depCmds;
        for(auto x: cmds){
          MarshallDeputy md(x);
          depCmds.push_back(md);
        }
        
        Log_debug("Inside commo for append entries : leader %d server %d",leaderId,site_id);
        Call_Async(proxy, AppendEntries,term, leaderId, prevLogIndex, prevLogTerm, leaderCommit, pendingTerms,depCmds, fuattr);
      }
    }
    return ev;
  }

  shared_ptr<IntEvent>
  RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string &msg, string *res)
  {
    auto proxies = rpc_par_proxies_[par_id];
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    for (auto &p : proxies)
    {
      if (p.first == site_id)
      {
        RaftProxy *proxy = (RaftProxy *)p.second;
        FutureAttr fuattr;
        fuattr.callback = [res, ev](Future *fu)
        {
          fu->get_reply() >> *res;
          ev->Set(1);
        };
        /* wrap Marshallable in a MarshallDeputy to send over RPC */
        Call_Async(proxy, HelloRpc, msg, fuattr);
      }
    }
    return ev;
  }
