void RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t candidateTerm,
                                locid_t candidateId,
                                uint64_t lastLogIndex,
                                uint64_t lastLogTerm,
                                const function<void(uint64_t, bool_t)>& callback) {
    /*
     * Example code for sending a single RPC to server at site_id
     * You may modify and use this function or just use it as a reference
     */
    auto proxies = rpc_par_proxies_[par_id];

    for (auto& p : proxies) {


        if (p.first != site_id) {  // provided site_id is sender's ID
            // Log_info("[server %d] Making RPC call to server %d", site_id, p.first);
            RaftProxy* proxy = (RaftProxy*)p.second;
            FutureAttr fuattr;
            fuattr.callback = [callback](Future* fu) {
                /* this is a handler that will be invoked when the RPC returns */
                /* retrieve RPC return values in order */
                uint64_t term;
                bool_t voteGranted;

                fu->get_reply() >> term;
                fu->get_reply() >> voteGranted;
                /* process the RPC response here */
                // Log_info("term %d voteGranted %d", term, voteGranted);
                callback(term, voteGranted);
            };
            /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
             * to asynchronously invoke RPCs */
            // Log_info("sending RequestVote RPC");
            Call_Async(proxy, RequestVote, candidateTerm, candidateId, lastLogIndex, lastLogTerm, fuattr);
        }
    }
}

void RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t leaderTerm,
                                  locid_t leaderId,
                                  uint64_t prevLogIndex,
                                  uint64_t prevLogTerm,
                                  vector<LogEntry> entries,
                                  uint64_t leaderCommit,
                                  const function<void(uint64_t, bool_t, uint64_t, uint64_t)>& callback) {
    /*
     * More example code for sending a single RPC to server at site_id
     * You may modify and use this function or just use it as a reference
     */
    auto proxies = rpc_par_proxies_[par_id];
    for (auto& p : proxies) {
        if (p.first == site_id) {
            RaftProxy* proxy = (RaftProxy*)p.second;
            FutureAttr fuattr;
            fuattr.callback = [callback](Future* fu) {
                uint64_t term;
                bool_t success;
                uint64_t conflictingIndex;
                uint64_t conflictingTerm;
                fu->get_reply() >> term;
                fu->get_reply() >> success;
                fu->get_reply() >> conflictingIndex;
                fu->get_reply() >> conflictingTerm;
                callback(term, success, conflictingIndex, conflictingTerm);
            };

            vector<uint64_t> entriesTerms;
            vector<MarshallDeputy> entriesCommands;
            for (auto entry : entries) {
                entriesTerms.push_back(entry.term);
                /* wrap Marshallable in a MarshallDeputy to send over RPC */
                entriesCommands.push_back(MarshallDeputy(entry.cmd));
            }
            Call_Async(proxy, AppendEntries, leaderTerm, leaderId, prevLogIndex, prevLogTerm, entriesTerms, entriesCommands, leaderCommit, fuattr);
        }
    }
}

void RaftCommo::SendEmptyAppendEntries(parid_t par_id,
                                       siteid_t site_id,
                                       uint64_t leaderTerm,
                                       locid_t leaderId,
                                       const function<void(uint64_t)>& callback) {
    auto proxies = rpc_par_proxies_[par_id];

    for (auto& p : proxies) {
        if (p.first != site_id) {  // provided site_id is sender's ID
            // Log_info("[server %d] Making EmptyAppendEntries RPC call to server %d", site_id, p.first);
            RaftProxy* proxy = (RaftProxy*)p.second;
            FutureAttr fuattr;
            fuattr.callback = [callback](Future* fu) {
                /* this is a handler that will be invoked when the RPC returns */
                uint64_t term;
                /* retrieve RPC return values in order */
                fu->get_reply() >> term;
                callback(term);
            };
            // Log_info("Sending heartbeat to %d", p.first);
            Call_Async(proxy, EmptyAppendEntries, leaderTerm, leaderId, fuattr);
        }
    }
}

shared_ptr<IntEvent>
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
    auto proxies = rpc_par_proxies_[par_id];
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    for (auto& p : proxies) {
        if (p.first == site_id) {
            RaftProxy* proxy = (RaftProxy*)p.second;
            FutureAttr fuattr;
            fuattr.callback = [res, ev](Future* fu) {
                fu->get_reply() >> *res;
                ev->Set(1);
            };
            /* wrap Marshallable in a MarshallDeputy to send over RPC */
            Call_Async(proxy, HelloRpc, msg, fuattr);
        }
    }
    return ev;
}
