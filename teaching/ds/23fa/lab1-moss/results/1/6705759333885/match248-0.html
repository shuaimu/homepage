shared_ptr<IntEvent>
RaftCommo::SendRequestVote(void *proxy_,
                                int64_t term,
                                int64_t candidateId,
                                int64_t lastLogIndex,
                                int64_t lastLogTerm,
                                int64_t *rsp_term,
                                int64_t *rsp_vote_granted) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
    RaftProxy *proxy = (RaftProxy *)proxy_;
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    FutureAttr fuattr;


    fuattr.callback = [rsp_term, rsp_vote_granted, ev](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        int64_t r_term;
        bool_t vote_granted;
        /* retrieve RPC return values in order */
        fu->get_reply() >> r_term;
        fu->get_reply() >> vote_granted;
        *rsp_term = r_term;
        *rsp_vote_granted = vote_granted;
        ev->Set(1);
        /* process the RPC response here */
    };

    /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
     * to asynchronously invoke RPCs */
    Call_Async(proxy, RequestVote, term, candidateId, lastLogIndex, lastLogTerm, fuattr);
    return ev;
}

shared_ptr<IntEvent>
RaftCommo::SendAppendEntries(void *proxy_,
                             int64_t term,
                             int64_t leader_id,
                             int64_t prev_log_index,
                             int64_t prev_log_term,
                             int64_t leader_commit,
                             vector <MarshallDeputy> mds,
                             vector <int64_t> terms,
                             int64_t *rsp_term,
                             int64_t *success,
                             int64_t *conflict_index,
                             int64_t *conflict_term
                             /*shared_ptr<Marshallable> cmd*/) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
    RaftProxy *proxy = (RaftProxy*) proxy_;
    auto ev = Reactor::CreateSpEvent<IntEvent>();
    if (mds.size() != 0) {
        FutureAttr fuattr;
        fuattr.callback = [ev, rsp_term, success, conflict_index, conflict_term](Future* fu) {
            int64_t r_term;
            fu->get_reply() >> r_term;
            *rsp_term = r_term;
            int64_t suc;
            fu->get_reply() >> suc;
            *success = suc;
            int64_t conflict_idx;
            fu->get_reply() >> conflict_idx;
            *conflict_index = conflict_idx;
            int64_t conflict_trm;
            fu->get_reply() >> conflict_trm;
            *conflict_term = conflict_trm;
            ev->Set(1);
        };
        /* wrap Marshallable in a MarshallDeputy to send over RPC */
        // MarshallDeputy md(cmd);
        Log_debug("[Server %d] Calling send append entries during term %d", leader_id, term);
        Call_Async(proxy, AppendEntries, term, leader_id, prev_log_index, prev_log_term,
                                     leader_commit, mds, terms, fuattr);
    } else {
        *success = 1;
        *conflict_index = -1;
        *conflict_term = 0;
        FutureAttr fuattr;
        fuattr.callback = [ev, rsp_term](Future* fu) {
            int64_t r_term;
            fu->get_reply() >> r_term;
            *rsp_term = r_term;
            ev->Set(1);
        };
        Log_debug("[Server %d] Calling send empty append entries during term %d", leader_id, term);
        Call_Async(proxy, EmptyAppendEntries, term, leader_id, prev_log_index, prev_log_term,
                                    leader_commit, fuattr);
    }
    return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      //Log_debug("sending TPC");
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res, ev](Future* fu) {
        //Log_debug("got reply");
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
        //Log_debug("Calling Async");
        Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
