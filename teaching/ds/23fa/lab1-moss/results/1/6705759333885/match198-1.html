void RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t* term,
                                uint64_t candidateId,
                                uint64_t last_log_term,
                                uint64_t last_log_index,
                                bool_t* is_leader,
                                uint64_t* number_of_votes,


                                bool_t* vote_granted,
                                uint64_t* ret1,
                                uint64_t* votedFor,
                                shared_ptr<IntEvent> ev) {
  //Log_info("\ncommo.cc SendRequestVote Called");  
  //Log_info("term is : %d",term);                          
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  //uint64_t number_of_votes = 0;
  //Log_info("Inside SEND REQUEST VOTE: *number_of_votes: %d", *number_of_votes);

  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      //Variables in Lambda passed by value
      fuattr.callback = [ev, ret1, vote_granted, number_of_votes, is_leader, term, votedFor, candidateId] (Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        //Log_info("Inside SEND REQUEST VOTE: *number_of_votes: %d", *number_of_votes);
        fu->get_reply() >> *ret1;
        fu->get_reply() >> *vote_granted;
        //Log_info("\n1\n");
        /* process the RPC response here */
        if(*vote_granted)
        {
          *number_of_votes = *number_of_votes + 1;
          //Log_info("\nVote granted is true, number_of_votes: %d\n", *number_of_votes);
        }
      
        if(*number_of_votes >= 3)
        {
          //Log_info("\n\n number of votes >= 3 for server%d - selected as leader", candidateId);
          *is_leader = true;
          ev->Set(1);
          //Log_info("Leader is going to be set");
        }
        // Log_info("Ret1: %d Term number: %d",*ret1,term);
        //Log_info("3. Old term is %d\n",*term);
        if(*ret1 > *term)
        {
          //Log_info("\nLeader reset as, %d no longer leader , returned term is %d \n",candidateId, *ret1);
          *votedFor = -1;
          *term = *ret1;
          *is_leader = false;
        }
        //Log_info("4. New term: %d\n",*term);
        //Log_info("In Send RV, set to %d", *term);
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      //Log_info("\nQWERTYUIOP\n");
      uint64_t t = *term;
      Call_Async(proxy, RequestVote, last_log_index, last_log_term, t, candidateId, fuattr);
      //Log_info("\n12345QWERTYUIOP\n");
    }
  }
}

  void RaftCommo::SendHeartbeat(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t& candidateId,
                                  uint64_t* term,
                                  bool_t* is_leader,
                                  uint64_t* votedFor,
                                  bool_t* timeout,
                                  uint64_t& last_applied,
                                  uint64_t* ret1
) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ret1, term, is_leader, votedFor, timeout](Future* fu) {
        fu->get_reply() >> *ret1;
        if(*ret1 > *term)
        {
          *votedFor = -1;
          *term = *ret1;
          *is_leader = false;
          *timeout = false;
        }
      };
      uint64_t t = *term;
      Call_Async(proxy, Heartbeat, t, candidateId, last_applied, fuattr);
    }
  }
}

void RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,
                                  uint64_t& prev_log_index,
                                  uint64_t& prev_log_term,
                                  uint64_t* next_index,
                                  uint64_t* term,
                                  bool_t* recieve_append_ok,
                                  uint64_t* number_append_ok,
                                  bool_t* consensus,
                                  uint64_t last_applied,
                                  uint64_t* follower_next_index,
                                  uint64_t& corresponding_term,
                                  bool_t* is_leader,
                                  uint64_t* votedFor,
                                  bool_t* timeout,
                                  uint64_t* ret1,
                                  vector<MarshallDeputy>& cmd,
                                  vector<uint64_t> & v_term) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ret1, term, is_leader, votedFor, timeout, follower_next_index, site_id,number_append_ok, consensus, recieve_append_ok, next_index](Future* fu) {
        bool_t followerAppendOK = false;
        fu->get_reply() >> followerAppendOK;
        fu->get_reply() >> *follower_next_index;
        fu->get_reply() >> *ret1;
        if(followerAppendOK)
        {
          (*number_append_ok)++;
          *recieve_append_ok = true;
        }
        if(*follower_next_index!=0)
        {
          *next_index = *follower_next_index;
          Log_info("Setting follower new index as: %d *follower_next_index: %d", *next_index, *follower_next_index);
        }
        else{
          Log_info("No connection  with server %d, returned 0", site_id);
        }
        if(*number_append_ok>=3)
        {
          *consensus = true;
        }
        if(*ret1 > *term)
        {
          *votedFor = -1;
          *term = *ret1;
          Log_info("term set to %d", *term);
          *is_leader = false;
          *timeout = false;
        }
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      //MarshallDeputy md(cmd);
      uint64_t t = *term;


      Call_Async(proxy, AppendEntries, prev_log_index, prev_log_term, t, last_applied, corresponding_term, cmd, v_term, fuattr);
    }
  }
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      Log_info("Commo.cc - SendString");
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
