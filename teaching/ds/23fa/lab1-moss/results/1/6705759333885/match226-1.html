}

shared_ptr<IntEvent>
RaftCommo::SendRequestVote(parid_t par_id,
                           siteid_t site_id,
                           uint64_t candidateTerm,
                           uint32_t candidateId,
                           uint64_t lastLogIndex,
                           uint64_t lastLogTerm,
                           uint64_t* term,
                           bool_t* voteGranted) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [term, voteGranted, ev](Future* fu) {
        fu->get_reply() >> *term >> *voteGranted;
        ev->Set(1);
      };
      Call_Async(proxy, RequestVote, candidateTerm, candidateId, lastLogIndex, lastLogTerm, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent>


RaftCommo::SendAppendEntries(parid_t par_id,
                             siteid_t site_id,
                             uint64_t leaderTerm,
                             uint32_t leaderId,
                             uint64_t prevLogIndex,
                             uint64_t prevLogTerm,
                             uint64_t leaderCommit,
                             uint64_t logTerm,
                             shared_ptr<Marshallable> cmd,
                             uint64_t *term,
                             bool_t *followerAppendOK) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto &p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy *)p.second;
      FutureAttr fuattr;
      fuattr.callback = [term, followerAppendOK, ev](Future *fu) {
        fu->get_reply() >> *term >> *followerAppendOK;
        ev->Set(1);
      };
      if (cmd == nullptr) {
        Call_Async(proxy, AppendEmptyEntry, leaderTerm, leaderId, prevLogIndex, prevLogTerm, leaderCommit, fuattr);
      } else {


        MarshallDeputy md{cmd};
        Call_Async(proxy, AppendEntries, leaderTerm, leaderId, prevLogIndex, prevLogTerm, leaderCommit, logTerm, md, fuattr);
      }
    }
  }
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
