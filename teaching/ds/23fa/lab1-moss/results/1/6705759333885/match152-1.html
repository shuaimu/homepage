shared_ptr<IntEvent> 
RaftCommo::SendRequestVote(parid_t par_id,
                          siteid_t site_id,
                          uint64_t currentTerm,
                          uint64_t CandidateId,
                          uint64_t candidateLogLen,
                          uint64_t candidateLogTerm,
                          uint64_t* peerTerm,
                          bool_t* vote_granted) {
/*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  //Log_info("In commo for SendRequestVote");
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [peerTerm,vote_granted, ev](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        /* retrieve RPC return values in order */
        fu->get_reply() >> *peerTerm;
        fu->get_reply() >> *vote_granted;
        ev->Set(1);
        //Log_info("In callback");
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      //Log_info("In commo for SendRequestVote, %d to %d",CandidateId,site_id);
      //Log_info("Making async call");
      Call_Async(proxy, RequestVote, currentTerm, CandidateId, candidateLogLen, candidateLogTerm, fuattr);
      //Log_info("Last line of send RequestVote commo");
      //Log_info("In commo call async done");
    }
  }
  return ev;
}

shared_ptr<IntEvent>
RaftCommo::SendAppendEntries(parid_t par_id,
                             siteid_t site_id,


                             uint64_t currentTerm,
                             uint64_t leaderId,
                             uint64_t prevLogIndex,
                             uint64_t prevLogTerm,
                             uint64_t commitLen,
                             vector<pair<uint64_t,shared_ptr<Marshallable>>> entries,
                             uint64_t* peerTerm,
                             uint64_t* ackLen,
                             bool_t* followerAppendOK) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  //Log_info("In commo for AppendEntries");
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [peerTerm, ackLen, followerAppendOK, ev](Future* fu) {


        fu->get_reply() >> *peerTerm;
        fu->get_reply() >> *ackLen;
        fu->get_reply() >> *followerAppendOK;
        ev->Set(1);
        //Log_info("In callback");
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      //MarshallDeputy md(cmd);
      vector<uint64_t>logTerms;
      vector<MarshallDeputy>logCommands;
      for(auto ent:entries){
        logTerms.push_back(ent.first);
        MarshallDeputy command(ent.second);
        logCommands.push_back(command);
      }
      
      //Log_info("Making async call");
      Call_Async(proxy, AppendEntries, currentTerm, leaderId, prevLogIndex, prevLogTerm, commitLen, logTerms, logCommands, fuattr);
      //Log_info("Last line of send AppendEntries commo");
    }
  }
  return ev;
}
