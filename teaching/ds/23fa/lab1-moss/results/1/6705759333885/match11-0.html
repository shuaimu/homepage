shared_ptr<IntEvent> RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                const uint64_t currTerm,
                                const uint64_t candidateId,
                                const uint64_t lastLogIndex,
                                const uint64_t lastLogTerm,
                                uint64_t* updatedTerm,
                                bool_t* voteResponse)
                                 {
  /* 
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [updatedTerm,voteResponse,ev,candidateId,site_id](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        uint64_t ret1;
        bool_t vote_granted;
        /* retrieve RPC return values in order */
        fu->get_reply() >> *updatedTerm;
        fu->get_reply() >> *voteResponse;
        
        ev->Set(2);
        /* process the RPC response here */
        if(*voteResponse==true){
          Log_info("%d :Received a vote %d", candidateId, site_id);
        }
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, currTerm, candidateId, lastLogIndex, lastLogTerm, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent> RaftCommo::SendAppendEntries(parid_t par_id,
                         siteid_t site_id,
                         const uint64_t currTerm,
                         const uint64_t candidateId,
                         const uint64_t lastLogIndex,
                         const uint64_t lastLogTerm,
                         const uint64_t leaderIndex,
                         const vector<uint64_t> logTerms,
                         const vector<shared_ptr<Marshallable>>& cmd,
                         uint64_t* updatedTerm,
                         bool_t* success) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev, updatedTerm,success](Future* fu) {
        bool_t followerAppendOK;
        fu->get_reply() >> *updatedTerm;
        fu->get_reply() >> *success;
        ev->Set(2);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      vector<MarshallDeputy> md;
      vector<uint64_t> terms;

      for(int i=0;i<cmd.size();i++){
        md.push_back(MarshallDeputy(cmd[i]));
        terms.push_back(logTerms[i]);
      }
      Call_Async(proxy, AppendEntries, currTerm, candidateId, lastLogIndex, lastLogTerm, leaderIndex, terms, md, fuattr);
    }
  }
  return ev;
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
