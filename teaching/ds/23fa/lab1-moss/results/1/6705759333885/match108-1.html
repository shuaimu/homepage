shared_ptr<IntEvent>
RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                const uint64_t& term,
                                const uint64_t& candidateId,
                                const uint64_t& lastLogIndex,
                                const uint64_t& lastLogTerm,
                                uint64_t* currentTerm,
                                bool_t* voteGranted) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev, currentTerm, voteGranted, candidateId](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        // int currentTerm;
        // bool vote_granted;
        /* retrieve RPC return values in order */
        fu->get_reply() >> *currentTerm;
        fu->get_reply() >> *voteGranted;
        ev->Set(1);
        //returnTuple = make_tuple(ev, *currentTerm, *voteGranted);
        /* process the RPC response here */
      };
      /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
      * to asynchronously invoke RPCs */
      Call_Async(proxy, RequestVote, term, candidateId, lastLogIndex, lastLogTerm, fuattr);
    }
  }
  return ev;
}

 shared_ptr<IntEvent>
 RaftCommo::SendHeartbeat(parid_t par_id, 
                          siteid_t site_id, 
                          const uint64_t& term, 
                          const uint64_t& leaderId,
                          const uint64_t& prevLogIndex,
                          const uint64_t& prevLogTerm, 
                          const uint64_t& leaderCommitIndex,
                          uint64_t* currentTerm, 
                          bool_t* followerAppendOK) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev, currentTerm, followerAppendOK](Future* fu) {
        fu->get_reply() >> *followerAppendOK;
        fu->get_reply() >> *currentTerm;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      // MarshallDeputy md(cmd);
      Call_Async(proxy, Heartbeat, term, leaderId, prevLogIndex, prevLogTerm, leaderCommitIndex, fuattr);
    }
  }
  return ev;
}



  void RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id,


                                  const uint64_t& term,
                                  const uint64_t& leaderId,
                                  const uint64_t& prevLogIndex,
                                  const uint64_t& prevLogTerm,
                                  const vector<shared_ptr<Marshallable>>& cmd,
                                  const uint64_t& leaderCommitIndex,
                                  uint64_t* currentTerm,
                                  bool_t* success) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  auto event = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [event, currentTerm, success](Future* fu) {
        fu->get_reply() >> *success;
        fu->get_reply() >> *currentTerm;
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      vector<MarshallDeputy> marshallDeputy = {};
      for (shared_ptr<Marshallable> cmd : cmd) {
        MarshallDeputy md(cmd);
        marshallDeputy.push_back(md);
      }
      Call_Async(proxy, AppendEntries, marshallDeputy, term, leaderId, prevLogIndex, prevLogTerm, leaderCommitIndex, fuattr);
    }
  }
  // event->Wait(100000);
   //event->Set(1);
}

shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, bool_t* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
