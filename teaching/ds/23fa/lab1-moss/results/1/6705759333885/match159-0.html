SiteProxyPair RaftCommo::FindProxy(parid_t par_id, siteid_t site_id) {
  auto proxies = rpc_par_proxies_[par_id];
  for (auto& p : proxies) {
      if (p.first == site_id) {
        return p;
      }
  }
  return pair<uint16_t, ClassicProxy *>(-1, nullptr);
}


vector<shared_ptr<BoxEvent<pair<bool_t, uint64_t>>>> RaftCommo::SendRequestVoteAll(parid_t par_id, 
                                                          locid_t src, 
                                                          uint64_t candidateTerm,
                                                          uint64_t candidateId,
                                                          uint64_t lastLogIndex, 
                                                          uint64_t lastLogTerm) {                                                      
  auto proxies = rpc_par_proxies_[par_id];
  vector<shared_ptr<BoxEvent<pair<bool_t, uint64_t>>>> events;
  for (auto& p : proxies) {
      if (p.first == src) {
        continue;
      }
      auto ev = Reactor::CreateSpEvent<BoxEvent<pair<bool_t, uint64_t>>>();


      events.push_back(ev);
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [ev](Future* fu) {
        uint64_t retTerm;
        bool_t vote_granted;
        fu->get_reply() >> retTerm;
        fu->get_reply() >> vote_granted;
        /* process the RPC response here */
        
        ev->Set(std::pair<uint64_t,uint64_t>(vote_granted, retTerm));
      };
      Call_Async(proxy, RequestVote, candidateTerm, candidateId, lastLogIndex, lastLogTerm, fuattr);
  }
  return events;
}

shared_ptr<BoxEvent<vector<uint64_t>>> RaftCommo::SendAppendEntries(parid_t par_id,
                                  locid_t src,
                                  siteid_t site_id,
                                  vector<MarshallDeputy> &cmds,
                                  vector<uint64_t> &logEntryTerms,
                                  uint64_t leaderTerm,
                                  uint64_t prevLogIndex,
                                  uint64_t prevLogTerm,
                                  uint64_t leaderCommitIndex,
                                  uint64_t fastBackupEnabled
                                  ) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */

  auto p = FindProxy(par_id, site_id);
  int64_t i = p.first;
  RaftProxy *proxy = (RaftProxy *)p.second;
  auto ev = Reactor::CreateSpEvent<BoxEvent<vector<uint64_t>>>();
  if (proxy == nullptr) {
    // Log_debug("Unable to find server %d for AppendEntries", site_id);
    return ev;
  }

  FutureAttr fuattr;
  fuattr.callback = [ev](Future* fu) {
    uint64_t retTerm;
    bool_t followerAppendOK;
    uint64_t idx;
    uint64_t conflictingTerm;
    uint64_t setNext;
        /* retrieve RPC return values in order */
        //Log_info("Returned from RPC for append entry");
    fu->get_reply() >> retTerm;
    fu->get_reply() >> followerAppendOK;
    fu->get_reply() >> idx;
    fu->get_reply() >> conflictingTerm;
    fu->get_reply() >> setNext;

    vector<uint64_t> res({0, idx, conflictingTerm, retTerm, setNext});
    if (followerAppendOK) {
      res.at(0) = 1;
    }
 
    ev->Set(res);
  };
  /* wrap Marshallable in a MarshallDeputy to send over RPC */
  // MarshallDeputy md(cmd);
  Call_Async(proxy, AppendEntries, cmds, logEntryTerms, leaderTerm, src, prevLogIndex, prevLogTerm, leaderCommitIndex, fastBackupEnabled, fuattr);

  return ev;
}

vector<shared_ptr<BoxEvent<pair<bool_t, uint64_t>>>> RaftCommo::SendHeartbeatAll(parid_t par_id, 
                                                          locid_t src,
                                                          uint64_t leaderTerm,
                                                          uint64_t prevLogIndex,
                                                          uint64_t prevLogTerm,
                                                          uint64_t leaderCommitIndex 
                                                          ) {
  // fprintf(stderr, "Leader with id %d: Sending sending append entries to all\n", src);                                                          
  auto proxies = rpc_par_proxies_[par_id];
  vector<shared_ptr<BoxEvent<pair<bool_t, uint64_t>>>> events;

  for (auto& p : proxies) {
    if (src == p.first) {
      continue;
    }
      auto ev = Reactor::CreateSpEvent<BoxEvent<pair<bool_t, uint64_t>>>();


      events.push_back(ev);

      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;

      fuattr.callback = [ev](Future* fu) {
        uint64_t retTerm;
        bool_t followerAppendOK;
        fu->get_reply() >> retTerm;
        fu->get_reply() >> followerAppendOK;

        ev->Set(std::pair<bool_t, uint64_t>(followerAppendOK, retTerm));
      };


      Call_Async(proxy, EmptyAppendEntries, leaderTerm, src, prevLogIndex, prevLogTerm, leaderCommitIndex, fuattr);
    }
  return events;
}



shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
