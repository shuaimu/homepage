shared_ptr<IntEvent>
RaftCommo::SendRequestVote(parid_t par_id,
                                siteid_t site_id,
                                uint64_t arg1,
                                uint64_t arg2, uint64_t arg3, uint64_t arg4, int *res, bool *votedg, bool *sendRPC) {
  /*
   * Example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  auto proxies = rpc_par_proxies_[par_id];
  //debugging comment-1&2 test
  //Log_info("Creating return event for server id %d with term number %d",arg2,arg1);
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  ev->target_=-1;
  for (auto& p : proxies) {
    //Log_info("Sending RPC from %d to %d",site_id,p.first);
    if(p.first == site_id) {


      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,site_id,proxies,ev,votedg,sendRPC](Future* fu) {
        /* this is a handler that will be invoked when the RPC returns */
        uint64_t ret1;
        bool_t vote_granted;
        bool_t gotRPC;
        /* retrieve RPC return values in order */
        fu->get_reply() >> ret1;
        fu->get_reply() >> vote_granted;
        fu->get_reply() >> gotRPC;
        /* process the RPC response here */
        *sendRPC=gotRPC;
        if(vote_granted) {
          *res=ret1;
          *votedg=vote_granted;
          ev->Set(1);
          //debugging comment-1&2 test
          //Log_info("votes for %d are %d",site_id,ev->value_);
        }
        else {
          *res=ret1;
          *votedg=vote_granted;
          ev->Set(-1);
        }
        
        // if((ev->value_)>(proxies.size()/2)) {
        //   //debugging comment-1&2 test
        //   //Log_info("*****%d******",proxies.size());
        //   ev->target_=ret1;
        //   //debugging comment-1&2 test
        //   //Log_info("******counting votes %d********* for %d      %d", ev->value_,site_id,ev->target_);
        // }
    };
  
    /* Always use Call_Async(proxy, RPC name, RPC args..., fuattr)
    * to asynchronously invoke RPCs */
    Call_Async(proxy, RequestVote, arg1, arg2, arg3, arg4, fuattr);
  }
}
  
  return ev;
}

shared_ptr<IntEvent>
RaftCommo::SendAppendEntries(parid_t par_id,
                                  siteid_t site_id, uint64_t arg1, uint64_t arg2, int *res) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */

  //debugging comment
  //Log_info("Entered SendAppendEntires for %d",site_id);
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  ev->target_=-1;
  for (auto& p : proxies) {
    if (p.first == site_id) {
      continue;
    }
    RaftProxy *proxy = (RaftProxy*) p.second;
    FutureAttr fuattr;
    fuattr.callback = [res,ev](Future* fu) {
      uint64_t ret1;
      bool_t followerAppendOK;
      fu->get_reply() >> followerAppendOK;
      fu->get_reply() >> ret1;
      //Log_info("the follower said %d and term is %d",followerAppendOK,ret1);
      if(!followerAppendOK) {
        *res = ret1;
      }
      else {
        *res = ret1;
      }

    };
    /* wrap Marshallable in a MarshallDeputy to send over RPC */
    //debugging comment-1&2 test
    //Log_info("marshalling deputy");
    //MarshallDeputy md(cmd);
    //debugging comment-1&2 test
    //Log_info("Calling the handler");
    Call_Async(proxy, AppendEntries, arg1, arg2, fuattr); 
  }

  return ev;
}
shared_ptr<IntEvent>
RaftCommo::SendAppendEntriesLog(parid_t par_id,
                                  siteid_t site_id, uint64_t arg1, uint64_t arg2, uint64_t arg3, uint64_t arg4, uint64_t arg5,vector<uint64_t> terms, vector<shared_ptr<Marshallable>> cmds,int *res,int *term_g,int *Index_g, bool *response, bool *sendAPE) {
  /*
   * More example code for sending a single RPC to server at site_id
   * You may modify and use this function or just use it as a reference
   */
  //Log_info("Entered SendAppendEntiresLog for %d",arg2);
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {


      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [term_g,Index_g,res,ev,site_id,response,sendAPE](Future* fu) {
        uint64_t ret1;
        uint64_t Xterm;
        uint64_t XIndex;
        bool_t followerAppendOK;
        bool_t gotAPE;
        fu->get_reply() >> followerAppendOK;
        fu->get_reply() >> ret1;
        fu->get_reply() >> Xterm;
        fu->get_reply() >> XIndex;
        fu->get_reply() >> gotAPE;
        //Log_info("coming inside callback for %d",site_id);
        *sendAPE=gotAPE;
        if(followerAppendOK) {
          //Log_info("setting event value_ to 1");
          *term_g=Xterm;
          *Index_g=XIndex;
          *res=ret1;
          *response=followerAppendOK;
          ev->Set(1);
        }
        else {
          Log_info("setting event value_ to -1 %d",ret1);
          *term_g=Xterm;
          *Index_g=XIndex;
          *res=ret1;
          *response=followerAppendOK;
          ev->Set(-1);
        }
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      //debugging comment-1&2 test
      //Log_info("marshalling deputy");
      
      vector<MarshallDeputy> mds;
      for(int i=0;i<cmds.size();i++) {
        MarshallDeputy md(cmds[i]);
        mds.push_back(md);
      }
      //debugging comment-1&2 test
      //Log_info("Calling the handler");
      Call_Async(proxy, AppendEntriesLog, arg1, arg2, arg3, arg4, arg5, terms, mds, fuattr);      
    }
  }
  return ev;
}


shared_ptr<IntEvent> 
RaftCommo::SendString(parid_t par_id, siteid_t site_id, const string& msg, string* res) {
  auto proxies = rpc_par_proxies_[par_id];
  auto ev = Reactor::CreateSpEvent<IntEvent>();
  for (auto& p : proxies) {
    if (p.first == site_id) {
      RaftProxy *proxy = (RaftProxy*) p.second;
      FutureAttr fuattr;
      fuattr.callback = [res,ev](Future* fu) {
        fu->get_reply() >> *res;
        ev->Set(1);
      };
      /* wrap Marshallable in a MarshallDeputy to send over RPC */
      Call_Async(proxy, HelloRpc, msg, fuattr);
    }
  }
  return ev;
}
