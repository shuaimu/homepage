int64_t ShardKvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;
  ret = ret << 32;
  ret = ret + op_id_cnt_++; 
  return ret;
}

int ShardKvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  auto m = dynamic_pointer_cast<Marshallable>(s);
  bool_t success = -1;
  auto ev = Reactor::CreateSpEvent<IntEvent>();


  events[to_string(oid)] = ev;
  Coroutine::CreateRun([this, &m, ev, &success]() {
    uint64_t index, term;
    success = this->GetRaftServer().Start(m, &index, &term);
    if (success == false) ev->Set(1);
  });
  ev->Wait(2000000);
  if (success == true && ev->status_ == Event::DONE) {
    db[k] = v;
  } else if (success == false) {
    return KV_NOTLEADER;
  } else if (ev->status_ != Event::DONE) {
    return KV_TIMEOUT;
  }
  fprintf(stderr, "%d    %d\n", success, ev->status_);
  return KV_SUCCESS;
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  fprintf(stderr, "%u %d %s %d\n", this->GetRaftServer().loc_id_, v->data_.size(), v->data_[0], events.size());
  
  if (events.count(v->data_[0]) == 1) events[v->data_[0]]->Set(1);
  else {
    if (v->data_[1] == "put") {
      db[v->data_[2]] = v->data_[3];
    } else if (v->data_[1] == "append") {
      db[v->data_[2]] += v->data_[3];
    }
  }
}

shared_ptr<ShardKvClient> ShardKvServer::CreateClient(Communicator* comm) {
  auto cli = make_shared<ShardKvClient>();
  cli->commo_ = comm;
  verify(cli->commo_ != nullptr);
  static uint32_t id = 0;
  id++;
  cli->cli_id_ = id; 
  return cli;
}
