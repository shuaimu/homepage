int ShardKvClient::Op(function<int(uint32_t*)> func) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);


    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      leader_idx_ = (leader_idx_+1);
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      leader_idx_ = (leader_idx_+1);
    }
  }

}

int ShardKvClient::Put(const string& k, const string& v) {
  uint32_t shardId = Key2Shard(k);
  auto masterCli = make_shared<ShardMasterClient>();
  masterCli->commo_ = commo_;
  ShardConfig latestConfig;
  masterCli->Query(-1, &latestConfig);
  auto shardMap = latestConfig.shard_group_map_;
  auto gid = shardMap[shardId];
  auto gidServerMap = latestConfig.group_servers_map_;
  leader_idx_ = gidServerMap[gid][0];
  Log_info("CLIENT PUT OPERATION => GID %d shardId %d Key %s Value %s trying server %d as leader", gid, shardId, k.c_str(), v.c_str(), leader_idx_);
  return Op([&](uint32_t* r)->int{
    return Proxy(leader_idx_).Put(GetNextOpId(), k, v, r);
  });
}

ShardKvProxy& ShardKvClient::Proxy(siteid_t site_id) {
  verify(commo_);
  auto p = (ShardKvProxy*)commo_->rpc_proxies_.at(site_id);
  return *p; 
}
