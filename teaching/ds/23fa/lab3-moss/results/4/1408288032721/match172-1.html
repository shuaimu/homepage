int ShardKvClient::Op(function<int(uint32_t*)> func) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    //Log_info("Before calling function for leader_idx_ %lli in Op",leader_idx_);
    r1 = func(&ret);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      leader_idx_ = (leader_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      //Log_info("Before updating leader_idx_ in Op:%lli",leader_idx_);


      leader_idx_ = (leader_idx_+1) % 5;
      //Log_info("After updating leader_idx_ in Op:%lli",leader_idx_);

    }
  }

}

int ShardKvClient::Put(const string& k, const string& v) {
  shardid_t shard_id = Key2Shard(k);
  Log_info("Leader before updating: %lli",leader_idx_);
  auto smc = make_shared<ShardMasterClient>();
  Log_info("Before creating client");
  smc->commo_ = commo_;
  Log_info("After creating client");
  //smc.leader_idx_ = sms.sp_log_svr_->site_id_;
  //auto cli = sms.CreateClient();
  ShardConfig config;
  ShardConfig* ptr_config = &config;
  //sms.Query(-1,ptr_ret,ptr_config,);
  Log_info("Before calling query from Shardmaster client");
  uint32_t ret = smc->Query(-1,ptr_config);
  Log_info("After calling query from Shardmaster client");
  uint32_t partition = config.shard_group_map_[shard_id];
  Log_info("Partition: %lli",partition);
  return Op([&](uint32_t* r)->int{
    return Proxy(partition*5 + leader_idx_).Put(GetNextOpId(), k, v, r);
  });
}

ShardKvProxy& ShardKvClient::Proxy(siteid_t site_id) {
  verify(commo_);


  auto p = (ShardKvProxy*)commo_->rpc_proxies_.at(site_id);
  return *p; 
}
