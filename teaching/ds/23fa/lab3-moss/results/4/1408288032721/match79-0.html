int ShardKvClient::Op(function<int(uint32_t*)> func) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      leader_idx_ = (leader_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {


      leader_idx_ = (leader_idx_+1) % 5;
    }
  }

}

int ShardKvClient::Put(const string& k, const string& v) {
  shardid_t shardId=Key2Shard(k);
  auto p = (ShardMasterProxy*)commo_->rpc_proxies_.at(0);
  ShardConfig shardConfig;
  uint32_t ans = -1;
  p->Query(-1, &ans, &shardConfig);
  uint32_t groupId = shardConfig.shard_group_map_[shardId];
  vector<uint32_t> gidList = shardConfig.group_servers_map_[groupId];
  if(gidList.size()>2) {
    leader_idx_ = gidList[2];
  }
  

  return Op([&](uint32_t* r)->int{
    return Proxy(leader_idx_).Put(GetNextOpId(), k, v, r);
  });
}

ShardKvProxy& ShardKvClient::Proxy(siteid_t site_id) {
  verify(commo_);
  ShardKvProxy *proxy;
  int numShards = 4;
  // auto p = (ShardKvProxy*)commo_->rpc_proxies_.at(site_id);
  for(int i=0;i<numShards;i++) {
    auto proxies = commo_->rpc_par_proxies_[i];
      for (auto& p : proxies) {
        if (p.first == site_id) {
          proxy = (ShardKvProxy*) p.second;
        }
      }
  }


  return *proxy; 
}
