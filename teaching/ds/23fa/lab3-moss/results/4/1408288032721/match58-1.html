int ShardKvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
    // lab_shard: fill in your code
    auto s = make_shared<MultiStringMarshallable>();
    string oidString = to_string(oid);


    s->data_.push_back(to_string(oid));
    s->data_.push_back("get");
    s->data_.push_back(k);
    shared_ptr<Marshallable> cmd = s;
    uint64_t index = 0, term = 0;

    if (!GetRaftServer().Start(cmd, &index, &term)) {
        return KV_NOTLEADER;
    }
    operationStatus[oidString] = false;
    Coroutine::Sleep(TIMEOUT_INTERVAL);


    if (operationStatus[oidString] && datastore.find(k) != datastore.end()) {
        *v = datastore[k];
        return KV_SUCCESS;
    }
    return KV_TIMEOUT;
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
    auto v = (MultiStringMarshallable *) (&m);
    /* your code here */
    string key, value, type, oid;
    oid = v->data_[0];
    type = v->data_[1];
    if (type != "get") {
        key = v->data_[2];
        value = v->data_[3];
    }
    if (type == "append") {
        if (datastore.find(key) != datastore.end()) {
            datastore[key] += value;
        } else {
            datastore[key] = value;
        }
    } else if (type == "put") {
//        string log_str="put : " +value;
//        Log_info(log_str.c_str());
        datastore[key] = value;
    }
    operationStatus[oid] = true;
}
