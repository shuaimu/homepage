int ShardKvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
    // lab_shard: fill in your code
    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(to_string(oid));
    s->data_.push_back("put");
    s->data_.push_back(k);
    s->data_.push_back(v);

    // Log_info("PUT: value is %s with oid %d on server %d (leader: %d)", v.c_str(), oid, GetRaftServer().loc_id_, GetRaftServer().state == ServerState::Leader);

    uint64_t cmdIndex, leaderTerm;
    shared_ptr<Marshallable> cmd = dynamic_pointer_cast<Marshallable>(s);
    bool startAgreement = GetRaftServer().Start(cmd, &cmdIndex, &leaderTerm);
    if (!startAgreement) {
      return KV_NOTLEADER;
    }
    this->outstanding_requests_[to_string(oid)] = Reactor::CreateSpEvent<IntEvent>();
    this->outstanding_requests_[to_string(oid)]->Wait(OPERATION_TIMEOUT);
    if (this->outstanding_requests_[to_string(oid)]->status_ == Event::DONE) {
      shared_ptr<IntEvent> waitForCommit = Reactor::CreateSpEvent<IntEvent>();
      waitForCommit->Wait(WAIT_FOR_COMMIT);
      kv_store_[k] = v;


      return KV_SUCCESS;
    }
    return KV_TIMEOUT;
}

int ShardKvServer::Append(const uint64_t& oid, 
                     const string& k,
                     const string& v) {
    // lab_shard: fill in your code
    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(to_string(oid));
    s->data_.push_back("append");
    s->data_.push_back(k);
    s->data_.push_back(v);

    uint64_t cmdIndex, leaderTerm;
    shared_ptr<Marshallable> cmd = dynamic_pointer_cast<Marshallable>(s);
    bool startAgreement = GetRaftServer().Start(cmd, &cmdIndex, &leaderTerm);
    if (!startAgreement) {
      // Log_info("server %d not leader, state is candidate (%d); follower (%d)", GetRaftServer().loc_id_, GetRaftServer().state == ServerState::Candidate, GetRaftServer().state == ServerState::Follower);
      return KV_NOTLEADER;
    }
    this->outstanding_requests_[to_string(oid)] = Reactor::CreateSpEvent<IntEvent>();
    this->outstanding_requests_[to_string(oid)]->Wait(OPERATION_TIMEOUT);
    if (this->outstanding_requests_[to_string(oid)]->status_ == Event::DONE) {
      shared_ptr<IntEvent> waitForCommit = Reactor::CreateSpEvent<IntEvent>();
      waitForCommit->Wait(WAIT_FOR_COMMIT);
      if (kv_store_.find(k) == kv_store_.end()) {
        kv_store_[k] = v;
      } else {


        kv_store_[k] += v;
      }
      return KV_SUCCESS;
    }
    return KV_TIMEOUT;
}

int ShardKvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
    // lab_shard: fill in your code
    auto s = make_shared<MultiStringMarshallable>();
    s->data_.push_back(to_string(oid));
    s->data_.push_back("get");
    s->data_.push_back(k);
    s->data_.push_back(*v);
    // Log_info("GET: with oid %d on server %d (leader: %d)", oid, GetRaftServer().loc_id_, GetRaftServer().state == ServerState::Leader);

    uint64_t cmdIndex, leaderTerm;
    shared_ptr<Marshallable> cmd = dynamic_pointer_cast<Marshallable>(s);
    bool startAgreement = GetRaftServer().Start(cmd, &cmdIndex, &leaderTerm);
    if (!startAgreement) {
      return KV_NOTLEADER;
    }

    this->outstanding_requests_[to_string(oid)] = Reactor::CreateSpEvent<IntEvent>();
    this->outstanding_requests_[to_string(oid)]->Wait(OPERATION_TIMEOUT);
    if (this->outstanding_requests_[to_string(oid)]->status_ == Event::DONE) {
      shared_ptr<IntEvent> waitForCommit = Reactor::CreateSpEvent<IntEvent>();
      waitForCommit->Wait(WAIT_FOR_COMMIT);
      if (kv_store_.find(k) != kv_store_.end()) {
        *v = kv_store_[k];
        // Log_info("GET: returning value %s", keyValueStore[k].c_str());
      } else {
        *v = "";
      }
      return KV_SUCCESS;
    }
    return KV_TIMEOUT;
}
