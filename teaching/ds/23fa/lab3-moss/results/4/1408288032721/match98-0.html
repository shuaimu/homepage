int ShardKvServer::Append(const uint64_t& oid, 
                     const string& k,
                     const string& v) {
    // lab_shard: fill in your code
    RaftServer& site = GetRaftServer();
   if(site.isLeader){
   // Log_info("in append %s value %s",k.c_str(),v.c_str());
         auto s = make_shared<MultiStringMarshallable>();


      s->data_.push_back(to_string(oid));
      s->data_.push_back("append");
      s->data_.push_back(k);
      s->data_.push_back(v);
         
      uint64_t index;
      uint64_t term;
      auto cmdPointer = dynamic_pointer_cast<Marshallable>(s);
     //  Log_info("Sleeping main Thread Append");
       auto ev = temp(cmdPointer,index,term,site);
       ev->Wait(500000);
       //Log_info("Waking main Thread Append");


       if (kv_store_.find(k) != kv_store_.end() &&
           kv_store_[k].compare(kv_store_[k].length() - v.length(), kv_store_[k].length(), v) == 0) {
          //  Log_info("Append returning success");
               return KV_SUCCESS;
           }else{
          //  Log_info("Append returning timeout");
           return KV_TIMEOUT;
       }

   }

    if(kv_store_.find(k)!=kv_store_.end()){
        kv_store_[k]=kv_store_[k].append(v);
    }else{
      kv_store_.insert({k,v});
    }
    return KV_SUCCESS;
}

int ShardKvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
    // lab_shard: fill in your code
    RaftServer& site = GetRaftServer();
   if(site.isLeader){
       isMajority = false;
     //  Log_info("in get key %s",k.c_str());


       auto s = make_shared<MultiStringMarshallable>();
       s->data_.push_back(to_string(oid));
       s->data_.push_back("get");
       s->data_.push_back(k);

      uint64_t index;
      uint64_t term;
      auto cmdPointer = dynamic_pointer_cast<Marshallable>(s);
       // Log_info("Sleeping main Thread GET");
       auto ev = temp(cmdPointer,index,term,site);
       ev->Wait(500000);
       // Log_info("Waking main Thread GET");

       if(isMajority && kv_store_.find(k)!=kv_store_.end()){
           *v=kv_store_[k];
         //  Log_info("Get returning success");
           return KV_SUCCESS;              
       } else {
         //  Log_info("Get returning timeout");
           return KV_SUCCESS;
       }
   }
   *v=kv_store_[k];
 //  Log_info("Get returning not Leader");
   return KV_SUCCESS;
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
    RaftServer& site = GetRaftServer();
      site.mtx_.lock();
       Log_info("In onNext for site %d ",site.site_id_);
       if(m.kind_==MarshallDeputy::CMD_MULTI_STRING){
       auto v = dynamic_cast<MultiStringMarshallable*>(&m);
       if (v->data_.empty()) {
         //  Log_error("Received an empty command.");
           return;
       }
       const string& command = v->data_[1];
       if (command == "put") {
           kv_store_[v->data_[2]] = v->data_[3];
       } else if (command == "append") {
         if(kv_store_.find(v->data_[2])!=kv_store_.end()){
           //  kv_store_[v->data_[2]] = kv_store_[v->data_[2]].append(v->data_[3]);
           }
           else{  
            kv_store_[v->data_[2]] = v->data_[3];
           }
           
       }else if (command == "get") {
          // Log_info("In OnNext GET for site %d for key %s", site.site_id_,v->data_[2].c_str());
           isMajority = true;
       }     
       }
       site.mtx_.unlock();
}
