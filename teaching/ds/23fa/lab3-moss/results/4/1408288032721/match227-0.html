int ShardKvClient::Op(function<int(uint32_t*)> func,const string& k) {
  uint64_t t1 = Time::now();
  auto cli = make_shared<ShardMasterClient>();
  cli->commo_ = commo_;
  Coroutine::CreateRun([this, &cli](){
    cli->Query(-1, &(this->config));
  });
  Coroutine::Sleep(1000000);
  offset = config.group_servers_map_[config.shard_group_map_[Key2Shard(k)]][0];
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);
    fprintf(stderr, "wjawdaf %d %u\n", offset+leader_idx_, ret);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      leader_idx_ = (leader_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      leader_idx_ = (leader_idx_+1) % 5;
    }
  }

}

int ShardKvClient::Put(const string& k, const string& v) {

  }, k);
}

ShardKvProxy& ShardKvClient::Proxy(siteid_t site_id) {
  verify(commo_);
  auto p = (ShardKvProxy*)commo_->rpc_proxies_.at(site_id);
  return *p; 
}
