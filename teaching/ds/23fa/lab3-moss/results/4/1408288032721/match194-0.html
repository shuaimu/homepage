int64_t ShardKvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;
  ret = ret << 32;
  ret = ret + op_id_cnt_++; 
  return ret;
}

bool ShardKvServer::IsLeader() {
  RaftServer& rs = GetRaftServer();
  bool is_leader;
  uint64_t term;
  rs.GetState(&is_leader, &term);
  return is_leader;
}

int ShardKvServer::PublishRaftCommand(const uint64_t& oid, const string& k, const string& v, const string& op) {
  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back(op);
  s->data_.push_back(k);
  s->data_.push_back(v);

  RaftServer& rs = GetRaftServer();
  shared_ptr<Marshallable> cmd = s;
  uint64_t index, term;
  rs.Start(cmd, &index, &term);
  Log_info("Shard KV Node %d :: %s %s %s", rs.site_id_, op.c_str(), k.c_str(), v.c_str());

  for(int i=1; i<=3; ++i) {
    Coroutine::Sleep(i*300000);
    if(uncommitted_ops_[k].find(oid) == uncommitted_ops_[k].end()) break;
  }
  return uncommitted_ops_[k].find(oid) != uncommitted_ops_[k].end() ? KV_TIMEOUT : KV_SUCCESS;
}

int ShardKvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
    // lab_shard: fill in your code
    if (!IsLeader()) return KV_NOTLEADER;
    uncommitted_ops_[k].insert(oid);
    return PublishRaftCommand(oid, k, v, "put");
}

int ShardKvServer::Append(const uint64_t& oid, 
                     const string& k,
                     const string& v) {
    // lab_shard: fill in your code
    if (!IsLeader()) return KV_NOTLEADER;
    uncommitted_ops_[k].insert(oid);
    return PublishRaftCommand(oid, k, v, "append");
}

int ShardKvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
    // lab_shard: fill in your code
    if(!IsLeader()) return KV_NOTLEADER;

    if(uncommitted_ops_[k].size() > 0 || kv_store_.find(k) == kv_store_.end()) {


      return KV_TIMEOUT;
    } else {
      *v = kv_store_[k];
      return KV_SUCCESS;
    }
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
    auto v = (MultiStringMarshallable*)(&m);
    uint64_t op_id = stoull(v->data_[0]);


    string op = v->data_[1];
    string k = v->data_[2];
    string v1 = v->data_[3];
    if(uncommitted_ops_[k].find(op_id) != uncommitted_ops_[k].end()) {
      uncommitted_ops_[k].erase(op_id);
    }
    if(op == "put") {
      kv_store_[k] = v1;
    } else if(op == "append") {
      kv_store_[k] += v1;
    }
}
