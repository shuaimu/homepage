void ShardMasterServiceImpl::Join(const map<uint32_t, std::vector<uint32_t>>& gid_server_map, uint32_t* ret, rrr::DeferredReply* defer) {
  // your code here
  
  uint32_t newGroupId = gid_server_map.begin()->first;
  Log_info("New group joined with ID: %d", newGroupId);
  uint32_t configSize = configs_.size();
  map<uint32_t, vector<uint32_t> > groupToListOfServers;
  map<uint32_t, uint32_t> shardToGroups;

  if(configSize == 0){
    for(int i=1;i<=10;i++){
      shardToGroups[i] = 1;
    }
    groupToListOfServers = gid_server_map;
  }
  else{
    shardToGroups = configs_[configSize].shard_group_map_;
    groupToListOfServers = configs_[configSize].group_servers_map_;

    if(groupToListOfServers.find(newGroupId) != groupToListOfServers.end()){
      *ret = KV_SUCCESS;
      defer->reply();
      return;
    }

    for(auto gid: gid_server_map){
      groupToListOfServers[gid.first] = gid.second;
    }

    map<uint32_t, vector<uint32_t> > groupToShards;
    for(auto i: shardToGroups){
      groupToShards[i.second].push_back(i.first);
    }
    
    uint32_t n = groupToListOfServers.size();
    uint32_t quotient = (uint32_t)(10)/(n);
    uint32_t remainder = (uint32_t)(10)%(n);
    
    for(auto i: groupToShards){
      int currCount = i.second.size();
      int required = quotient;
      if(remainder){
        required++;
        remainder--;
      }
      int difference = currCount-required;
      // Remove difference number of elements from the back
      for(int j=0;j<difference;j++){
        uint32_t shard = i.second[i.second.size()-1];
        shardToGroups[shard] = newGroupId;
        i.second.pop_back();
      }
    }
  }

    ShardConfig shardConfig;
    shardConfig.number = configSize+1;
    shardConfig.shard_group_map_ = shardToGroups;
    shardConfig.group_servers_map_ = groupToListOfServers;
    configs_[shardConfig.number] = shardConfig;

    *ret = KV_SUCCESS;
     defer->reply();
}

void ShardMasterServiceImpl::Leave(const std::vector<uint32_t>& gids, uint32_t* ret, rrr::DeferredReply* defer) {
  // your code here
  uint32_t configSize = configs_.size();
  uint32_t latestConfigNum = configs_[configSize].number;

  map<uint32_t, uint32_t> shardToGroups = configs_[configSize].shard_group_map_;
  map<uint32_t, vector<uint32_t>> groupToListOfServers = configs_[configSize].group_servers_map_;
  
  map<uint32_t, vector<uint32_t> > groupToShards;
  for(auto i: shardToGroups){
    groupToShards[i.second].push_back(i.first);
  }

  vector<uint32_t>allocateShards;
  for(auto groupID: gids){
    allocateShards.insert(allocateShards.end(),groupToShards[groupID].begin(),groupToShards[groupID].end());
    groupToShards.erase(groupID);
    groupToListOfServers.erase(groupID);
  }

  int groupCount = groupToShards.size();
  uint32_t quotient = (uint32_t)(10)/(groupCount);
  uint32_t remainder = (uint32_t)(10)%(groupCount);

  for(auto gts: groupToShards){
    int grp = gts.first;
    int currCount = gts.second.size();
    int required = quotient;
    if(remainder){
      required++;
      remainder--;
    }
    int difference = required - currCount; 
    for(int j=0; j < difference; j++){
      int shard = allocateShards.back();
      shardToGroups[shard] = grp;
      allocateShards.pop_back();
    }
  }

  ShardConfig shardConfig;
  shardConfig.number = configSize+1;
  shardConfig.group_servers_map_ = groupToListOfServers;
  shardConfig.shard_group_map_ = shardToGroups;
  configs_[shardConfig.number] = shardConfig;

  *ret = KV_SUCCESS;
  defer->reply();
}
