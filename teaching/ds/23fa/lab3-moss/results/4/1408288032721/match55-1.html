void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
  auto v = (MultiStringMarshallable*)(&m);
  string opid=v->data_[0];


  string op=v->data_[1];
  string key=v->data_[2];
  string value=v->data_[3];
  RaftServer &ser= GetRaftServer();
  index_size++;
  if(ack.find(opid) != ack.end()){
    return ;
  }
  ack[opid]=1;
  if(op=="put"){
    db[key]=value;
    Log_info("database size %d",db.size());
  }
  else if(op=="append"){
    if(db.count(key)==0){
      db[key]=value;
    }
    else{
      db[key]=db[key]+value;
    }
    Log_info("append database size %d",db.size()); 
  }
 
}
