int ShardKvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
    // lab_shard: fill in your code
  resPut=0;
  Log_info("the value of oid is %d",oid);
  auto evPut = Reactor::CreateSpEvent<IntEvent>();
  auto commd = make_shared<MultiStringMarshallable>();


  commd->data_.push_back(to_string(oid));
  commd->data_.push_back("put");
  commd->data_.push_back(k);
  commd->data_.push_back(v);
  janus::RaftServer& svr_ = GetRaftServer();
  //Log_info("the leader is %d current_term is %d log index is %d",svr_.loc_id_,svr_.current_term,svr_.log.size()-1);
  shared_ptr<Marshallable> cmd_raft = commd;
  uint64_t index;
  uint64_t term;
  Log_info("THe site_id_ of server %d",svr_.site_id_);
  if(!svr_.Start(cmd_raft,&index,&term)) {
    Log_info("THe site_id_ of not leader server %d",svr_.site_id_);
    resPut=KV_NOTLEADER;
    return resPut;
  }
  else {
      Event_map.insert({to_string(oid),evPut});
      Log_info("the leader is %d",svr_.site_id_);
  }
  evPut->Wait(3000000);
  Log_info("Came back to put with %d",evPut->get());
  if(evPut->get()) {
    resPut=KV_SUCCESS;
  }
  else {
    resPut=KV_TIMEOUT;
  }
  return resPut;
}

void ShardKvServer::OnNextCommand(Marshallable& m) {
    // lab_shard: fill in your code
    auto v = (MultiStringMarshallable*)(&m);
  if(v->data_[1] == "put") {
    //mtx_.lock();
    New_Map[v->data_[2]]=v->data_[3];
    Log_info("the key is %s and the value is %s",v->data_[2].c_str(),v->data_[3].c_str());
    if(Event_map.find(v->data_[0])!=Event_map.end()) {
      //Log_info("PUT-the oid of the command %d",stoi(v->data_[0]));
        Event_map[v->data_[0]]->Set(1);
    }
    //mtx_.unlock();
  }

  else if(v->data_[1] == "append") {
    //mtx_.lock();
    //Log_info("Append-the oid of the command %d",stoi(v->data_[0]));
    if(New_Map.find(v->data_[2])==New_Map.end()) {
      Log_info("the value is not present in map adding it now");
      New_Map[v->data_[2]]=v->data_[3];
    }
    else {
      Log_info("the Appending value to %d",stol(New_Map[v->data_[2]]));
      Log_info("the value to be appended %d",stol(v->data_[3]));
      New_Map[v->data_[2]]+=v->data_[3];
      Log_info("the Appended value is %d",stol(New_Map[v->data_[2]]));
    }
    if(Event_map.find(v->data_[0])!=Event_map.end()) {
        Event_map[v->data_[0]]->Set(1);
    }
    //mtx_.unlock();
  }
  else if(v->data_[1] == "get") {
    //mtx_.lock();
    //Log_info("GET-the oid of the command %d",stoi(v->data_[0]));
    if(Event_map.find(v->data_[0])!=Event_map.end()) {      
          if(New_Map.find(v->data_[2])!=New_Map.end()) {
             Log_info("the get value is %d",stol(New_Map[v->data_[2]]));
             Event_map[v->data_[0]]->Set(1);
          }
    }
    //mtx_.unlock();
  }
}
