int ShardKvClient::Op(function<int(uint32_t*)> func) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      leader_idx_ = (leader_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      leader_idx_ = (leader_idx_+1) % 5;
    }
  }

}
int ShardKvClient::OpNew(function<int(uint32_t*)> func, vector<uint32_t> group) {
  uint64_t t1 = Time::now();
  while (true) {
    uint64_t t2 = Time::now();
    if (t2 - t1 > 10000000) {
      return KV_TIMEOUT;
    }
    uint32_t ret = 0;
    int r1; 
    r1 = func(&ret);
    //Log_info("In shardKv for PUT?GET REQUESRT TO SERVER %d",group[group_idx_]);
    if (r1 == ETIMEDOUT || ret == KV_TIMEOUT) {
      group_idx_ = (group_idx_+1) % 5;
      return KV_TIMEOUT;
    }
    if (ret == KV_SUCCESS) {
      return KV_SUCCESS;
    }
    if (ret == KV_NOTLEADER) {
      group_idx_ = (group_idx_+1) % 5;
    }
  }

}

int ShardKvClient::Put(const string& k, const string& v) {
  shardid_t shard = Key2Shard(k);
  auto cli = GetNewConfig();
  cli->Query(-1,&shard_config);
  uint32_t gid = shard_config.shard_group_map_[shard];


  vector<uint32_t> group = shard_config.group_servers_map_[gid];
  return OpNew([&](uint32_t* r)->int{
    return Proxy(group[group_idx_]).Put(GetNextOpId(), k, v, r);
  },group);
}
