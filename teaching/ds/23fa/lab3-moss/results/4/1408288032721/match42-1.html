  const int TIMEOUT = 3 * 1000 * 1000;

  int ShardKvServer::Put(const uint64_t &oid,
                         const string &k,
                         const string &v)
  {
    // lab_shard: fill in your code


    string oidStr = to_string(oid);
    eventMap[oidStr] = Reactor::CreateSpEvent<IntEvent>();
    auto s = make_shared<MultiStringMarshallable>();
    uint64_t servIndex, servIerm;
    RaftServer &raftServ = GetRaftServer();
    Log_info("Initialised event");

    s->data_.push_back(to_string(oid));
    s->data_.push_back("put");
    s->data_.push_back(k);
    s->data_.push_back(v);



    shared_ptr<janus::Marshallable> ptr = s;
    // auto ptr = dynamic_pointer_cast<janus::Marshallable>(s);

    bool retMessage = raftServ.Start(ptr, &servIndex, &servIerm);
    Log_info("Start called with Put and oid %s", oidStr.c_str());

    if (retMessage == false)
    {
      Log_info("KV_NOTLEADER");
      return KV_NOTLEADER;
    }

    eventMap[oidStr]->Wait(TIMEOUT);

    Log_info("Wait done");

    if (eventMap[oidStr]->status_ == Event::TIMEOUT)
    {
      Log_info("timeout happens");
      return KV_NOTLEADER;
    }
    else
    {
      kvStore[k] = v;
      Log_info("KV_SUCCESS");
      return KV_SUCCESS;
    }
  }

  int ShardKvServer::Get(const uint64_t &oid,
                         const string &k,
                         string *v)
  {
    // lab_shard: fill in your code
    auto s = make_shared<MultiStringMarshallable>();
    string oidStr = to_string(oid);
    eventMap[oidStr] = Reactor::CreateSpEvent<IntEvent>();
    uint64_t servIndex, servIerm;
    RaftServer &raftServ = GetRaftServer();

    s->data_.push_back(to_string(oid));
    s->data_.push_back("get");
    s->data_.push_back(k);

    shared_ptr<janus::Marshallable> ptr = s;
    // auto ptr = dynamic_pointer_cast<janus::Marshallable>(s);
    bool retMessage = raftServ.Start(ptr, &servIndex, &servIerm);
    Log_info("Start called with Get and oid %s", oidStr);

    if (retMessage == false)
    {
      return KV_NOTLEADER;
    }

    eventMap[oidStr]->Wait(TIMEOUT);

    if (eventMap[oidStr]->status_ == Event::TIMEOUT)
    {
      Log_info("timeout happens");
      return KV_NOTLEADER;
    }
    else
    {
      if (kvStore.find(k) == kvStore.end())
      {
        *v = "";
      }
      else
      {
        *v = kvStore[k];
      }
      return KV_SUCCESS;
    }
  }

  void ShardKvServer::OnNextCommand(Marshallable &m)
  {
    // lab_shard: fill in your code


    auto v = (MultiStringMarshallable *)(&m);
    string oid = v->data_[0];
    string operation = v->data_[1];
    Log_info("OnNextCommand operation - %s", operation.c_str());
    if (operation == "put")
    {
      string key = v->data_[2];
      string value = v->data_[3];
      kvStore[key] = value;
      Log_info("OnNextCommand Put, oid - %s, operation - %s, key - %s", oid.c_str(), operation.c_str(), key.c_str());
    }
    else if (operation == "append")
    {
      string key = v->data_[2];
      string value = v->data_[3];
      if (kvStore.find(key) == kvStore.end())
      {
        kvStore[key] = value;
      }
      else
      {
        string oldValue = kvStore[key];
        kvStore[key] = oldValue + value;
      }
      Log_info("OnNextCommand Append, oid - %s, operation - %s, key - %s", oid.c_str(), operation.c_str(), key.c_str());
    }
    if (eventMap.find(oid) == eventMap.end())
      return;
    eventMap[oid]->Set(1);
  }
