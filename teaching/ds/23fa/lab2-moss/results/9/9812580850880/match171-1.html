    return KV_TIMEOUT;
  }

}

void KvServer::OnNextCommand(Marshallable& m) {


  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  // extract oid, op, k, v from v
  // if op == "put", then put k, v into data_
  // if op == "append", then append v to data_[k]
  // if op == "get", then get v from data_[k]
  auto opid = v->data_[0];
  auto op = v->data_[1];
  auto k = v->data_[2];
  string val;
  if (op != "get") {
    val = v->data_[3];
  }
  if (op == "put") {
    data_[k] = val;
  } else if (op == "append") {
    data_[k] += val;
  } else if (op == "get") {
    val = data_[k];
  }
  completed_ops_.insert(stoull(opid));
}
