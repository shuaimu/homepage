int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {

  Log_info("Received Get request with OID : %d , Key : %s",oid,k.c_str());
  RaftServer& raftPeer = GetRaftServer();
  if(raftPeer.state != 2){
    Log_info("Inside Get, the KV server is not the leader");
    return KV_NOTLEADER;
  }
  // Now we know that the KV server is leader here
  auto operation = make_shared<MultiStringMarshallable>();
  operation->data_.push_back(to_string(oid));
  operation->data_.push_back("GET");
  operation->data_.push_back(k);
  
  getCommitSuccess = 0;
  uint64_t index = 0;
  uint64_t term = 0;
  auto op = dynamic_pointer_cast<Marshallable>(operation);

  Log_info("Calling start of raft peer with id %d for OID %d",raftPeer.loc_id_,oid);


  auto ev = Reactor::CreateSpEvent<IntEvent>();
  raftPeer.Start(op,&index,&term);
  ev->Wait(400000);
  
  // Check for success -> Return KV_SUCCESS
  // Check for failure -> Return KV_NOTLEADER
  

  if(getCommitSuccess){
    *v = keyValueStore[k];
    Log_info("Get successful : OP = %d, K = %s , V = %s",oid,k.c_str(),(*v).c_str());
    return KV_SUCCESS;
  }
  else{
    Log_info("Get timed out : OP = %d, K = %s",oid,k.c_str());
    return KV_TIMEOUT;
  }
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  string op = v->data_[1];
  if(op == "PUT"){
    string key = v->data_[2];
    string value = v->data_[3]; 
    keyValueStore[key] = value;
    getPutSuccess = 1;
  }
  if(op == "APPEND"){
    Log_info("In OnNext of append for OP ID : %d, Key = %s and Value = %s",v->data_[0],v->data_[2].c_str(),v->data_[3].c_str());
    string key = v->data_[2];
    string value = v->data_[3]; 
    keyValueStore[key].append(value);
    getAppendSuccess = 1;
  }
  if(op == "GET"){
    getCommitSuccess = 1;
    Log_info("Get operation with OP ID : %d is also committed",v->data_[0]);
  }
}
