int KvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
  /* 
  Your are recommended to use MultiStringMarshallable as the format

  s->data_.push_back(to_string(op_id));
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  */
  /* your code here */
  cout<<"KvServer Put: "<<oid<<" "<<k<<" "<<v<<" Raft server id "<<GetRaftServer().site_id_<<endl;
  shared_ptr<Marshallable> command = GenerateMarshallable(oid, PUT, k, v);
  bool start_response;

  uint64_t index, term;
  start_response = GetRaftServer().Start(command, &index, &term);
  
  if (!start_response) {
    cout<<"Put is not leader for "<<GetRaftServer().site_id_<<endl;
    return KV_NOTLEADER;
  }

  int retries = 1000;

  bool op_commited;
  string oid_str = to_string(oid);

  while(--retries){
    mtx_.lock();
    op_commited = commited_oids.count(oid_str);
    mtx_.unlock();

    if (op_commited){
      cout<<"Put is Success"<<endl;
      return KV_SUCCESS;
    }
    Coroutine::Sleep(10);
  }
  
  cout<<"Put is Timeout"<<endl;
  return KV_TIMEOUT;
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  cout<<"KvServer Get: "<<oid<<" "<<k<<" Raft server id "<<GetRaftServer().site_id_<<endl;
  shared_ptr<Marshallable> command = GenerateMarshallable(oid, GET, k, "");
  uint64_t index, term;
  bool start_response = GetRaftServer().Start(command, &index, &term);
  
  if (!start_response) {
    cout<<"Get is not leader for "<<GetRaftServer().site_id_<<endl;
    return KV_NOTLEADER;
  }

  int retries = 1000;


  bool op_commited;
  string oid_str = to_string(oid);

  while(--retries){
    mtx_.lock();
    op_commited = commited_oids.count(oid_str);
    mtx_.unlock();

    if (op_commited){
      mtx_.lock();
      *v = database[k];
      mtx_.unlock();
      cout<<"Get is Success"<<endl;
      return KV_SUCCESS;
    }
    Coroutine::Sleep(10);
  }

  cout<<"Get is Timeout"<<endl;
  return KV_TIMEOUT;
}

void KvServer::KvDataUpdate(vector<string> data) {
  string operation = data[1], key = data[2], value = data[3];
  if (operation == PUT) {
    database[key] = value;
  } else if(operation == APPEND) {
    database[key] += value;
  }
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  cout<<"KvServer id: "<<v->data_[0]<<" -> "<<v->data_[1]<<": "<<v->data_[2]<<" "<<v->data_[3]<<endl;
  mtx_.lock();
  KvDataUpdate(v->data_);
  commited_oids.insert(v->data_[0]);
  cout<<"Commited id "<<v->data_[0]<<endl;
  cout<<"Database value for "<<v->data_[2]<<" = "<<database[v->data_[2]]<<endl;
  mtx_.unlock();
  Coroutine::Sleep(10);
}
