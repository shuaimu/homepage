int64_t KvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;
  ret = ret << 32;
  ret = ret + op_id_cnt_++; 
  return ret;
}

                  const string& k,

  bool is_leader;
  uint64_t term, index;

  auto cmd = std::make_shared<MultiStringMarshallable>();
  cmd->data_.push_back(to_string(oid));
  cmd->data_.push_back("put");
  cmd->data_.push_back(k);
  cmd->data_.push_back(v);

  auto cmd_ptr = dynamic_pointer_cast<Marshallable>(cmd);

  is_leader = GetRaftServer().Start(cmd_ptr, &index, &term);

  if (!is_leader) {
    return KV_NOTLEADER;
  }

  auto ev = Reactor::CreateSpEvent<IntEvent>();
  op_id_event_map_[oid] = ev;
  ev->Wait(15*100000);
  // Coroutine::Sleep(10*100000);
  // Log_info("wait for %u", ev->status_ == Event::TIMEOUT);

  if (ev->status_ == Event::TIMEOUT) {
    op_id_event_map_.erase(oid);
    return KV_TIMEOUT;
  }

  op_id_event_map_.erase(oid);
  return KV_SUCCESS;
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  bool is_leader;
  uint64_t term, index;

  auto cmd = std::make_shared<MultiStringMarshallable>();
  cmd->data_.push_back(to_string(oid));
  cmd->data_.push_back("get");
  cmd->data_.push_back(k);

  auto cmd_ptr = dynamic_pointer_cast<Marshallable>(cmd);

  is_leader = GetRaftServer().Start(cmd_ptr, &index, &term);

  if (!is_leader) {
    return KV_NOTLEADER;
  }

  auto ev = Reactor::CreateSpEvent<IntEvent>();
  op_id_event_map_[oid] = ev;
  ev->Wait(15*100000);

  if (ev->status_ == Event::TIMEOUT) {
    op_id_event_map_.erase(oid);
    return KV_TIMEOUT;
  }

  op_id_event_map_.erase(oid);
  if (kv_map_.find(k) != kv_map_.end()) {
    *v = kv_map_[k];
  } else {
    *v = "";
  }
  return KV_SUCCESS;
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  // Log_info("on next command %s", v->data_[1].c_str());


  if (v->data_[1] == "put") {
    // Log_info("put %s %s", v->data_[2].c_str(), v->data_[3].c_str());
    kv_map_[v->data_[2]]=v->data_[3];
  } else if (v->data_[1] == "append") {
    // Log_info("append %s %s", v->data_[2].c_str(), v->data_[3].c_str());
    if (kv_map_.find(v->data_[2]) == kv_map_.end()) {
      kv_map_[v->data_[2]] = "";
    }
    kv_map_[v->data_[2]]+=v->data_[3];
  } else if (v->data_[1] == "get") {
    // Log_info("get %s %s", v->data_[2].c_str(), kv_map_[v->data_[2]].c_str());
    // do nothing
  } else {
    verify(0);
  }
  uint64_t oid = stoull(v->data_[0]);
  if(op_id_event_map_.find(oid) != op_id_event_map_.end()){
    op_id_event_map_[oid]->Set(1);
  }
}
