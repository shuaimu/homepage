    Log_info("--------------OnNextCommand------------------%d", sp_log_svr_->site_id_);


  // auto coro4 = Coroutine::CreateRun([&] () {
  std::mutex mut;
  mut.lock();

  auto v = (MultiStringMarshallable*)(&m);
            std::string op = v->data_[0];
            std::string optype = v->data_[1];
            std::string key = v->data_[2];
            std::string val = v->data_[3];
  // q.push(v);
    if (optype ==  "put"){
      storage[key] = val;
      if (putevent!= nullptr && putevent->status_ == Event::WAIT){
        putevent->Set(1);
      }
    }else if(optype ==  "append") {
      if (storage.find(key) != storage.end()) {
              storage[key] = storage[key] + val;
          } else {  
              storage[key] = val;
          }
      if (appendevent!= nullptr && appendevent->status_ == Event::WAIT){
        std::cout<<"in next command  append "<<storage[key]<<std::endl;
          appendevent->Set(1);
      }
    } else if (optype == "get"){
      if (getevent!= nullptr && getevent->status_ == Event::WAIT){
        getevent->Set(1);
      }
    }
  mut.unlock();


}
