bool KvServer::KVOperation(string oid){
  RaftServer& raftServer = this->GetRaftServer();
  lock_guard<recursive_mutex> lock(raftServer.mtx_);
  Timer waitTimer;
  waitTimer.start();
  while(waitTimer.elapsed()*1e6 < this->agreementTimeOut && this->completedCmds.count(oid)==0){
    Log_info("Time Elapsed: %f", waitTimer.elapsed());
    Coroutine::Sleep(this->agreementTimeOut/10);
  }
  // waitTimer.reset();
  return (this->completedCmds.count(oid));
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  RaftServer& raftServer = this->GetRaftServer();
  lock_guard<recursive_mutex> lock(raftServer.mtx_);
  string oid_string  = to_string(oid);
  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  // s->data_.push_back(*v);
  shared_ptr<Marshallable> cmd = s;
  uint64_t index, term;
  bool result = raftServer.Start(cmd, &index, &term);
  if(!result){
    Log_info("Get Not Leader %d", oid);
    return KV_NOTLEADER;
  }
  bool ret  = KVOperation(oid_string);
  if(ret){
    Log_info("Get Success %d", oid);
    if(completedCmds.count(oid_string)){
      *v = kvStore[k];
      return KV_SUCCESS;
    }
  }
  Log_info("Get Timeout %d", oid);
  return KV_TIMEOUT;
}

void KvServer::OnNextCommand(Marshallable& m) {
  RaftServer& raftServer = this->GetRaftServer();
  lock_guard<recursive_mutex> lock(raftServer.mtx_);
  auto v = (MultiStringMarshallable*)(&m);
  string oid = (v->data_[0]);
  string op = v->data_[1];
  string key = v->data_[2];
  completedCmds.insert(oid);
  if(op == "get"){

  }
  else if(op == "put"){
    string newVal = v->data_[3];
    kvStore[key] = newVal;
    Log_info("Put - Key %s, Val %s", key, kvStore[key]);
  }
  else if(op == "append"){
    string newVal = v->data_[3];
    if(kvStore.count(key)){
      try{
        string orignal = kvStore[key];
        kvStore[key]= orignal + newVal;
      }
      catch(...){
        Log_info("Nikunj Mittal catches error!!!");
      }
    }
    else{
      kvStore[key]=newVal;
    }
    Log_info("Put - Key %s, Val %s, appendVal: %s", key, kvStore[key], newVal);
  }
}
