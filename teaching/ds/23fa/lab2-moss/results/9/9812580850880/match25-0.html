shared_ptr<Marshallable> KvServer::GetMarshallableLogCmd(uint64_t operationId, CommandType cmdType, string k, string v){
  auto logEntry = make_shared<MultiStringMarshallable>();
  logEntry->data_.push_back(to_string(operationId));
  logEntry->data_.push_back(to_string(cmdType));
  logEntry->data_.push_back(k);


  logEntry->data_.push_back(v);
  return dynamic_pointer_cast<Marshallable>(logEntry);
}

int KvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
  auto logEntry = GetMarshallableLogCmd(oid, CommandType::PUT, k, v);
  uint64_t index, term;
  // Log_info("[%d] in put, before eventmap [%d], (%d:%d)", sp_log_svr_->loc_id_, oid, k, v);
  auto isLeader = GetRaftServer().Start(logEntry, &index, &term);
  if (!isLeader) {
    return KV_NOTLEADER;
  }

  eventMap[oid] = Reactor::CreateSpEvent<IntEvent>();
  eventMap[oid]->Wait(EVENT_TIMEOUT);
  if (eventMap[oid]->IsTimeout()) {
    eventMap.erase(oid);
    return KV_TIMEOUT;
  }

  eventMap.erase(oid);
  // Log_info("[%d] done with put %d:%d (%d)", sp_log_svr_->loc_id_, k, v, dataStore[k]);
  return KV_SUCCESS;
}

int KvServer::Append(const uint64_t& oid, 
                     const string& k,
                     const string& v) {
  auto logEntry = GetMarshallableLogCmd(oid, CommandType::APPEND, k, v);
  uint64_t index, term;
  auto isLeader = GetRaftServer().Start(logEntry, &index, &term);
  if (!isLeader)
  {
    return KV_NOTLEADER;
  }

  // Log_info("[%d] in append, before eventmap [%d], (%d:%d)", sp_log_svr_->loc_id_, oid, k, v);
  eventMap[oid] = Reactor::CreateSpEvent<IntEvent>();
  eventMap[oid]->Wait(EVENT_TIMEOUT);
  if (eventMap[oid]->IsTimeout())
  {
    // Log_info("[%d]append timed out [%d] - (%d:%d)", sp_log_svr_->loc_id_, oid, k, v);
    eventMap.erase(oid);
    return KV_TIMEOUT;
  }

  eventMap.erase(oid);
  // Log_info("[%d] done with append %d:%d (%d)", sp_log_svr_->loc_id_, k, v, dataStore[k]);
  return KV_SUCCESS;
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  auto logEntry = GetMarshallableLogCmd(oid, CommandType::GET, k);
  uint64_t index, term;
  auto isLeader = GetRaftServer().Start(logEntry, &index, &term);
  if (!isLeader)
  {
    return KV_NOTLEADER;
  }

  // Log_info("[%d] in get before eventmap %d", sp_log_svr_->loc_id_, k);
  eventMap[oid] = Reactor::CreateSpEvent<IntEvent>();
  eventMap[oid]->Wait(EVENT_TIMEOUT);
  if (eventMap[oid]->IsTimeout())
  {
    eventMap.erase(oid);
    return KV_TIMEOUT;
  }
  eventMap.erase(oid);
  
  if (dataStore.find(k) != dataStore.end())
  {
    // Log_info("in get");
    *v = dataStore.at(k);
  }
  else
  {
    *v = "";
  }

  // Log_info("[%d] done with get %d:%d (%d)", sp_log_svr_->loc_id_, k, *v, dataStore[k]);
  return KV_SUCCESS;
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here  */ 
  auto opId = uint64_t(std::stoul(v->data_.at(0)));
  auto operation = CommandType(std::stoi(v->data_.at(1)));
  auto key = v->data_.at(2);
  auto value = v->data_.at(3);
  

  switch (operation) {
    case CommandType::PUT:
      dataStore[key] = value;
      break;
    case CommandType::GET:
      break;
    case CommandType::APPEND:
      if (dataStore.find(key) != dataStore.end())
      {
        dataStore[key] += value;
      }
      else
      {
        dataStore[key] = value;
      }
      break;
    default:
      return;
  }
  if (eventMap.find(opId) != eventMap.end()) {
    eventMap.at(opId)->Set(1);
  }
}
