int KvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
  /* 
  Your are recommended to use MultiStringMarshallable as the format 
  for the log entries. Here is an example how you can use it.
  */

  RaftServer& raft = this->GetRaftServer();

  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);

  uint64_t index, term;
  auto c = dynamic_pointer_cast<Marshallable>(s);
  if (!raft.Start(c, &index, &term)) {
    return KV_NOTLEADER;
  }

  Coroutine::CurrentCoroutine()->Sleep(200000); // 200 ms

  if (raft.commitIndex >= index) {


    return KV_SUCCESS;
  }

  return KV_TIMEOUT;
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  RaftServer& raft = this->GetRaftServer();

  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);

  uint64_t index, term;
  auto c = dynamic_pointer_cast<Marshallable>(s);
  if (!raft.Start(c, &index, &term)) {
    return KV_NOTLEADER;
  }

  Coroutine::CurrentCoroutine()->Sleep(200000); // 200 ms

  if (raft.commitIndex >= index) {
    *v = store[k];


    return KV_SUCCESS;
  }

  return KV_TIMEOUT;
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);

  string op = v->data_[1];
  string k = v->data_[2];

  // do nothing if we get a "get"
  if (op == "put") {
    store[k] = v->data_[3];
  } else if (op == "append") {
    store[k] += v->data_[3];
  }
}
