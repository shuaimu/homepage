void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */

  uint64_t oid = strToUint64(v->data_[0]);
  std::string opType = v->data_[1];
  // uint64_t cmdIdx = stoi(v->data_[2]);

  // Log_info("[+] %d | OnNextCommand | oid: %d", sp_log_svr_->site_id_, oid);

  if (IsOIDDuplicated(oid)) {

  } else {
    std::lock_guard<std::recursive_mutex> lk(mtx_);
    if (opType == "put") {
      std::string key   = v->data_[2];
      std::string value = v->data_[3];
      kv[key] = value;
      executedRequests[oid] = "done";
    } else if (opType == "append") {
      std::string key   = v->data_[2];
      std::string value = v->data_[3];
      kv[key] += value;
      executedRequests[oid] = "done";
    } else if (opType == "get") {
      std::string key   = v->data_[2];
      executedRequests[oid] = kv[key];
    } else {
      verify(0);
    }
  }
  bool isLeader;
  uint64_t currentTerm;
  GetRaftServer().GetState(&isLeader, ¤tTerm);

  std::lock_guard<std::recursive_mutex> lk(mtx_);
  if (isLeader &&   
      currentTerm == termReceived[oid]) {
    verify(notifiers.find(oid) != notifiers.end());
    notifiers[oid] = true;
  }
}
