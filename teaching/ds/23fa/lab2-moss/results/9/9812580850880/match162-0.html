  /* your code here */
  
  RaftServer* raftServer = &GetRaftServer();


  auto s = make_shared<MultiStringMarshallable>();
  string soid = to_string(oid);
  s->data_.push_back(soid);
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  shared_ptr<Marshallable> ss = static_pointer_cast<Marshallable>(s);
  uint64_t index;
  uint64_t term;
  // *index = *term = 0;
  if(!raftServer->Start(ss, &index, &term)){
    Log_info("KV Server is not leader");
    return KV_NOTLEADER;
  }
  int x = 10000;
  while(x){
    if(checkOPId[soid])
      break;
    x--;
    Log_info("OP ID not done in PUT, op id: %d",oid);
    Coroutine::Sleep(2000);
  }
  if(x <= 0){
    Log_info("Timeout happened in PUT, op id: %d",oid);
    return KV_TIMEOUT;
  }
  return KV_SUCCESS;
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  if(v->data_.size() < 3)
    return;
  string opId = v->data_[0];
  string cmd = v->data_[1];
  string key = v->data_[2];
  if(cmd.compare("put") == 0){
    string value = v->data_[3];
    kvStore[key] = value;
  }else if(cmd.compare("append") == 0){
    string value = v->data_[3];
    if(kvStore.count(key) > 0)
      kvStore[key] += value;
    else
      kvStore[key] = value;
  }else if(cmd.compare("get") == 0){
    if(kvStore.count(key) == 0)
      getMap[opId] = "";
    else
      getMap[opId] = kvStore[key];
  }
  checkOPId[opId] = true;
  Coroutine::Sleep(2000);
}
