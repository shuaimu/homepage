int KvServer::Put(const uint64_t& oid, 
                  const string& k,
                  const string& v) {
  /* 
  Your are recommended to use MultiStringMarshallable as the format 
  for the log entries. Here is an example how you can use it.
 */
  RaftServer& raftServer = GetRaftServer();
  if(raftServer.state != LEADER)
    return KV_NOTLEADER;
  
  string request = "put" + to_string(oid) + k + v;
  myRequests.insert(request);

  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  uint64_t index;
  uint64_t term;
  auto marsh = dynamic_pointer_cast<Marshallable>(s);
  raftServer.Start(marsh, &index, &term);
  Log_info("PUT AT LEADER %d => oid is %d key is %s value is %s index of cmd is %d leader term is %d", raftServer.loc_id_, oid, k.c_str(), v.c_str(), index, term);
  //cout<<"Thread id : "<<this_thread::get_id()<<" for server : "<<GetRaftServer().loc_id_<<endl;
  
  putEvent = Reactor::CreateSpEvent<IntEvent>();
  putEvent->Wait(800000); //timeout after 800 ms
  if (putEvent->status_ == Event::TIMEOUT) {
    Log_info("TIMEOUT PUT oid %d leader %d", oid, raftServer.loc_id_);
    return KV_TIMEOUT;
  } else if(putEvent->get() == 1) {
    Log_info("SUCCESS PUT oid %d leader %d", oid, raftServer.loc_id_);


    return KV_SUCCESS; 
  }
  // raftServer.mtx_.unlock();
  return KV_TIMEOUT;
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  RaftServer& raftServer = GetRaftServer();
  if(raftServer.state != LEADER)
    return KV_NOTLEADER;

  string request = "get" + to_string(oid) + k;
  myRequests.insert(request);

  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  s->data_.push_back(k);
  s->data_.push_back(*v);
  uint64_t index;
  uint64_t term;
  auto marsh = dynamic_pointer_cast<Marshallable>(s);
  raftServer.Start(marsh, &index, &term);
  Log_info("GET AT LEADER %d => oid is %d key is %s index of cmd is %d leader term is %d", raftServer.loc_id_, oid, k.c_str(), index, term);
  //cout<<"Thread id : "<<this_thread::get_id()<<" for server : "<<GetRaftServer().loc_id_<<endl;
  
  getEvent = Reactor::CreateSpEvent<IntEvent>();
  getEvent->Wait(800000); //timeout after 800 ms

    Log_info("TIMEOUT GET oid %d leader %d", oid, raftServer.loc_id_);
    return KV_TIMEOUT;
  } else if(getEvent->get() == 1) {
    *v = kvstore[k];
    Log_info("SUCCESS GET oid %d leader %d", oid, raftServer.loc_id_);
    return KV_SUCCESS; 
  }
  return KV_TIMEOUT;
}

void KvServer::OnNextCommand(Marshallable& m) {
  RaftServer& raftServer = GetRaftServer();
  
  Coroutine::CreateRun([this, &m, &raftServer](){
    auto v = (MultiStringMarshallable*)(&m);
    string opid = v->data_[0];
    string op = v->data_[1];
    string key = v->data_[2];
    string value = v->data_[3];
    //Log_info("INSIDE ONNEXTCOMMAND server %d => opid : %s op : %s k : %s v : %s", raftServer.loc_id_, opid.c_str(), op.c_str(), key.c_str(), value.c_str());

    raftServer.mtx_.lock();
    if(op == "put"){
      Log_info("PUTTING kvstore server %d k %s v %s", raftServer.loc_id_, key.c_str(), value.c_str());
      kvstore[key] = value;
    } else if(op == "append"){
      Log_info("APPENDING kvstore server %d k %s v %s", raftServer.loc_id_, key.c_str(), value.c_str());
      kvstore[key] += value;
    }
    raftServer.mtx_.unlock();

    string request = op + opid + key;
    if(op != "get") request += value;

    if(raftServer.state == LEADER && myRequests.find(request) != myRequests.end()){
      myRequests.erase(myRequests.find(request));
      Log_info("SET => %d is leader, setting %s event to 1 with k %s v %s", raftServer.loc_id_, op.c_str(), key.c_str(), value.c_str());
      if(op == "put"){
        putEvent->Set(1);
      } else if(op == "append"){
        appendEvent->Set(1);
      } else if(op == "get"){
        getEvent->Set(1);
      }
    }
  });
  
}
