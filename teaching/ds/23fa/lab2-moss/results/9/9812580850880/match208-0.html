int64_t KvServer::GetNextOpId() {
  verify(sp_log_svr_);
  int64_t ret = sp_log_svr_->site_id_;

  ret = ret + op_id_cnt_++;

bool KvServer::IsLeader() {
  bool isLeader;
  uint64_t termAtReqCheck;
  this->GetRaftServer().GetState(&isLeader, &termAtReqCheck);
  return isLeader;
}

                  const string& k,

  Log_info("PUT called %d %s %s", oid, k.c_str(), v.c_str());

  if (!this->IsLeader()) {
    Log_info("Not leader!");
    return KV_NOTLEADER;
  }

  s->data_.push_back("put");
  s->data_.push_back(k);
  s->data_.push_back(v);
  auto t = dynamic_pointer_cast<Marshallable>(s);
  uint64_t term;
  uint64_t index;
  // send put request
  this->GetRaftServer().Start(
      t,
      &index,
      &term);

  // await call to return or timeout
  auto event = Reactor::CreateSpEvent<IntEvent>();
  this->opCodeResolution[oid] = event;
  event->Wait(RAFT_COMMIT_TIMEOUT);
  if (event->status_ == Event::TIMEOUT)
  {
    Log_info("Put request timed out!");
    return KV_TIMEOUT;
  }
  else
  {
    Log_info("Put request seemingly succeeded?");
    return KV_SUCCESS;
  }
}

int KvServer::Get(const uint64_t& oid, 
                  const string& k,
                  string* v) {
  /* your code here */
  Log_info("GET called %d %s", oid, k.c_str());
  // MAKE SURE you enter gets into the log as well. this garauntees that you are part of the 
  // majority
  if (!this->IsLeader()) {
    Log_info("Not leader!");
    return KV_NOTLEADER;
  }
  // FIRST - perform an append rpc to raft. this will ensure you sync your local table.
  auto s = make_shared<MultiStringMarshallable>();
  s->data_.push_back(to_string(oid));
  s->data_.push_back("get");
  auto t = dynamic_pointer_cast<Marshallable>(s);
  // send get request
  uint64_t term;
  uint64_t index;
  this->GetRaftServer().Start(
      t,
      &index,
      &term);

  // await call to return or timeout
  auto event = Reactor::CreateSpEvent<IntEvent>();
  this->opCodeResolution[oid] = event;
  event->Wait(RAFT_COMMIT_TIMEOUT);
  if (event->status_ == Event::TIMEOUT)
  {
    Log_info("Get request timed out!");
    return KV_TIMEOUT;
  }
  else
  {
    Log_info("Get request seemingly succeeded?");
    if (this->localMap.find(k) != this->localMap.end()) {
      *v = this->localMap[k];
    }
    return KV_SUCCESS;
  }
}

void KvServer::OnNextCommand(Marshallable& m) {
  auto v = (MultiStringMarshallable*)(&m);
  /* your code here */
  uint64_t opCode = std::stoull(v->data_.at(0));
  string opType = v->data_.at(1);
  Log_info("Command received: %s (opcode=%d)", opType.c_str(), opCode);
  if (opType.compare("get") == 0) {
    // get operation - do nothing specific
  }
  else if (opType.compare("put") == 0) {
    // update the actual map
    string key = v->data_.at(2);
    string value = v->data_.at(3);
    this->localMap[key] = value;
  }
  else if (opType.compare("append") ==0 ) {
    // update the actual map
    string key = v->data_.at(2);
    string value = v->data_.at(3);
    if (this->localMap.find(key) == this->localMap.end()) {
      this->localMap[key] = value;
    }
    else {
      this->localMap[key] = this->localMap[key].append(value);
    }
  }
  // in all cases - resolve the corresponding opCode.
  if (this->opCodeResolution.find(opCode) != this->opCodeResolution.end()) {
    auto event = this->opCodeResolution[opCode];
    event->Set(1);
    // TODO - delete opCodes at some time;
  }
}
