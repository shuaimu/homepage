//
// A Go object implementing a single Raft peer.
//
type Raft struct {
	mu        sync.Mutex          // Lock to protect shared access to this peer's state
	peers     []*labrpc.ClientEnd // RPC end points of all peers
	persister *Persister          // Object to hold this peer's persisted state
	me        int                 // this peer's index into peers[]

	state rfstate
	// Your data here (2A, 2B, 2C).
	// Look at the paper's Figure 2 for a description of what
	// state a Raft server must maintain.

	//same as raft.pdf
	// State
	// Persistent state on all servers:
	currentTerm int //latest term server has seen (initialized to 0	on first boot, increases monotonically)
	votedFor    int //candidateId that received vote in current term (or null if none)
	// peer i vote for peer j, then i.votedFor = j
	// set -1 for null because integer do not have nil
	voteCount int

	log []LogItem //store logs
	//LastIncludedIndex int
	//LastIncludedTerm  int

	//Volatile state on leaders: (Reinitialized after election)
	//


	nextIndex []int //for each server, index of the next log entry to send to that server (initialized to leader last log index + 1)
	//index[i] means peer[i]'s next index
	matchIndex []int //for each server, index of highest log entry known to be replicated on server (initialized to 0, increases monotonically)
	//index[i] means peer[i]'s match index

	//Volatile state on all servers:
	commitIndex int //index of highest log entry known to be committed (initialized to 0, increases	monotonically)
	//commitIndex is log's index, not actual rf index (consider snapshot)
	lastApplied int //index of highest log entry applied to state machine (initialized to 0, increases monotonically)

	//channels
	chanHeartbeat chan bool
	chanSetLeader chan bool
	chanApply     chan ApplyMsg
	chanCommit    chan bool
	//hbtimer       *time.Timer //heartbeat timeout timer, used by follower
	//candtimer     *time.Timer //candidate timeout timer, used by candidate
	shouldStop bool //dead loop kill switch
}

// return currentTerm and whether this server
// believes it is the leader.
func (rf *Raft) GetState() (int, bool) {

	var term int
	var isleader bool
	// Your code here (2A).
	rf.mu.Lock()
	term = rf.currentTerm
	isleader = (rf.state == LEADER)
	rf.mu.Unlock()

	return term, isleader
}

func (rf *Raft) getLastCommitedIdx() int {
	//get last commited index
	//if a server fail and recover, the safest way to set hintIdx is to set it as commitIndex
	// if rf.currentTerm > 9 && rf.commitIndex == 0 {
	// 	EPrintf("ERR: commit index is 0\n")
	// }
	return rf.commitIndex
}
func (rf *Raft) getLastCommitedLogPos() int {
	//get last log position of commited log's index
	//for example
	//rf.log = [5,6,7,8,9], and commitIndex is 7, return 2 because rf.log[2].Idx==7
	//if a server fail and recover, the safest way to set hintIdx is to set it as commitIndex
	for i := 0; i < len(rf.log); i++ {
		if rf.log[i].Idx == rf.commitIndex {
			return i
		}
	}
	return -2
}

func truncateLog(lastIncludedIndex int, lastIncludedTerm int, log []LogItem) []LogItem {

	var trucLog []LogItem
	trucLog = append(trucLog, LogItem{Idx: lastIncludedIndex, Term: lastIncludedTerm})

	for index := len(log) - 1; index >= 0; index-- {
		//reverse search
		if log[index].Idx == lastIncludedIndex && log[index].Term == lastIncludedTerm {
			trucLog = append(trucLog, log[index+1:]...)
			break
		}
	}

	return trucLog
}

//StartSnapshot .


func (rf *Raft) StartSnapshot(snapshot []byte, index int) {

	rf.mu.Lock()
	defer rf.mu.Unlock()
	//EPrintf("[%v]snapshot begin\n", rf.me)
	//rf.PrintSnapshotData(snapshot)

	baseIndex := rf.log[0].Idx
	lastIndex := rf.getLastLogIdx()

	if index <= baseIndex || len(rf.log) == 0 || (index-baseIndex) >= len(rf.log) {
		// in case having installed a snapshot from leader before snapshotting
		// second condition is a hack
		EPrintf("BUG: not able to snapshot\n")
		return
	}

	var newLogEntries []LogItem

	newLogEntries = append(newLogEntries, LogItem{Idx: index, Term: rf.log[index-baseIndex].Term})

	for i := index + 1; i <= lastIndex; i++ {
		newLogEntries = append(newLogEntries, rf.log[i-baseIndex])
	}

	rf.log = newLogEntries

	rf.persist()

	w := new(bytes.Buffer)
	e := gob.NewEncoder(w)
	e.Encode(newLogEntries[0].Idx)


	e.Encode(newLogEntries[0].Term)
	//e.Encode(rf.currentTerm)
	//e.Encode(rf.votedFor)
	//e.Encode(rf.log)
	data := w.Bytes()
	data = append(data, snapshot...)
	rf.persister.SaveStateAndSnapshot(rf.persister.ReadRaftState(), data)
	//EPrintf("[%v]snapshot done\n", rf.me)
	//rf.PrintSnapshotData(snapshot)
	//

}

//InstallSnapshot example from RequestVote RPC handler
func (rf *Raft) InstallSnapshot(args InstallSnapshotArgs, reply *InstallSnapshotReply) {
	//rpc call
	rf.mu.Lock()
	defer rf.mu.Unlock()
	//EPrintf("SNAPSHOT:[%v] called, log len = (%v)\n", rf.me, len(rf.log))
	if args.Term < rf.currentTerm {
		reply.Term = rf.currentTerm
		EPrintf("Snapshot: reject, arg term(%v) smaller than current term(%v)", args.Term, rf.currentTerm)
		//reject because term small
		return
	}
	// keep alive
	rf.chanHeartbeat <- true
	DPrintf("Snapshot: HB sent")

	w := new(bytes.Buffer)
	e := gob.NewEncoder(w)
	e.Encode(rf.currentTerm)
	e.Encode(rf.votedFor)
	e.Encode(rf.log)
	e.Encode(rf.commitIndex)
	//e.Encode(rf.LastIncludedIndex)
	//e.Encode(rf.LastIncludedTerm)
	data := w.Bytes()
	//EPrintf("SNAPSHOT:[%v] snapshot DB data = ", rf.me)
	//rf.PrintSnapshotData(args.Data)

	rf.persister.SaveStateAndSnapshot(data, args.Data)
	rf.log = truncateLog(args.LastIncludedIndex, args.LastIncludedTerm, rf.log)

	//rf.log = truncateLog(args.LastIncludedIndex, args.LastIncludedTerm, rf.log)

	msg := ApplyMsg{HaveSnapshot: true, Snapshot: args.Data, CommandValid: false}

	rf.lastApplied = args.LastIncludedIndex
	//rf.LastIncludedIndex = args.LastIncludedIndex
	rf.commitIndex = args.LastIncludedIndex

	rf.persist()
	rf.chanApply <- msg
	//EPrintf("SNAPSHOT: rf.commitIndex=%v, rf.log=%v\n", rf.commitIndex, rf.log)
}

//sendInstallSnapshot .
func (rf *Raft) sendInstallSnapshot(server int, args InstallSnapshotArgs, reply *InstallSnapshotReply) bool {

	//EPrintf("SNAPSHOT: ready to call, len(rf.log)=%v\n", len(rf.log))
	var ok = false


	ok = rf.peers[server].Call("Raft.InstallSnapshot", args, reply)

	rf.mu.Lock()
	defer rf.mu.Unlock()
	if ok {
		if reply.Term > rf.currentTerm {
			//see higher term, self back to follower
			rf.currentTerm = reply.Term
			rf.state = FOLLOWER
			rf.votedFor = -1
			return ok
		}
		//EPrintf("SNAPSHOT: len(rf.log)=%v\n", len(rf.log))
		rf.nextIndex[server] = args.LastIncludedIndex + 1
		rf.matchIndex[server] = args.LastIncludedIndex
	} else {
		DPrintf("SNAPSHOT: RPC failed\n")
	}
	return ok
}

				DPrintf("[rejectNoMatch]Entry PrevLogIndex(%v) == ConflictPos(%v), Not at least one Match, unreliable; reject with retry, reset hintIdx = LastCommitIdx(%v), args.entries = %v\n", args.PrevLogIndex, conflictPos, rf.getLastCommitedIdx(), args.Entries)
				reply.HintIdx = rf.getLastCommitedIdx()
			} else {
				reply.HintIdx = 1
			}
			reply.Term = -1
			reply.Success = false //reject - false success
			return

		}

		if touchedConf == true || len(rf.log) == 1 || len(args.Entries) == 1 {
			isSpecialpassed := false
			//means we checked the logs
			//make sure no conflict place have this flag true
			if conflictPos == -1 {
				//no conflict
				//conflict pos is [1, 2] + [2]
				// or 			  [1, 2] + [2, 3, 4]
				DPrintf("No conflict\n")
				//conflictPos = len(rf.log)
				appendbase = entryIdx
				//rf.printLogEntry()
				DPrintf("args.Entries=%v", args.Entries)

			}
			if len(args.Entries) == 1 && len(rf.log) > 1 {
				//something like [1, 2] + [3]
				EPrintf("[SPECIAL] args.entries only have one log(%v), args.prevlogidx=%v, args.prevlogterm=%v, startpos=%v, len(rf.log)=%v, rf.lastlogIdx=%v, conflictpos=%v, entryIdx=%v\n", args.Entries, args.PrevLogIndex, args.PrevLogTerm, startPos, len(rf.log), rf.getLastLogIdx(), conflictPos, entryIdx)
				//rf.printLogEntry()
				if args.PrevLogTerm == rf.log[len(rf.log)-1].Term && args.PrevLogIndex == rf.log[len(rf.log)-1].Idx {
					EPrintf("[SPECIAL] args.entries only have one log--passed\n")
					appendbase = 0 //start from first element of args.Entries
					cutpos = len(rf.log) - 1
					isSpecialpassed = true
				} else {
					EPrintf("[SPECIAL] args.entries only have one log--NOT passed\n")
				}

			}

			if conflictPos == rf.getLastCommitedLogPos()+1 {
				//for testing, not action
				//DPrintf("[append-after-cmted]Entry PrevLogIndex(%v), ConflictPos(%v), cmtidx=(%v), args.Entries=%v", args.PrevLogIndex, conflictPos, rf.getLastCommitedIdx(), args.Entries)
			}

			if conflictPos <= rf.getLastCommitedLogPos() && conflictPos != -1 {
				//this is request from disconnected leader
				//Follower [1, 2, 3, 4], term 2, commited at 3 (pos=2)
				//Mal-leader [#2, #3, #4], term 5
				//conflictpos=1, but it is wrong to overwrite them all, because commited is commited, just reject
				EPrintf("Mal-Leader, conflictPos(%v) < rf[%v].commitIndex(%v), last commited log pos = %v, reject with no retry", conflictPos, rf.me, rf.commitIndex, rf.getLastCommitedLogPos())
				//reply.HintIdx = rf.getLastCommitedIdx() + 1
				reply.Term = -1
				reply.HintIdx = -2

				reply.Success = false //reject - false success
				return
			}
			//DPrintf("touchedConf= True, len(Entries)=%v, ConflictPos(%v), endPos(%v), modify appendBase\n", len(args.Entries), conflictPos, endPos)
			if conflictPos != -1 && rf.log[conflictPos].Idx == args.Entries[0].Idx && isSpecialpassed == false &&
				(conflictPos > rf.getLastCommitedLogPos()) &&
				args.PrevLogTerm != rf.log[conflictPos].Term {
				//conflict at the very first point, not reliable
				//maybe log=[1, 2, #3, #4, #5] but Entries=[#4, #5]
				//should reject, and hint with commitIdx
				EPrintf("[rejectAppend]Entry PrevLogIndex(%v), ConflictPos(%v),  conflict at beginning, unreliable; reject with retry, reset hintIdx = LastCommitIdx(%v)--\n", args.PrevLogIndex, conflictPos, rf.getLastCommitedIdx())
				EPrintf("args.entries=%v, rf.log=%v\n\n", args.Entries, rf.log)
				if rf.getLastCommitedIdx() > 0 {
					reply.HintIdx = rf.getLastCommitedIdx()
				} else {
					reply.HintIdx = 1
				}
				reply.Term = -1 //force to go into reply.term <= rf.currentterm

				reply.Success = false //reject but retry - false success
				return
			}
			//rf.log = rf.log[:conflictPos]

			if rf.getLastCommitedLogPos() != -2 {
				lastCommitedTerm := rf.log[rf.getLastCommitedLogPos()].Term
				for _, item := range args.Entries {
					if item.Term < lastCommitedTerm && item.Idx > rf.getLastCommitedIdx() {
						EPrintf("Disconnected Leader, term(%v) smaller than commited term(%v) at rf.log[%v], reject to follower--", item.Term, lastCommitedTerm, rf.getLastCommitedLogPos())
						EPrintf("arg term (%v), rf term(%v), args.entries=%v, rf.log=%v, reject append\n", args.Term, rf.currentTerm, args.Entries, rf.log)
						reply.Term = -1
						//hintIdx == -2 means reject without retry, just make caller go back to follower
						reply.HintIdx = -2
						reply.Success = false
						return
					}
				}

			}

			//DPrintf("3. If an existing entry conflicts with a new one (same index but different terms)\n")
			//DPrintf("Found conflict position at log pos %v, truncating rf.log ( original len %v)\n", conflictPos, len(rf.log))
			//not including rf.log[conflictPos]
			if len(rf.log) > 0 {
				//deal with rf.log
				rf.log = rf.log[:cutpos]
			} else {
				//rf.log is null
				//appendbase = 0
			}
			//rf.printLogEntry() //test
			//now start append from conflictPos
			loglist := make([]LogItem, 0)
			//if appendbase == len(args.Entries)
			if appendbase == len(args.Entries) {
				//make it null
				DPrintf("no conflict, appending 0-len list\n")
			} else {

				DPrintf("Entries len = %v, appendBase = %v, args.PrevLogIdx=%v\n--from Leader[%v]start appending entries starting from %v\n", len(args.Entries), appendbase, args.PrevLogIndex, args.LeaderID, args.Entries[appendbase])
				DPrintf("")
			}
			for _, item := range loglist {
				DPrintf("appending item %v\n", item)
				rf.log = append(rf.log, item)
				DPrintf("len(rf.log)=%v\n", len(rf.log))
				//rf.printLogEntry()
			}

		}

		reply.Success = true
		reply.HintIdx = rf.getLastLogIdx() + 1
		//reply.HintIdx = rf.getLastCommitedIdx() + 1
		//rf.printLogEntry()

		if len(args.Entries) == 1 && len(rf.log) > 1 {
			//something like [1, 2] + [3]
			EPrintf("[SPECIAL] args.entries only have one log(%v), args.prevlogidx=%v, args.prevlogterm=%v, startpos=%v, len(rf.log)=%v, rf.lastlogIdx=%v, conflictpos=%v, entryIdx=%v\n", args.Entries, args.PrevLogIndex, args.PrevLogTerm, startPos, len(rf.log), rf.getLastLogIdx(), conflictPos, entryIdx)
			rf.printLogEntry()
			if args.PrevLogTerm == rf.log[len(rf.log)-1].Term && args.PrevLogIndex == rf.log[len(rf.log)-1].Idx {
				EPrintf("[SPECIAL] args.entries only have one log--passed\n")

			} else {
				EPrintf("[SPECIAL] args.entries only have one log--NOT passed\n")
			}

		}
		//updating commit index
		//only when updating something
		// heartbeat do not update anything, so should not check this.
		//	Otherwise - bad situation:
		// 			RF0 have [1,2]commit but [2A,2B,2C] not commit (wrong), rf.commitIndex==2
		// 			but leader RF1 have [1,2,3] commit (leadercommit==3)
		// 			after one heartbeat RF0 will set commitIndex to 3 and commit[2A],
		//			which is WRONG
		//
		if args.LeaderCommit >= rf.commitIndex {
			last := rf.getLastLogIdx()
			if args.LeaderCommit > last {
				rf.commitIndex = last
				EPrintf("[1]Leader commit large +1 round, args.LeaderCommit=%v, rf.commitIdx=%v\n", args.LeaderCommit, rf.commitIndex)
				//
			} else {
				rf.commitIndex = args.LeaderCommit
			}
			DPrintf("rf(%v) lastlogIdx=%v, leadercommit=%v, upgraded to %v\n ", rf.log, rf.getLastLogIdx(), args.LeaderCommit, rf.commitIndex)
			//select min(last, leadercommit) as rf.commitIndex
			rf.chanCommit <- true
		} else {

			//EPrintf("[2]Leader commit small +1 round, args.LeaderCommit=%v, rf.commitIdx=%v\n", args.LeaderCommit, rf.commitIndex)
		}
		//DPrintf("args.LeaderCommit=%v, rf.commitIdx=%v\n", args.LeaderCommit, rf.commitIndex)
	}

	return
}

//
func (rf *Raft) RequestVote(args *RequestVoteArgs, reply *RequestVoteReply) {
	// Your code here (2A, 2B).
	rf.mu.Lock()
	defer rf.mu.Unlock()
	defer rf.persist()
	DPrintf("(%v)get one request vote from (%v)\n", rf.me, args.CandidateID)
	argisuptodate := false
	revote := false
	if args.Term < rf.currentTerm {
		//rf term bigger, **force reject**
		//reply with rf term to notify RPC caller
		reply.Term = rf.currentTerm
		reply.VoteGranted = false
		DPrintf("RF: %v; current term: %v;\n args: %v; term: %v;\n arg term small - reject\n", rf.me, rf.currentTerm, args.CandidateID, args.Term)
		return
	}
	if args.Term > rf.currentTerm {
		//args term bigger,
		//If RPC request or response contains term T > currentTerm: set currentTerm = T, convert to follower (§5.1)
		// 1. update term (forced)
		// 2. set self to follower
		// 3. but check logs before vote for args
		DPrintf("arg(%v,term%v) term bigger than (rf%v, term%v), rf(%v) conv to follower\n", args.CandidateID, args.Term, rf.me, rf.currentTerm, rf.me)
		rf.currentTerm = args.Term
		rf.state = FOLLOWER
		rf.votedFor = -1

	} else if args.Term == rf.currentTerm {
		//equal, do nothing
		DPrintf("[RF%v,argRF%v]args term(%v) same as RF\n", rf.me, args.CandidateID, args.Term)
	}
	reply.Term = rf.currentTerm

	//decide whether to revote
	if args.LastLogTerm > rf.getlastTerm() {
		//1. larger term wins, ignore idx length
		argisuptodate = true
		//DPrintf("[RF%v,argRF%v]arg(%v) term(%v) bigger, set argisuptodate = true", rf.me, args.CandidateID, args.CandidateID, args.Term)

	} else if args.LastLogTerm == rf.getlastTerm() {
		//2. if same term, larger idx (longer log) wins
		//DPrintf("--Same Log Term, longer log will win")
		if args.LastLogIndex > rf.getLastLogIdx() {
			argisuptodate = true
			//DPrintf("arg(%v) logterm(%v) same but log longer, set argisuptodate = true\n", args.CandidateID, args.LastLogTerm)
		} else if args.LastLogIndex == rf.getLastLogIdx() {
			argisuptodate = true
			//DPrintf("arg(%v) logterm(%v) same, arg idx same, set argisuptodate = true\n", args.CandidateID, args.LastLogTerm)

		} else {
			argisuptodate = false
			/*
				##caution
			*/
			//DPrintf("arg(%v) logterm(%v) same but log shorter, set argisuptodate = FALSE\n", args.CandidateID, args.LastLogTerm)
		}

	} else {
		//not up to date, do not revote
		argisuptodate = false
		//DPrintf("arg(%v) term(%v) log stale, setting revote to FALSE\n", args.CandidateID, args.Term)
	}
	// If votedFor is null or candidateId, and candidate’s log is at least as up-to-date as receiver’s log, grant vote
	if (rf.votedFor == args.CandidateID || rf.votedFor == -1) && (argisuptodate == true) {
		revote = true
		//DPrintf("(rf.votedFor == args.CandidateID || rf.votedFor == -1) && (argisuptodate == true), setting revote=true\n")
		//revote to same one, prevent bugs
		/*
			if args.CommitIndex < rf.getLastCommitedIdx() && len(rf.peers) != 3 {
				DPrintf("ERR: args commitidx(%v) < rf commitidx(%v)", args.CommitIndex, rf.getLastCommitedIdx())
				//args commit index small, means an outdated leader.
				// prevent wrong appending from vote
				revote = false
			}*/
	}

	if revote == true {
		//do the revote - vote to args peer
		rf.state = FOLLOWER
		reply.VoteGranted = true

		rf.votedFor = args.CandidateID
		//DPrintf("RF %v will votefor: %v\n", rf.me, args.CandidateID)
		//DPrintf("In RF term: %v,arg term %v, RF: %v will Revote to %v\n", rf.currentTerm, args.Term, rf.me, args.CandidateID)
	} else {
		//not revote

		//DPrintf("RF: %v DO NOT revote to %v, rf.votefFor = %v, args.CanditateID= %v, argisuptodate= %v\n",
		//	rf.me, args.CandidateID, rf.votedFor, args.CandidateID, argisuptodate)
	}

	return
}

//
// example code to send a RequestVote RPC to a server.
// server is the index of the target server in rf.peers[].
// expects RPC arguments in args.
// fills in *reply with RPC reply, so caller should
// pass &reply.
// the types of the args and reply passed to Call() must be
// the same as the types of the arguments declared in the
// handler function (including whether they are pointers).
//
// The labrpc package simulates a lossy network, in which servers
// may be unreachable, and in which requests and replies may be lost.
// Call() sends a request and waits for a reply. If a reply arrives
// within a timeout interval, Call() returns true; otherwise
// Call() returns false. Thus Call() may not return for a while.
// A false return can be caused by a dead server, a live server that
// can't be reached, a lost request, or a lost reply.
//
// Call() is guaranteed to return (perhaps after a delay) *except* if the
// handler function on the server side does not return.  Thus there
// is no need to implement your own timeouts around Call().
//
// look at the comments in ../labrpc/labrpc.go for more details.
//
// if you're having trouble getting RPC to work, check that you've
// capitalized all field names in structs passed over RPC, and
// that the caller passes the address of the reply struct with &, not
// the struct itself.
//
func (rf *Raft) sendRequestVote(server int, args *RequestVoteArgs, reply *RequestVoteReply) bool {

	defer func() {
		// recover from panic if one occured. Set err to nil otherwise.
		if recover() != nil {
			DPrintf("catched error in sendRequestVote\n")
			go rf.sendRequestVote(server, args, reply)
		}
	}()
	ok := rf.peers[server].Call("Raft.RequestVote", args, reply)
	rf.mu.Lock()
	defer rf.mu.Unlock()
	if rf.state != CANDIDATE {
		DPrintf("RF No longer candidate\n")
	}
	if ok {
		if rf.currentTerm < reply.Term {
			//current term smaller, rejected from otherside

			DPrintf("(%v) gets rejected, back down to follower\n", rf.me)
			rf.currentTerm = reply.Term
			rf.state = FOLLOWER
			rf.votedFor = -1
			rf.persist()
		}
		if reply.VoteGranted == true {
			if rf.state == CANDIDATE && rf.currentTerm != args.Term {
				DPrintf("WARN: candidate term(%v) changed, recv old vote(%v)\n", rf.currentTerm, args.Term)
				return ok
			}
			if rf.state == CANDIDATE && rf.currentTerm == args.Term {
				DPrintf("(%v) gets voted +1 from %v, previous vote=%v\n", rf.me, server, rf.voteCount)
				rf.voteCount++

				if rf.voteCount > len(rf.peers)/2 {
					// over half: 3/2=1, 5/2=2
					DPrintf("(%v) gets majority\n", rf.me)
					select {
					case rf.chanSetLeader <- true:
						DPrintf("sent set Leader\n")
						rf.voteCount = 0
					default:
						DPrintf("chanSetLeader already full, discarding setleader\n")
					}

				}

			} else {
				DPrintf("RF (%v) no longer candidate\n", rf.me)
			}
		}
	} else {
		DPrintf("INFO: vote RPC failed on %v request vote from %v\n", rf.me, server)

	}

	return ok
}

//

// if it's ever committed. the second return value is the current
// term. the third return value is true if this server believes it is

//
func (rf *Raft) Start(command interface{}) (int, int, bool) {
	index := -1
	term := -1
	//isLeader := true
	rf.mu.Lock()
	defer rf.mu.Unlock()

	//DPrintf("---try to append as leader---")
	isLeader := (rf.state == LEADER)

	if isLeader {
		index = rf.getLastLogIdx() + 1
		term = rf.currentTerm
		DPrintf("RF(%v) begin as leader to append log(%v) \n", rf.me, command)

		rf.log = append(rf.log, LogItem{Idx: index, Term: term, Ops: command})
		go rf.broadcastAppendEntries()
		rf.persist()
	} else {
		//DPrintf("---RF(%v) is not leader - not appending \n", rf.me)
	}

	// Your code here (2B).

	return index, term, isLeader
}

	//Leader broadcast log
	rf.mu.Lock()
	defer rf.mu.Unlock()
	cmtIdx := rf.commitIndex
	last := rf.getLastLogIdx()
	baseIdx := 1
	if len(rf.log) > 0 {
		baseIdx = rf.log[0].Idx



		for i := rf.commitIndex + 1; i <= last; i++ {
			cnt := 1
			for j := range rf.peers {
				if j != rf.me && rf.matchIndex[j] >= i && rf.log[i-baseIdx].Term == rf.currentTerm {
					//DPrintf("[leader]rf(%v).matchIndex[j=%v]=%v >= i=%v, rf.log[i-baseIdx].Term=%v, rf.currentTerm=%v\n", rf.me, j, rf.matchIndex[j], i, rf.log[i-baseIdx].Term, rf.currentTerm)
					//match index have some problem
					cnt++
				}
				//count how many RF have full log
			}
			if (2 * cnt) > len(rf.peers) {
				//DPrintf("cnt =%v, len(rf.peers)=%v\n", cnt, len(rf.peers))
				cmtIdx = i
				//if over half of RF have full log to log[i], then update cmdIdx to higher value
			}
		}

		if cmtIdx > rf.commitIndex {
			// if cmtIdx is updated to higher value
			// update rf.commitIndex, means commited more logs
			//EPrintf("\n---[COMMIT]commited logs from Idx(%v) to Idx(%v), updating rf(%v).commitIndex to %v\n", rf.commitIndex, cmtIdx, rf.me, cmtIdx)

			rf.commitIndex = cmtIdx
			//rf.printLogEntry()
			rf.chanCommit <- true
		}
	} else {
		DPrintf("LOG BASE IDX == 0\n")
		//return
	}

	//broadcast to peers
	for i := range rf.peers {
		if i != rf.me && rf.state == LEADER {
			//if not leader, cannot broadcast
			//broadcast appendEntries
			//DPrintf()


			if rf.nextIndex[i] > baseIdx {
				//for normal log entry append
				var args AppendEntriesArgs
				args.Term = rf.currentTerm
				args.LeaderID = rf.me
				args.PrevLogIndex = rf.nextIndex[i] - 1 // guess rf.nextIndex as prevlogIndex
				if len(rf.log) == 0 {
					args.PrevLogTerm = 0
					//init use
				} else if args.PrevLogIndex == 0 {
					args.PrevLogTerm = 0

				} else {
					if args.PrevLogIndex-baseIdx >= len(rf.log) {
						DPrintf("Not Enough (Commited) Log\n")
					} else {
						//EPrintf("rf[%v].nextindex=%v, args.PrevLogIndex = %v, baseIdx=%v, sendto=>%v\n", rf.me, rf.nextIndex, args.PrevLogIndex, baseIdx, i)
						if args.PrevLogIndex-baseIdx+1 > 0 && args.PrevLogIndex-baseIdx+1 < len(rf.log) {
							args.PrevLogTerm = rf.log[args.PrevLogIndex-baseIdx+1].Term
						}
					}
				}
				//TODO: Reply false if log doesn’t contain an entry at prevLogIndex
				//		whose term matches prevLogTerm
				if args.PrevLogIndex+1-baseIdx > len(rf.log) {
					args.PrevLogIndex = baseIdx
					DPrintf("Not Enough (Commited) Log-send whole log over\n")
					args.Entries = make([]LogItem, len(rf.log))
					copy(args.Entries, rf.log)
				} else {
					args.Entries = make([]LogItem, len(rf.log[args.PrevLogIndex+1-baseIdx:]))
					copy(args.Entries, rf.log[args.PrevLogIndex+1-baseIdx:])
				}
				//args.Entries = make([]LogItem, 0) //zero log entries, empty for heartbeat
				args.LeaderCommit = rf.commitIndex
				EPrintf("leader(%v)[term(%v)] assembling args: len(Entries)=%v[%v], prevlogIdx=%v, baseIdx=%v, leadercommit=%v =>RF(%v), leader.rf.log=%v\n", rf.me, rf.currentTerm, len(args.Entries), args.Entries, args.PrevLogIndex, baseIdx, rf.commitIndex, i, rf.log)
				//call appendentries RPC
				go func(i int, args AppendEntriesArgs) {
					var reply AppendEntriesReply
					//DPrintf("calling AppendEntries RPC\n")
					rf.sendAppendEntries(i, args, &reply)

				}(i, args)
			} else {
				/*
					EPrintf("\nSEND ALL LOG, baseIdx = %v, rf.nextIndex[i]=%v\n", baseIdx, rf.nextIndex[i])
					var args AppendEntriesArgs
					args.Term = rf.currentTerm
					args.LeaderID = rf.me
					args.PrevLogIndex = baseIdx
					//rf.nextIndex[i] - 1
					args.PrevLogTerm = 0 //no use in HB
					//args.Entries = make([]LogItem, 0)
					args.Entries = make([]LogItem, len(rf.log))
					copy(args.Entries, rf.log)
					args.LeaderCommit = rf.commitIndex

					//call appendentries RPC
					go func(i int, args AppendEntriesArgs) {
						var reply AppendEntriesReply
						DPrintf("calling AppendEntries RPC\n")
						rf.sendAppendEntries(i, &args, &reply)

					}(i, args)*/

				if rf.state == LEADER {
					//for snapshot
					var shouldsnapshot bool
					shouldsnapshot = true
					/*
						if rf.getLastCommitedIdx() > 40 {
							//EPrintf("SNAPSHOT: log len = %v\n", len(rf.log))
							//EPrintf("SNAPSHOT: ready to take one snapshot to (%v)\n", i)
							shouldsnapshot = false
						}*/
					if shouldsnapshot == true {
						var argsnap InstallSnapshotArgs


						argsnap.Term = rf.currentTerm
						argsnap.LeaderID = rf.me
						argsnap.LastIncludedIndex = rf.log[0].Idx
						argsnap.LastIncludedTerm = rf.log[0].Term
						argsnap.Data = rf.persister.ReadSnapshot()
						//EPrintf("Send InstallSnapshot data:")
						//rf.PrintSnapshotData(argsnap.Data)
						go func(server int, args InstallSnapshotArgs) {
							var reply InstallSnapshotReply
							rf.sendInstallSnapshot(server, argsnap, &reply)
						}(i, argsnap)
					}
				}

			}

		}

	}

}
func (rf *Raft) broadcastRequestVote() {

	//EPrintf("Enter broadcastRequestVote\n")
	rf.mu.Lock()
	defer rf.mu.Unlock()
	var args RequestVoteArgs
	args.Term = rf.currentTerm
	args.CandidateID = rf.me
	//args.CommitIndex = rf.getLastCommitedIdx()
	args.LastLogTerm = rf.getlastTerm()
	args.LastLogIndex = rf.getLastLogIdx()

	for i := range rf.peers {
		//broadcast
		if (i != rf.me) && (rf.state == CANDIDATE) {
			go func(i int, args RequestVoteArgs) {
				var reply RequestVoteReply
				rf.sendRequestVote(i, &args, &reply)
			}(i, args)

		}
	}

}

func (rf *Raft) getStateStr() string {
	str := ""
	switch rf.state {
	case LEADER:
		str = "LEADER"
	case FOLLOWER:
		str = "FOLLOWER"
	case CANDIDATE:
		str = "CANDIDATE"
	}
	return str
}
func (rf *Raft) readSnapshot(spdata []byte) {

	rf.mu.Lock()


	defer rf.mu.Unlock()
	rf.readPersist(rf.persister.ReadRaftState())
	//rf.currentTerm/votedfor have already stored into raftstate, no need to do it in snapshot state

	if len(spdata) == 0 {
		return
	}

	r := bytes.NewBuffer(spdata)
	d := gob.NewDecoder(r)

	LastIncludedIndex := 0
	LastIncludedTerm := 0

	d.Decode(&LastIncludedIndex)
	d.Decode(&LastIncludedTerm)
	//d.Decode(&rf.currentTerm)
	//d.Decode(&rf.votedFor)
	//d.Decode(rf.log)
	rf.commitIndex = LastIncludedIndex
	rf.lastApplied = LastIncludedIndex
	//rf.LastIncludedIndex = LastIncludedIndex
	//rf.LastIncludedTerm = LastIncludedTerm

	rf.log = truncateLog(LastIncludedIndex, LastIncludedTerm, rf.log)

	//commandvalid = false means this is a snapshot data
	//EPrintf("SNAPSHOT:sending spdata:")
	//rf.PrintSnapshotData(spdata)
	msg := ApplyMsg{HaveSnapshot: true, Snapshot: spdata, CommandValid: false}

	go func() {
		rf.mu.Lock()
		rf.chanApply <- msg
		rf.mu.Unlock()
	}()
}

				//close(rf.chanSetLeader)
				break
			}
			rf.mu.Unlock()

			case LEADER:

				}

						} else {

					DPrintf("RF %v Broadcasting HB and log\n", rf.me)
					go rf.broadcastAppendEntries()

				}

			}

		}
	}()

	//rf.readPersist(persister.ReadRaftState())
	go rf.doapply()
	return rf
}

func (rf *Raft) doapply() {

	defer func() {
		// recover from panic if one occured. Set err to nil otherwise.
		if recover() != nil {
			EPrintf("catched error in apply\n")
			go rf.doapply()
		}
	}()

	for {
		select {
		case <-rf.chanCommit:
			rf.mu.Lock()
			commitIndex := rf.commitIndex


			baseIndex := rf.log[0].Idx

			for i := rf.lastApplied + 1; i <= commitIndex; i++ {
				msg := ApplyMsg{CommandIndex: i, Command: rf.log[i-baseIndex].Ops, CommandValid: true}
				EPrintf("RF[%v]sending msg:(%v), rf.nextidx=%v\n", rf.me, msg, rf.nextIndex)
				rf.chanApply <- msg
				rf.printLogEntry()
				rf.lastApplied = i
			}

			rf.mu.Unlock()

			// if rf.getLastCommitedIdx() > 100 {
			// 	DPrintf("rf[%v] applied msg, commit=%v, rf.nextIdx = %v\n", rf.me, rf.getLastCommitedIdx(), rf.nextIndex)
			// }
		case <-time.After(1000 * time.Millisecond):
			// time out just in case dead in here
			continue
		}
	}
